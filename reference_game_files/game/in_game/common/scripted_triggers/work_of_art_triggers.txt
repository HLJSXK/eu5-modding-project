work_of_art_exists = {
	exists = $work_of_art$
	$work_of_art$ = { is_art_destroyed = no }
}

imperial_work_of_art_can_be_moved = {
	exists = $work_of_art$
	$work_of_art$ = {
		is_art_destroyed = no
		owner ?= {
			this != $country$
			is_member_of_international_organization = $international_organization$
		}
	}
}
