	if = {
		limit = {
			OR = {
				current_year < 1345
				has_cooldown = ai_spam_protection
			}
		}
		add = {
			desc = "INERTIA"
			value = -10000
		}
	}
	else = {
		value = 0
	}
}

assassinate_character_cost_multiplier = {
	value = total_abilities
	divide = 300
	
	if = {
		limit = {
			has_estate = estate_type:peasants_estate
		}
		multiply = 0.5
	}
	if = {
		limit = { is_ruler = yes }
		multiply = 2.0
	}
	if = {
		limit = { is_heir = yes }
		multiply = 1.5
	}
	if = {
		limit = { is_regent = yes }
		multiply = 1.5
	}
	if = {
		limit = { is_general = yes }
		multiply = 2.0
	}
	if = {
		limit = { is_admiral = yes }
		multiply = 2.0
	}
	if = {
		limit = { 
			exists = owner.ruler
			is_close_relative = owner.ruler 
		}
		multiply = 2.0
	}
}

	}

hegemon_demand_base_will_do = {
	if = {
		limit = { scope:actor = { has_casus_belli_of_type_on = { type = casus_belli:cb_hegemon_ultimatum target = scope:recipient } } }
		subtract = {
			desc = "DIPLOREASON_WE_HAVE_CASUS_BELLI"
			value = 1000
		}
	}
	add = {
		desc = "DIPLOREASON_THEY_WOULD_ACCEPT"
		value = "scope:actor.offer_relation_acceptance(scope:scripted_relation_type|scope:recipient)"
		add = {
			value = prestige_extreme_penalty
			multiply = 2
		}
	}
	if = {
		limit = { scope:actor = { wants_casus_belli_with = scope:recipient } }
		add = {
			desc = "DIPLOREASON_WE_WANT_WAR"
			value = 100
		}
	}
}

alliance_upkeep_cost = {
	add = {
		desc = "UPKEEP_BASE_VALUE"
		value = 0.1
	}
	add = {
		desc = "RELATIVE_POPULATION"
		value = scope:second.total_population
		divide = {
			value = scope:first.total_population
			min = 1 
		}
		multiply = 0.5
		min = 0.2
	}
	if = {
		limit = {
			scope:first ?= { is_a_threat_for_us = scope:second }
		}
		multiply = {
			desc = "AUC_THREAT"
			value = 2.0
		}
	}
	divide = {
		scope:first = {
			scope:second = {
				value = this_opinion_of_prev
			}
		}
		divide = 200
		min = 1
		desc = "UPKEEP_TARGET_OPINION"
	}

	max = {
		value = scope:second.country_rank_level
		desc = "UPKEEP_RANK"
	}

}
