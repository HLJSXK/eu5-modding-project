give_subject_location_to_other_subject = { 
	type = subject
	category = CATEGORY_SUBJECT_ACTIONS

	ai_tick = never
	
	use_enroute = no

	potential = {
	}

	allow = {
	}

	select_trigger = {
		looking_for_a = country
		interaction_source_list = {
			scope:actor = {
				every_subject = {
					add_to_list = source
				}
			}
		}
		target_flag = recipient
		name = "choose_country"
		column = {
			data = name
		}
		visible = {
			NOR = {
				country_type = building
				country_type = pop
			}
		}
        enabled = {
			num_locations > 1
			at_war = no
		}
	}

	select_trigger = {
		looking_for_a = country
		interaction_source_list = {
			scope:actor = {
				every_subject = {
					add_to_list = source
				}
			}
		}
		target_flag = country_that_gets_the_location
		name = "select_a_subject_to_receive_a_location"
		column = {
			data = name
		}
		visible = {
            is_neighbor_of = scope:recipient
			NOR = {
				country_type = building
				country_type = pop
			}
		}
        enabled = {
			at_war = no
			trigger_if = {
				limit = {
					is_subject_type = tusi
				}
				num_locations < 15
			}
		}
	}

	select_trigger = {
		looking_for_a = location
		source = recipient
		target_flag = target_location
		source_flags = only_actual_locations
		name = "select_a_location_to_give_to_other_subject"
		none_available_msg_key = "give_location_to_subject_no_locations"
		column = {
			data = name
		}
		column = {
			data = population
		}
		visible = {
			is_capital = no
			OR = {
				any_coast_border_location = {
					owner ?= scope:country_that_gets_the_location
				}
				area = scope:country_that_gets_the_location.capital.area
			}
		}
	}

	effect = {
        ##do this before you give the location away or you'll get the wrong number
		if = {
			limit = {
                exists = scope:recipient
			}
			scope:recipient = {	
				add_opinion = {
                    target = scope:actor
                    modifier = opinion_gave_away_our_lands
                    scale = {
                        value = 1
                        divide = num_locations
                    }
                }
			}
		}
        #now give the location away
		if = {
			limit = {
                exists = scope:country_that_gets_the_location
				exists = scope:target_location
			}
			scope:target_location = {	
				change_location_owner = scope:country_that_gets_the_location
			}
		}
		else = {
			custom_tooltip = select_a_location_to_give_to_other_subject
		}
	}
}
