namespace = council_of_trent

council_of_trent.2 = { #on end event
	type = country_event
	category = situation_event
	title = council_of_trent.2.title
	desc = council_of_trent.2.desc
	
	illustration_tags = {
		10 = regular
		10 = interior
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:peasants_estate background = estate_type:peasants_estate }
		international_organization:catholic_church = {
			every_current_policy_in_international_organization = {
				if = {
					limit = { has_tag = anti_reformation }
					root = { change_variable = { name = council_of_trent_pro_reformation_policies add = 1 } }
					if = {
						limit = { has_tag = pro_reformation_severe }
						root = { change_variable = { name = council_of_trent_pro_reformation_policies add = 1 } }
					}
				}
				else_if = {
					limit = { has_tag = pro_reform }
					root = { change_variable = { name = council_of_trent_anti_reformation_policies add = 1 } }
					if = {
						limit = { has_tag = anti_reformation_severe }
						root = { change_variable = { name = council_of_trent_anti_reformation_policies add = 1 } }
					}
				}
				if = {
					limit = { has_tag = conciliatory_policy }
					root = { change_variable = { name = council_of_trent_pro_reformation_policies add = 3 } }
				}
				else_if = {
					limit = { has_tag = harsh_policy }
					root = { change_variable = { name = council_of_trent_anti_reformation_policies add = 3 } }
				}
			}
		}
		set_variable = {
			name = council_of_trent_value
			value = {
				value = var:council_of_trent_pro_reformation_policies
				subtract = var:council_of_trent_anti_reformation_policies
				subtract = {
					value = international_organization:catholic_church.var:council_of_trent_pro_reformation_policies
					subtract = international_organization:catholic_church.var:council_of_trent_anti_reformation_policies
				}
			}
		}
	}

	after = {
		hidden_effect = {
			remove_variable = council_of_trent_value
			remove_variable = council_of_trent_pro_reformation_policies
			remove_variable = council_of_trent_anti_reformation_policies
			international_organization:catholic_church = {
				remove_variable = council_of_trent_reforms_done_counter
				remove_variable = council_of_trent_pro_reformation_policies
				remove_variable = council_of_trent_anti_reformation_policies
			}
		}
	}
	
	option = { #Neutral "Historical" Outcome - not full hatred but not tolerant
		name = council_of_trent.2.a
		trigger = {
			AND = {
				var:council_of_trent_value <= 2
				var:council_of_trent_value >= -2
			}
		}
		religion:catholic = {
			change_protestant_view = { value = negative }
		}
	}

	option = { #Hostile Aggressive Outcome 
		name = council_of_trent.2.b
		trigger = {
			var:council_of_trent_value < -2
		}
		religion:catholic = {
			change_protestant_view = { value = enemy }
		}
	}

	option = { #Friendly Reconiciliation Outcome
		name = council_of_trent.2.c
		trigger = {
			var:council_of_trent_value > 2
		}
		religion:catholic = {
			change_protestant_view = { value = kindred }
		}
	}
}
