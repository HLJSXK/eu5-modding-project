namespace = eu5mp_sample

# Sample Event: Project Welcome
# This event demonstrates the basic structure of an EU5 event
eu5mp_sample.1 = {
	type = country_event
	title = eu5mp_sample.1.title
	desc = eu5mp_sample.1.desc
	fire_only_once = yes
	
	# This event is triggered manually, not by MTTH
	# In EU5, events must be explicitly triggered via on_actions or other scripts
	
	illustration_tags = {
		10 = happy
		10 = exterior
	}
	
	# Trigger conditions - when can this event fire?
	trigger = {
		is_ai = no  # Only for human players
		has_game_started = yes
	}
	
	# Option 1: Acknowledge the event
	option = {
		name = eu5mp_sample.1.option_a
		
		# Effects: what happens when this option is chosen
		add_treasury = 100
		add_prestige = 10
		
		# Custom tooltip for clarity
		custom_tooltip = eu5mp_sample.1.option_a.tooltip
	}
	
	# Option 2: Alternative choice
	option = {
		name = eu5mp_sample.1.option_b
		
		# Different effects for this option
		add_manpower = 1000
		add_legitimacy = 5
		
		custom_tooltip = eu5mp_sample.1.option_b.tooltip
	}
}

# Sample Event: Conditional Event
# This demonstrates more complex trigger logic
eu5mp_sample.2 = {
	type = country_event
	title = eu5mp_sample.2.title
	desc = eu5mp_sample.2.desc
	fire_only_once = no
	
	illustration_tags = {
		10 = neutral
		10 = interior
	}
	
	# More complex trigger conditions
	trigger = {
		is_ai = no
		has_game_started = yes
		
		# Check treasury level
		gold >= 1000
		
		# Check if at peace
		is_at_war = no
		
		# Check technology level
		technology_level >= 5
	}
	
	# Single option event
	option = {
		name = eu5mp_sample.2.option_a
		
		# Conditional effects based on current state
		if = {
			limit = { gold >= 2000 }
			add_prestige = 20
		}
		else = {
			add_prestige = 10
		}
	}
}
