		}

			type = estate_type:nobles_estate

		}

		name = polish_mission_events.6.b
		custom_tooltip = polish_mission_events.6.b.tt
		
		# Get CB on Teutonic Order capital
		add_casus_belli = {
			target ?= c:TEU
			type = casus_belli:cb_conquer_province
			years = 30
		}
	}
}
