# ==================== EVENT 1: THE WALLS OF CONSTANTINOPLE ====================
byz_mission_events.1 = {
	type = country_event
	category = mission_event
	title = byz_mission_events.1.t
	desc = byz_mission_events.1.d
	fire_only_once = yes
	
	illustration_tags = {
		10 = defensive
		10 = exterior
	}

	immediate = {
		event_illustration_poptype_effect = {
			foreground = pop_type:soldiers
			background = pop_type:laborers
		}
		capital = {
			save_scope_as = constantinople_location
		}
	}

	option = {
		name = byz_mission_events.1.a
		custom_tooltip = byz_mission_events.1.a.tt
		
		add_stability = 20
		
		ruler_or_regent = {
			add_character_modifier = {
				modifier = secure_throne
				years = 40
				mode = add
			}
		}
	}

	option = {
		name = byz_mission_events.1.b
		custom_tooltip = byz_mission_events.1.b.tt

		location:constantinople = {
			owner ?= root
			add_location_modifier = {
				modifier = practical_fortifications
				years = 35
				mode = add
			}
		}
		
		add_manpower = {
			value = max_manpower
			multiply = 0.2
		}
	}

	option = {
		name = byz_mission_events.1.c
		custom_tooltip = byz_mission_events.1.c.tt
		
		add_navy_tradition = navy_tradition_mild_bonus
		
		location:constantinople = {
			owner ?= root
			add_location_modifier = {
				modifier = reinforced_sea_walls
				years = 35
				mode = add
			}
		}
		
		change_societal_value = {
			type = land_vs_naval
			value = societal_value_move_to_right
		}
	}
}

# ==================== EVENT 2: THE HEXAMILION WALL ====================
byz_mission_events.2 = {
	type = country_event
	category = mission_event
	title = byz_mission_events.2.t
	desc = byz_mission_events.2.d
	fire_only_once = yes
	
	illustration_tags = {
		10 = construction
		10 = exterior
	}

	immediate = {
		event_illustration_estate_effect = {
			foreground = estate_type:peasants_estate
			background = estate_type:nobles_estate
		}
		
		location:corinth = {
			save_scope_as = corinth_location
		}
	}

	option = {
		name = byz_mission_events.2.a
		custom_tooltip = byz_mission_events.2.a.tt

		add_country_modifier = {
			modifier = fortified_isthmus
			years = 20
			mode = add
		}
				
		add_gold = 500
		add_prestige = prestige_mild_bonus
	}

	option = {
		name = byz_mission_events.2.b
		custom_tooltip = byz_mission_events.2.b.tt
				
		area:morea_area = {
			ordered_location_in_area = {
				limit = {
					owner ?= root
				}
				order_by = development
				max = 3
				if = {
					limit = {
						root = { has_advance = bastion_advance }
						NOT = { has_building_with_at_least_one_level = bastion }
					}
					construct_building = {
						building_type = building_type:bastion
						cost_multiplier = 0.3
						instant = yes
					}
				}
				else_if = {
					limit = {
						root = { has_advance = castle_advance }
						NOT = { has_building_with_at_least_one_level = castle }
					}
					construct_building = {
						building_type = building_type:castle
						cost_multiplier = 0.2
						instant = yes
					}
				}
			}
		}
		
		add_manpower = {
			value = max_manpower
			multiply = 0.1
		}
	}

	option = {
		name = byz_mission_events.2.c
		custom_tooltip = byz_mission_events.2.c.tt
		
		add_country_modifier = {
			modifier = peloponnese_watchtowers
			years = 25
			mode = add
		}
		
		area:morea_area = {
			ordered_location_in_area = {
				limit = {
					owner ?= root
						OR = {
							topography = mountains
							topography = hills
						}
				}
				order_by = development
				max = 5
				add_location_modifier = {
					modifier = watchtower_network
					years = 25
					mode = add
				}
			}
		}
	}
}

# ==================== EVENT 3: THE NICENE RESURRECTION ====================
byz_mission_events.3 = {
	type = country_event
	category = mission_event
	title = byz_mission_events.3.t
	desc = byz_mission_events.3.d
	fire_only_once = yes
	
	illustration_tags = {
		10 = religious
		10 = interior
	}

	immediate = {
		event_illustration_estate_effect = {
			foreground = estate_type:clergy_estate
			background = estate_type:nobles_estate
		}
		if = {
			limit = { country_exists = c:TUR }
			c:TUR = {
				save_scope_as = turkish_threat
			}
		}
	}

	option = {
		name = byz_mission_events.3.a
		custom_tooltip = byz_mission_events.3.a.tt
		
		# Religious Restoration
		location:iznik = {
			if = {
				limit = { NOT = { has_building_with_at_least_one_level = cathedral } }
				construct_building = {
					building_type = building_type:cathedral
					cost_multiplier = 0.4
					instant = yes
				}
			}
			add_location_modifier = {
				modifier = nicaea_patriarchate_restored
				years = 20
				mode = add
			}
			change_development = 3
		}
	}

	option = {
		name = byz_mission_events.3.b
		custom_tooltip = byz_mission_events.3.b.tt
		
		# Strategic Bridgehead
		location:iznik = {
			if = {
				limit = { root = { has_advance = castle_advance } }
				construct_building = {
					building_type = building_type:castle
					cost_multiplier = 0.3
					instant = yes
				}
			}
			add_location_modifier = {
				modifier = nicaea_fortified_bridgehead
				years = 25
				mode = add
			}
		}
		if = {
			limit = { exists = scope:turkish_threat }
			add_casus_belli = {
				target = scope:turkish_threat
				type = casus_belli:cb_conquer_province
				months = 180
			}
		}
	}

	option = {
		name = byz_mission_events.3.c
		custom_tooltip = byz_mission_events.3.c.tt
		
		# Cultural Renaissance
		location:iznik = {
			if = {
				limit = { NOT = { has_building_with_at_least_one_level = library } }
				construct_building = {
					building_type = building_type:library
					cost_multiplier = 0.3
					instant = yes
				}
			}
			add_location_modifier = {
				modifier = greek_resettlement_program
				years = 10
				mode = add
			}
			every_pop = {
				limit = { culture != culture:greek_culture }
				split_pop = {
					fraction = 0.3
					culture = culture:greek_culture
				}
			}
		}
	}
}

				limit = { owner ?= root NOT = { is_core_of = root }}

			}

			dip = { 80 90 }

			add_trait = trait:conqueror
			age = 45
			save_scope_as = anatolian_conqueror
			create_in_limbo = yes
		}
		if = {
			limit = { exists = scope:anatolian_conqueror }
			scope:anatolian_conqueror = {
				move_country = root

		}
