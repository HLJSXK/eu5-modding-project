# ==================== EVENT 2: DEVELOP LONDON ====================
england_mission_events.2 = {
	type = country_event
	category = mission_event
	title = england_mission_events.2.t
	desc = england_mission_events.2.d
	
	fire_only_once = yes
	
	illustration_tags = {
		10 = regular
		10 = interior
	}

	immediate = {
		event_illustration_estate_effect = { 
			foreground = estate_type:burghers_estate 
			background = estate_type:burghers_estate 
		}
		
		capital = {
			save_scope_as = london_location
			market = {
				save_scope_as = london_market
			}
		}
	}

	option = {
		name = england_mission_events.2.a
		
		custom_tooltip = england_mission_events.2.a.tt
		
		if = {
			limit = { exists = scope:london_location }
			scope:london_location = {
				add_location_modifier = {
					modifier = london_commercial_hub
					years = 25
					mode = add
				}
				change_development = development_mild_bonus
			}
		}
		
		if = {
			limit = { exists = scope:london_market }
			scope:london_market = {
				add_merchant_power = {
					country = root
					key = london_trade_expansion
					years = 10
					power = merchant_power_mild_bonus
				}
			}
		}
	}

	option = {
		name = england_mission_events.2.b
		custom_tooltip = england_mission_events.2.b.tt
		
		if = {
			limit = { exists = scope:london_location }
			scope:london_location = {
				add_location_modifier = {
					modifier = london_cultural_center
					years = 25
					mode = add
				}
				change_development = development_weak_bonus
			}
		}
		
		culture = {
			add_cultural_tradition = cultural_tradition_severe_bonus
		}
		
		add_gold = {
			value = {
				add = monthly_income_trade_and_tax
				multiply = 4
			}
		}
		add_prestige = prestige_mild_bonus
	}
}

	option = {

		if = {
			limit = {
					region:deccan_region = {
							any_location_in_region = {
									raw_material ?= goods:pepper
									has_owner = yes
							}
					}
			}
			region:deccan_region = {
					random_location_in_region = {
							limit = {
									raw_material ?= goods:pepper

									}
							}
							save_scope_as = target_location
							owner ?= { save_scope_as = target_country }
					}
			}
		}
		
		# Grant CB using saved scopes
		if = {
			limit = {
					exists = scope:target_location
					exists = scope:target_country
			}
			add_casus_belli = {
					target = scope:target_country
					type = casus_belli:cb_conquer_province
					province = scope:target_location.province
					months = 120
			}
