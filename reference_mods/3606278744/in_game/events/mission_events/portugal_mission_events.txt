	}

		event_illustration_estate_effect = {

	}

	option = {
		name = portugal_mission_events.3.a
		add_country_modifier = {
			modifier = atlantic_settlement
			years = 10
			mode = add
		}
		area:azores_area = {
			every_location_in_area = {
				limit = {
					owner ?= root
				}
				#it works but does not show ingame before we control it
				change_control = control_mild_bonus
				change_prosperity = prosperity_very_weak_bonus
				change_development = development_weak_bonus
			}
		}
	}

	option = {
		name = portugal_mission_events.3.b
		add_country_modifier = {
			modifier = naval_supply_bases
			years = 10
			mode = add
		}
		add_navy_tradition = 20
		ordered_owned_location = {
			limit = { is_coastal = yes }
			order_by = development
			max = 2
			add_location_modifier = {
				modifier = naval_supply_location
				years = 5
				mode = add
			}
		}
	}
}

portugal_mission_events.13 = {
	type = country_event
	category = mission_event
	title = portugal_mission_events.13.t
	desc = portugal_mission_events.13.d
	
	# Add illustration tags like the official example
	illustration_tags = {
		10 = regular
		10 = exterior
	}

	immediate = {
		event_illustration_poptype_effect = { 
			foreground = pop_type:soldiers 
			background = pop_type:soldiers 
		}

		# Save the target location for the CB
		location:taunat = {
			save_scope_as = target_location
			owner ?= {
				save_scope_as = target_country
			}
		}
	}

	option = {
		name = portugal_mission_events.13.a
		
		# Add CB using the saved scope
		if = {
			limit = { 
				scope:target_country = { exists = yes } 
			}
			add_casus_belli = {
				target = scope:target_country
				type = casus_belli:cb_conquer_province
				months = 120
			}
		}
		
		# Fix estate satisfaction (use proper values)
		add_estate_satisfaction = {
			type = estate_type:burghers_estate
			value = -0.15
		}
		
		# Fix stability (should be between -3 and +3 typically)
		add_stability = -25
		add_gold = -150
	}

	option = {
		name = portugal_mission_events.13.b
		add_stability = 25
	}
}
