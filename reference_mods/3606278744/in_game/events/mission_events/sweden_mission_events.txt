# Event 3: Challenging Danish Hegemony
sweden_mission_events.3 = {
	type = country_event
	category = mission_event
	title = sweden_mission_events.3.t
	desc = sweden_mission_events.3.d
	
	illustration_tags = {
		10 = diplomacy
		10 = interior
	}

	immediate = {
		event_illustration_estate_effect = { 
			foreground = estate_type:nobles_estate 
			background = estate_type:nobles_estate 
		}
		
		# Save Denmark scope for potential CB
		if = {
			limit = { country_exists = c:DAN }
			c:DAN = {
				save_scope_as = danish_rival
			}
		}
	}

	option = {
		name = sweden_mission_events.3.a
		if = {
			limit = { exists = scope:danish_rival }
			add_casus_belli = {
				target = scope:danish_rival
				type = casus_belli:cb_conquer_province
				months = 60
			}
		}
		add_country_modifier = {
			modifier = scandinavian_ambition
			years = 8
			mode = add
		}
		add_prestige = 20
	}

	option = {
		name = sweden_mission_events.3.b
		add_country_modifier = {
			modifier = diplomatic_isolation
			years = 15
			mode = add
		}
		add_gold = 150
	}
}
