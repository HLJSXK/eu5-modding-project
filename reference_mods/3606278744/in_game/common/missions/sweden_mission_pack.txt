	mission_counter_russian_expansion = {
		icon = diplomatic_influence
		requires = { mission_secure_finland }
		
		enabled = {
    OR = {
			# Case 1: Russia exists and is rival, check strength against Russia
			AND = {
				custom_tooltip = {
						text = check_russia_exists_and_strength
						country_exists = c:RUS
						c:RUS = {
								is_rival_of = root
						}
				}
				relative_strength = {
						target = c:RUS
						value >= 0.75
				}
		}
		# Case 2: Russia doesn't exist, check combined strength against MOS and NOV
		AND = {
				custom_tooltip = {
						text = check_muscovy_novgorod_strength
						NOT = { country_exists = c:RUS }
						country_exists = c:MOS
						country_exists = c:NOV
				}
				relative_strength = {
						target = c:MOS
						value >= 0.6
				}
				relative_strength = {
						target = c:NOV
						value >= 0.6
				}
		}
		# Case 3: Russia doesn't exist, only MOS exists
		AND = {
				custom_tooltip = {
						text = check_muscovy_only_strength
						NOT = { country_exists = c:RUS }
						country_exists = c:MOS
						NOT = { country_exists = c:NOV }
				}
				relative_strength = {
						target = c:MOS
						value >= 0.95
				}
		}
		# Case 4: Russia doesn't exist, only NOV exists
		AND = {
				custom_tooltip = {
						text = check_novgorod_only_strength
						NOT = { country_exists = c:RUS }
						country_exists = c:NOV
						NOT = { country_exists = c:MOS }
				}
				relative_strength = {
						target = c:NOV
						value >= 0.95
				}
			}
    }
	}
		
		duration = 0

		on_completion = {
			add_country_modifier = {
				modifier = russian_counterweight
				years = 30
				mode = add
			}
		}
		ai_will_do = { value = 100 }
