wales_independence_mission_pack = {
	icon = generic_capital_economy
	chance = 3600
	repeatable = no
	player_playstyle = diplomatic

	#ai_will_do = {}

	visible = {
		game_has_missions_enabled = yes
		tag = "WLS"
	}

	on_start = {
		set_variable = {
			name = mission_target_development
			value = 15
		}
	}

	on_completion = {
		add_prestige = 25
		remove_variable = mission_target_development
	}

	on_abort = {
		remove_variable = mission_target_development
	}

	# === WELSH UNIFICATION PATH ===
	
	mission_develop_welsh_heartland = {
		icon = cultural_acceptance_advance
		requires = { }
		
		enabled = {
			capital = {
				development >= 12
			}
			monthly_income_trade_and_tax >= 8
		}
		
		duration = 0

		on_completion = {
			add_gold = 75
			capital = {
				change_development = development_very_weak_bonus
			}
			trigger_event_non_silently = wales_independence_events.1
		}
		#ai_will_do = {}
	}

	mission_unite_welsh_principalities = {
		icon = diplomatic_influence
		requires = { mission_develop_welsh_heartland }
		
		enabled = {
			any_owned_location = {
			 #it shows a weird tooltipingame when i use region:wales_region here but works fine otherwise
				# region = region:wales_region
				count >= 16
			}
			total_development >= 160	
		}
		
		duration = 0

		on_completion = {
			add_country_modifier = {
				modifier = welsh_unification
				years = 5
			}
			trigger_event_silently = wales_independence_events.2
		}
		#ai_will_do = {}
	}

	mission_centralize_welsh_rule = {
		icon = court_accounting
		requires = { mission_unite_welsh_principalities }
		
		enabled = {
			"estate_power(estate_type:crown_estate)" >= 0.35	# 35% crown power
			average_control_in_home_region >= 0.8
		}
		
		duration = 0

		on_completion = {
			add_country_modifier = {
				modifier = centralized_welsh_rule
				years = 10
			}
			add_prestige = 25
			# global_monthly_control = 0.03
		}
		#ai_will_do = {}
	}

	# === DEFENSIVE PREPARATION PATH ===
	
	mission_fortify_welsh_borders = {
		icon = defensive_army
		requires = { mission_centralize_welsh_rule }
		
		enabled = {
			any_owned_location = {
				# region = region:wales_region
				has_building = building_type:castle
				count >= 5
			}
		}
		
		duration = 0

		on_completion = {
			add_country_modifier = {
				modifier = welsh_mountain_defenses
				years = -1
			}
		}
		#ai_will_do = {}
	}

	mission_raise_welsh_levies = {
		icon = medieval_military
		requires = { mission_fortify_welsh_borders }
		
		enabled = {
			# total levies does not exist
			# total_levies >= 12
			army_tradition >= 25
		}
		
		duration = 0

		on_completion = {
			add_country_modifier = {
				modifier = welsh_longbow_traditions
				years = 10
			}
		}
		#ai_will_do = {}
	}

	# === DIPLOMATIC PREPARATION PATH ===
	
	mission_strengthen_welsh_identity = {
		icon = bill_of_rights
		requires = { mission_centralize_welsh_rule }
		
		enabled = {
			# culture = culture:welsh
			#test dominant culture first then only culture
			dominant_culture = culture:welsh
			# primary_culture = culture:welsh
			# primary culture = culture:welsh check does not work but the modifer exists so i am not sure why:
			# 			primary_culture = {
			# 	alias = { primary_culture_short primary_cultures }
			# 	family = culture
			# 	texture = "gfx/interface/icons/culture/primary_culture.dds"
			# }
			stability >= 60
		}
		
		duration = 0

		on_completion = {
			culture = { add_cultural_tradition = cultural_tradition_mild_bonus }
			add_country_modifier = {
				modifier = welsh_cultural_revival
				years = 15
			}
		}
		#ai_will_do = {}
	}

	# === INDEPENDENCE STRUGGLE ===
	
	mission_declare_welsh_independence = {
		icon = glorious_arms
		requires = { mission_raise_welsh_levies mission_strengthen_welsh_identity }
		
		enabled = {
			NOT = { is_subject_of = c:ENG }
			is_at_war_with = c:ENG
		}
		
		duration = 0

		on_completion = {
			add_war_exhaustion = -15
			add_country_modifier = {
				modifier = welsh_independence_fervor
				years = 5
			}
			trigger_event_silently = wales_independence_events.10
		}
		#ai_will_do = {}
	}

	mission_secure_welsh_borders = {
		icon = defensive_army
		requires = { mission_declare_welsh_independence }
		
		enabled = {
			NOT = { is_at_war_with = c:ENG }
			any_owned_location = {
				# region = region:wales_region
				count >= 30
			}
			monthly_income_trade_and_tax >= 85
		}
		
		duration = 0

		on_completion = {
			add_prestige = 30
			add_legitimacy = 50
			add_country_modifier = {
				modifier = secure_welsh_borders
				years = 10
			}
		}
		#ai_will_do = {}
	}

	mission_establish_welsh_kingdom = {
		icon = bill_of_rights
		requires = { mission_secure_welsh_borders }
		final = yes
		
		enabled = {
			# NOT = { is_subject_of = c:ENG }
			NOT = { is_subject_type = vassal }
			country_rank = country_rank:rank_kingdom
		}
		
		duration = 0

		on_completion = {
			add_country_modifier = {
				modifier = kingdom_of_wales
				years = -1
			}
			add_prestige = 75
			trigger_event_non_silently = wales_independence_events.11
			custom_tooltip = welsh_kingdom_established_tt
		}
		#ai_will_do = {}
	}
}
