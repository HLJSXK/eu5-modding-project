namespace = muzaffarids_injuids_flavor

muzaffarids_injuids_flavor.1 = {
    type = country_event
    title = muzaffarids_injuids_flavor.1.title
    desc = muzaffarids_injuids_flavor.1.desc
    fire_only_once = yes

    dynamic_historical_event = {
		tag = MZF
		from = 1338.1.1
		to = 1340.1.1
		monthly_chance = 75
	}

    immediate = {
        root.ruler_or_regent = {
            add_character_modifier = {
                modifier = muzaffarid_rise_to_power
                days = -1
                mode = add_and_extend
            }
        }
    }

    trigger = { country_exists = c:KRM }

    option = {
        name = muzaffarids_injuids_flavor.1.a
        historical_option = yes

        root.capital = {
            while = {
                count = 10
                create_sub_unit_with_owner = {
                    type = a_feudal_levy
                    owner = root
                }
            }
        }

        declare_war_with_cb = {
            target = c:KRM
            type = casus_belli:cb_annex
        }
    }

    option = {
        name = muzaffarids_injuids_flavor.1.b

    }
    
}

muzaffarids_injuids_flavor.4 = {
    type = country_event
    title = muzaffarids_injuids_flavor.4.title
    desc = muzaffarids_injuids_flavor.4.desc
    historical_info = {
        first_valid = {
            triggered_desc = {
                desc = muzaffarids_injuids_flavor.4.historical_info
                trigger = {
                    tag = MZF
                }
            }
        }
    }
    fire_only_once = yes

    option = {
        name = muzaffarids_injuids_flavor.4.a

        if = {
            limit = {
                tag = MZF
            }
            set_capital = location:shiraz
            add_prestige = prestige_weak_bonus
        }
        else = {
            add_prestige = prestige_weak_bonus
        }

        every_owned_location = {
            add_core = root
        }

        region:persia_region = {
            every_country_with_capital_in_geography = {
                limit = {
                    NOT = { this = root }
                    OR = {
                        capital.area = area:fars_area
                        capital.area = area:kerman_area
                        capital.area = area:khuzestan_area
                    }         
                }
                root = {
                    add_casus_belli = {
                        target = prev
                        type = casus_belli:cb_annex
                        years = -1
                    }
                }
            }
        }

        set_country_rank = country_rank:rank_duchy
    }
}

    dynamic_historical_event = {
        tag = MZF
		from = 1360.1.1
		to = 1380.1.1
		monthly_chance = 70
	}

    trigger = {
        NOT = {
            is_subject_of = c:TIM
        }
        OR = {
            is_neighbor_of = c:TIM
            any_subject_or_below = {
                is_neighbor_of = c:TIM
            }
            any_neighbor_country = {
                is_subject_of = c:TIM
            }
        }
    }

    option = {
        name = muzaffarids_injuids_flavor.5.a

        save_scope_as = fars_ruler

        c:TIM = {
            trigger_event_non_silently = { id = muzaffarids_injuids_flavor.6 }
        }
    }

    option = {
        name = muzaffarids_injuids_flavor.5.b

        add_prestige = prestige_weak_bonus
    }
}
