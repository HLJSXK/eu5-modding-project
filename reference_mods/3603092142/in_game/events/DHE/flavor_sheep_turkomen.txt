scripted_effect setup_country = {
    if = {
        limit = { tag = QAR }
        create_qara_qoyunlu_family = yes
    }
    else_if = {
        limit = { tag = AKK }
        create_aq_qoyunlu_family = yes
    }

    #generic setup
    add_gold = 3000
    add_estate_satisfaction = { type = estate_type:nobles_estate value = estate_satisfaction_radical_bonus }
    add_estate_satisfaction = { type = estate_type:clergy_estate value = estate_satisfaction_radical_bonus }
    add_estate_satisfaction = { type = estate_type:peasants_estate value = estate_satisfaction_radical_bonus }
    add_estate_satisfaction = { type = estate_type:burghers_estate value = estate_satisfaction_radical_bonus }
    add_estate_satisfaction = { type = estate_type:dhimmi_estate value = estate_satisfaction_radical_bonus }
    add_estate_satisfaction = { type = estate_type:tribes_estate value = estate_satisfaction_radical_bonus }
    change_government_type = government_type:steppe_horde
    setup_societal_values = {
        centralization_vs_decentralization = -60
        traditionalist_vs_innovative = 40
        spiritualist_vs_humanist = -50
        aristocracy_vs_plutocracy = -60
        serfdom_vs_free_subjects = 50
        mercantilism_vs_free_trade = 50
        belligerent_vs_conciliatory = -80
        quality_vs_quantity = -50
        offensive_vs_defensive = -90
        land_vs_naval = -80
        capital_economy_vs_traditional_economy = 100
        individualism_vs_communalism = 60
        outward_vs_inward = -20
    }
    change_heir_selection = heir_selection:tribal_oldest_male
    every_owned_location = {
        change_control = control_radical_bonus
    }
    set_new_ruler = scope:target_character
    set_country_rank = country_rank:rank_kingdom
    change_country_type = army
    add_horde_unity = horde_unity_extreme_bonus
    add_country_modifier = {
        modifier = turkomen_rise_to_power
        years = 40
        mode = add_and_extend
    }
}

flavor_sheep_turkomen.1 = {
    type = country_event
    title = flavor_sheep_turkomen.1.title
    desc = flavor_sheep_turkomen.1.desc

    trigger = {
        has_game_rule = historical_turkomen
    }

    immediate = {
        every_neighbor_country = {
            limit = { this != c:TIM }
            add_truce_with = {
                target = root
                years = 20
            }
        }
        root.capital = {
            while = {
                count = 20
                create_sub_unit_with_owner = {
                    type = a_steppe_horse_archers
                    owner = root
                }
            }
            create_dynasty_from_location = qara_qoyunlu_dynasty
        }
        setup_country = yes

        every_known_country = {
            limit = {
                OR = {
                    capital.sub_continent = sub_continent:middle_east
                    capital.region = region:caucasus_region
                }
            }
            trigger_event_silently = flavor_sheep_turkomen.2
        }
    }

    option = {
        name = flavor_sheep_turkomen.1.a
    }
}

flavor_sheep_turkomen.3 = {
    type = country_event
    title = flavor_sheep_turkomen.3.title
    desc = flavor_sheep_turkomen.3.desc
    fire_only_once = yes

    dynamic_historical_event = {
		tag = QAR
		from = 1415.1.1
		to = 1460.1.1
		monthly_chance = 2
	}

    trigger = {
        has_game_rule = historical_turkomen
        country_exists = c:TIM
        at_war = no
        OR = {
            is_neighbor_of = c:TIM
            any_neighbor_country = {
                is_subject_of = c:TIM
            }
        }
    }

    option = {
        name = flavor_sheep_turkomen.3.a
        historical_option = yes
        root.capital = {
            while = {
                count = 40
                create_sub_unit_with_owner = {
                    type = a_steppe_horse_archers
                    owner = root
                }
            }
        }
        declare_war_with_cb = {
            target = c:TIM
            type = casus_belli:cb_tribal_feud
        }
    }
    option = {
        name = flavor_sheep_turkomen.3.b
    }
}

flavor_sheep_turkomen.100 = {
    type = country_event
    title = flavor_sheep_turkomen.100.title
    fire_only_once = yes
    hidden = yes

    trigger = {
        has_game_rule = historical_turkomen
        current_year > 1374
        c:TIM = { num_locations < 200 }
        NOT = { country_exists = c:QAR }
    }

    immediate = {
        every_location_in_the_world = {
            limit = {
                owner ?= {
                    is_ai = yes
                }
                OR = {
                    this.province_definition = province_definition:garmian_province
                    this.province_definition = province_definition:nusaybin_province
                    this = location:duhok
                    this = location:mosul
                    this = location:balad
                    AND = {
                        this.area = area:armenian_highlands_area
                        NOT = { 
                            OR = {
                                this.province_definition = province_definition:harput_province 
                                this.province_definition = province_definition:erzurum_province
                            }
                        }
                    }
                    this.area = area:armenia_area
                    this.area = area:azerbaijan_area
                }
            }
            add_core = c:QAR
            change_location_owner = c:QAR
        }

        c:QAR = { trigger_event_non_silently = flavor_sheep_turkomen.1 }
    }
}
