namespace = wrath_of_timur

wrath_of_timur.1 = {
    type = country_event
    title = wrath_of_timur.1.title
    hidden = yes
    fire_only_once = yes

    dynamic_historical_event = {
		tag = TIM
		from = 1360.1.1
		to = 1365.1.1
		monthly_chance = 100
	}

    trigger = {
        character:chg_timur_lame_borjigin ?= {
			is_alive = yes
			owner = root
        }
		owns = location:samarkand
    }

    immediate = {
        location:samarkand = {
            while = {
                count = 100
                create_sub_unit_with_owner = {
                    type = a_steppe_horse_archers
                    owner = root
                }
            }
        }

        if = {
            limit = {
                is_ai = yes
                has_game_rule = historical_timur_conquests
            }
            add_country_modifier = {
                modifier = ai_timurids_limiter
                years = 50
                mode = add_and_extend
            }
        }

        add_country_modifier = {
            modifier = ai_force_annexation
            years = -1
            mode = add_and_extend
        }

        if = {
            limit = { country_exists = c:CHG }

            c:CHG = {
                every_owned_location = {
                    limit = {
                        OR = {
                            this.area = area:transoxiana_area
                            this.area = area:badakhshan_area
                        }
                    }
                    change_location_owner = root
                }
            }
        }

        if = {
            limit = { 
                country_exists = c:CHG
                c:CHG = {
                    in_civil_war = yes
                }
            }

            c:CHG = {
                every_country_at_war_with = {
                    limit = {
                        is_rebel_country = yes
                    }
                    every_owned_location = {
                        limit = {
                            OR = {
                                this.area = area:transoxiana_area
                                this.area = area:badakhshan_area
                            }
                        }
                        change_location_owner = root
                    }
                }
            }
        }

        if = {
            limit = {
                is_ai = yes
            }

            region:persia_region = {
                every_location_in_region = {
                    add_core = root
                }
            }
            region:caucasus_region = {
                every_location_in_region = {
                    add_core = root
                }
            }
            area:transoxiana_area = {
                every_location_in_area = {
                    add_core = root
                }
            }
            area:khwarazm_area = {
                every_location_in_area = {
                    add_core = root
                }
            }
            area:badakhshan_area = {
                every_location_in_area = {
                    add_core = root
                }
            }
            area:western_khorasan_area = {
                every_location_in_area = {
                    add_core = root
                }
            }
            area:eastern_khorasan_area = {
                every_location_in_area = {
                    add_core = root
                }
            }
            area:iraq_arabi_area = {
                every_location_in_area = {
                    add_core = root
                }
            }
            area:jazira_area = {
                every_location_in_area = {
                    add_core = root
                }
            }
        }

        region:khorasan_region = {
            every_owner_in_region = {
                limit = {
                    OR = {
                        capital.area = area:transoxiana_area
                        capital.area = area:badakhshan_area
                        capital.area = area:eastern_khorasan_area
                    }
                    has_primary_or_accepted_culture = culture:mongolian_culture
                    is_rebel_country = no
                }
                root = {
                    annex_country = {
                        country = prev
                        reason = Diplomatic
                    }
                }
            }
        }
        region:persia_region = {
            every_owner_in_region = {
                limit = {
                    capital.area = area:sistan_area
                    has_primary_or_accepted_culture = culture:mongolian_culture
                    is_rebel_country = no
                }
                root = {
                    annex_country = {
                        country = prev
                        reason = Diplomatic
                    }
                }
            }
        }
    }
}

#marriage to Saray
wrath_of_timur.2 = {
    type = country_event
    title = wrath_of_timur.2.title
    desc = wrath_of_timur.2.desc

    fire_only_once = yes

    dynamic_historical_event = {
		tag = TIM
		from = 1360.1.1
		to = 1370.1.1
		monthly_chance = 10
	}

    trigger = {
        root.ruler = character:chg_timur_lame_borjigin
    }

    option = {
        name = wrath_of_timur.2.a
        historical_option = yes

        if = {
            limit = { NOT = { has_policy = muslim_marriage } }
            add_policy = policy:muslim_marriage
        }
        create_character = {
            first_name = name_saray_mulk
            dynasty = dynasty:borjigin_dynasty
            culture = culture:mongolian_culture
            religion = religion:sunni
            father = character:chg_qazan_borjigin
            birth_date = 1341.1.1
            female = yes
            save_scope_as = timur_wife
        }
        ruler = {
            if = { 
                limit = { exists = this.first_spouse }
                divorce_character = this.first_spouse
            } 
            found_dynasty = gurkani_dynasty
            every_descendant = { change_dynasty = dynasty:gurkani_dynasty }
            marry_character = scope:timur_wife
        }
    }

    option = {
        name = wrath_of_timur.2.b
    }
}

				}

        }
    }
}

#This specifically exists because IF a player exists in persia, the ai seldom clicks on the button to select core region and ultimately change government_type
wrath_of_timur.13 = {
    type = country_event
    title = wrath_of_timur.13.title
    hidden = yes
    fire_only_once = yes

    dynamic_historical_event = {
		tag = TIM
		from = 1385.1.1
		to = 1390.1.1
		monthly_chance = 25
	}
    
    trigger = {
        has_game_rule = historical_timur_conquests
        is_ai = yes
        root.government_type != government_type:monarchy
    }

    immediate = {
        change_country_type = location
        change_government_type = government_type:monarchy
        unlock_government_reform_effect = { type = timurid_empire }
        add_reform = government_reform:timurid_empire
        set_country_rank = country_rank:rank_empire
    }
}
