wot_ottoman_extras.1 = {
    type = country_event
	category = situation_event

    title = wot_ottoman_extras.1.title
	desc = wot_ottoman_extras.1.desc

	immediate = {
		ruler_or_regent ?= { save_scope_as = target_character1 }
		scope:target.owner = {
			save_scope_as = target_country
			ruler_or_regent ?= { save_scope_as = target_character2 }
		}
	}

	option = {
		name = wot_ottoman_extras.1.a
		add_casus_belli = { target = scope:target.owner type = casus_belli:cb_ottoman_core_expansion province = scope:target }
		refresh_map_colors = yes
	}
}
