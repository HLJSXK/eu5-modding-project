scripted_effect cede_aqqoyunlu_lands = {
    if = {
        limit = { NOT = { country_exists = c:AKK } }
        every_owned_location = {
            remove_core = c:AKK
        }
        every_owned_location = {
            limit = {
                OR = {
                    this.province_definition = province_definition:diyarbekir_province
                    this.province_definition = province_definition:ar_ruha_province
                    this.province_definition = province_definition:raqqa_province
                    this.province_definition = province_definition:mardin_province
                    this.province_definition = province_definition:hasankeyf_province
                    this.province_definition = province_definition:harput_province
                    this.province_definition = province_definition:erzurum_province
                }
            }
            add_core = c:AKK
        }
        create_country_from_cores_in_our_locations = c:AKK
    }
    else = {
        every_owned_location = {
            limit = {
                OR = {
                    this.province_definition = province_definition:diyarbekir_province
                    this.province_definition = province_definition:ar_ruha_province
                    this.province_definition = province_definition:raqqa_province
                    this.province_definition = province_definition:mardin_province
                    this.province_definition = province_definition:hasankeyf_province
                    this.province_definition = province_definition:harput_province
                    this.province_definition = province_definition:erzurum_province
                }
            }
            add_core = c:AKK
            change_location_owner = c:AKK
        }
    }
    c:AKK = { trigger_event_silently = flavor_sheep_turkomen.1 }
}

crisis_of_the_timurids.1 = {
    type = country_event
	category = disaster_event
	title = crisis_of_the_timurids.1.title
	desc = crisis_of_the_timurids.1.desc

    immediate = {
        every_subject = {
            limit = {
                NOT = {
                    OR = {
                        capital.area = area:transoxiana_area
						capital.area = area:khwarazm_area
						capital.area = area:badakhshan_area
						capital.area = area:western_khorasan_area
						capital.area = area:eastern_khorasan_area
                        AND = {
                            capital.region = region:persia_region
                            NOT = { capital.area = area:azerbaijan_area }
                        }
                    }
                }
            }
            root = {
                cancel_subject = prev
            }
        }

        activate_situation = situation:rise_of_persia

        #release some historical nations
        cede_nogai_lands = yes
        cede_shirvan_lands = yes
        cede_georgian_lands = yes
        cede_aqqoyunlu_lands = yes
        cede_qaraqoyunlu_lands = yes
        cede_northern_caucasus_lands = yes
        cede_makran_lands = yes
        cede_hormuz_lands = yes
        cede_sindh_lands = yes
        cede_punjab_lands = yes
    }

    option = {
        name = crisis_of_the_timurids.1.a

        add_stability = stability_extreme_penalty
    }
}

crisis_of_the_timurids.5 = {
    type = country_event
	category = disaster_event
	title = crisis_of_the_timurids.5.title
	desc = crisis_of_the_timurids.5.desc

    trigger = {
        any_active_disaster = { disaster_type = disaster_type:crisis_of_the_timurids }
    }
    
    immediate = {
        random_subject = {
            limit = {
                subject_loyalty < 40
            }
            save_scope_as = target_subject
        }
    }

    option = {
        name = crisis_of_the_timurids.5.a

        cancel_subject = scope:target_subject

        ai_will_select  = { value = 70 }
    }
    option = {
        name = crisis_of_the_timurids.5.b

        add_gold = -150

        scope:target_subject = {
            add_liberty_desire = -10

        }
        
        ai_will_select  = { value = 30 }
    }
}
