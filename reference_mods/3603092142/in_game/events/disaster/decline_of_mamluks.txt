flavor_mamluk_stagnation.100 = {
    type = country_event
    title = flavor_mamluk_stagnation.100.title
    category = disaster_event
    desc = flavor_mamluk_stagnation.100.desc

    option = {
        name = flavor_mamluk_stagnation.100.a
        historical_option = yes

        custom_tooltip = {
            text = mamluk_collapse_tt
            c:MAM = {
                every_country_in_hierarchy = {
                    every_owned_location = {
                        limit = {
                            OR = {
                                region = region:anatolia_region
                                region = region:egypt_region
                                region = region:crescent_region
                                area = area:hedjaz_area
                                area = area:north_yemen_area
                            }
                        }
                        change_location_owner = c:TUR
                    }
                    
                }
                every_subject = {
                    c:MAM = { cancel_subject = prev }
                }
            }

            c:MAM = {
                while = {
                    limit = { num_locations > 0 }

                    random_owned_location = {
                        save_scope_as = capital_location
                    } 
                    scope:capital_location = {
                        create_country_from_location = {
                            save_scope_as = new_state
                            set_capital = prev                      
                        }

                        area = {
                            every_location_in_area = {
                                limit = { owner ?= c:MAM }
                                add_core = scope:new_state
                                change_location_owner = scope:new_state
                            }
                        }
                    }
                        
                }
            }
        }  
    }

    option = {
        name = flavor_mamluk_stagnation.100.b

        add_prestige = -10
        c:MAM = { add_gold = -4000 }
        add_gold = 4000
    }
}
