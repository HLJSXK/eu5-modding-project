wot_turkish_expansion = {
    monthly_spawn_chance = 100
	hint_tag = hint_wot_turkish_expansion

    can_start = {
		current_date > 1444.1.1
		current_date < 1500.1.1
        country_exists = c:TUR
        situation:rise_of_the_ottomans = { situation_is_active = no }
        c:TUR = { num_locations > 500 }
	}

    can_end = {
		wot_turkish_expansion_end_trigger = yes
	}

    visible = {
		OR = {
            is_neighbor_of = c:TUR
			is_subject_of = c:TUR
            is_enemy_of = c:TUR
		    is_rival_of = c:TUR
			has_presence_in = sub_continent:middle_east
            has_presence_in = sub_continent:north_africa
            has_presence_in = sub_continent:eastern_europe
		}
	}

    on_start = {
        every_country = {
            limit = {
                this != c:TUR
                OR = {
                    is_neighbor_of = c:TUR
                    is_subject_of = c:TUR
                    is_enemy_of = c:TUR
                    is_rival_of = c:TUR
                    has_presence_in = sub_continent:middle_east
                    has_presence_in = sub_continent:north_africa
                    has_presence_in = sub_continent:eastern_europe
                }
            }
            trigger_event_silently = wot_ottoman_extras.10
        }

        c:TUR = {
            trigger_event_silently = wot_ottoman_extras.9
        }
    }

    on_ended = {
        c:TUR = {
            trigger_event_silently = wot_ottoman_extras.11
        }
        every_country = {
            limit = {
                this != c:TUR
                OR = {
                    is_neighbor_of = c:TUR
                    is_subject_of = c:TUR
                    is_enemy_of = c:TUR
                    is_rival_of = c:TUR
                    has_presence_in = sub_continent:middle_east
                    has_presence_in = sub_continent:north_africa
                    has_presence_in = sub_continent:eastern_europe
                }
            }
            trigger_event_silently = wot_ottoman_extras.12
        }
    }

    tooltip = {
		if = {
			limit = { owner = c:TUR }
			custom_tooltip = WOT_OTTOMAN
		}
		else_if = {
			limit = { owner ?= { is_allied_with = { target = c:TUR } }}
			custom_tooltip = WOT_OTTOMAN_ALLIES
		}
		else_if = {
			limit = { owner ?= { is_subject_of = c:TUR }}
			custom_tooltip = WOT_OTTOMAN_SUBJECTS
		}
		else_if = {
			limit = { owner ?= { is_at_war_with = c:TUR } }
			custom_tooltip = WOT_CURRENTLY_INVADED
		}
		else_if = {
			limit = { owner ?= { has_target_casus_belli_on_us = c:TUR } }
			custom_tooltip = WOT_FUTURE_INVASION_TARGET
		}
	}

    map_color = {
		if = {
			limit = { owner ?= c:TUR }
			value = map_ottomans
		}
		else_if = {
			limit = {
				OR = {
					owner ?= { is_allied_with = { target = c:TUR } }
					owner ?= { is_subject_of = c:TUR }
				}
			}
			value = blue
		}
		else_if = {
			limit = { owner ?= { is_at_war_with = c:TUR } }
			value = map_serbian
		}
		else_if = {
			limit = { owner ?= { has_target_casus_belli_on_us = c:TUR } }
			value = orange
		}
		else = {
			value = define:NMapColors|DEFAULT_COLOR
		}
	}
	
	legend_key = { 
		desc = "TUR"
		color = map_ottomans
		require_color_on_map = yes
	}
	legend_key = { 
		desc = "game_concept_ally" #Until loc fix
		color = blue
		require_color_on_map = yes
	}
	legend_key = { 
		desc = "game_concept_rival_enemy"
		color = map_serbian
		require_color_on_map = yes
	}
	legend_key = { 
		desc = "game_concept_casus_belli"
		color = orange
		require_color_on_map = yes
	}
}
