rise_of_persia_recalculate_top_3 = {
	every_country = {
		limit = {
			NOT = { tag = TIM }
			NOT = {
				is_target_in_global_variable_list = {
					name = eligible_amirs_list
					target = this
				}
			}
			any_primary_or_accepted_culture ?= { has_culture_group = culture_group:iranian_group }
		}
		add_to_global_variable_list = { name = eligible_amirs_list target = this }

		add_country_modifier = {
			modifier = ai_force_annexation
			years = -1
			mode = add_and_extend
		}
	}
	ordered_in_global_list = {  
		variable = eligible_amirs_list
		limit = { is_subject = no }
		order_by = {
			add = {
				value = military_strength
			}
			add = {
				value = monthly_income_trade_and_tax
			}
			if = {
				limit = { is_subject = yes }
				multiply = 0.75
			}
			if = {
				limit = { tag = TIM }
				multiply = 0.25
			}
			if = {
				limit = { tag = ABL }
				multiply = 2
			}
			if = {
				limit = { has_reform = government_reform:legacy_of_genghis }
				multiply = 0.35
			}
		} 
		max = 1
		check_range_bounds = no 
		situation:rise_of_persia = {
			set_variable = {
				name = strongest_amir_variable
				value = prev
			}
			set_variable = {
				name = strongest_strongest_amir_total_score_variable
				value = {
					add = var:strongest_amir_variable.military_strength
					add = var:strongest_amir_variable.monthly_income_trade_and_tax 
				}
			}
		}
	}
	
	if = {
		limit = {
			global_variable_list_size = {
				name = eligible_amirs_list
				value > 1
			}
		}
		ordered_in_global_list = {	#2nd strongest  
			variable = eligible_amirs_list
			limit = { 
				this != situation:rise_of_persia.var:strongest_amir_variable 
				is_subject = no
			}
			order_by = {	
				add = {
					value = military_strength
				}
				add = {
					value = monthly_income_trade_and_tax
				}
			} 
			max = 1
			check_range_bounds = no
			situation:rise_of_persia = {
				set_variable = {
					name = second_strongest_strongest_amir_variable
					value = prev
				}
				set_variable = {
					name = second_strongest_strongest_amir_total_score_variable
					value = {
						add = var:second_strongest_strongest_amir_variable.military_strength
						add = var:second_strongest_strongest_amir_variable.monthly_income_trade_and_tax 
					}
				}
			}
		}
	}
	else = {
		situation:rise_of_persia = {
			remove_variable = second_strongest_strongest_amir_variable
			remove_variable = second_strongest_strongest_amir_total_score_variable
		}
	}

	if = {
		limit = {
			global_variable_list_size = {
				name = eligible_amirs_list
				value > 2
			}
		}
		ordered_in_global_list = { #3rd Strongest 
			variable = eligible_amirs_list
			limit = { 
				this != situation:rise_of_persia.var:strongest_amir_variable 
				this != situation:rise_of_persia.var:second_strongest_strongest_amir_variable 
				is_subject = no
			}
			order_by = {
				add = {
					value = military_strength
				}
				add = {
					value = monthly_income_trade_and_tax
				}
			} 
			max = 1
			check_range_bounds = no 
			situation:rise_of_persia = {
				set_variable = {
					name = third_strongest_strongest_amir_variable
					value = prev
				}
				set_variable = {
					name = third_strongest_strongest_amir_total_score_variable
					value = {
						add = var:third_strongest_strongest_amir_variable.military_strength
						add = var:third_strongest_strongest_amir_variable.monthly_income_trade_and_tax
					}
				}
			}
		}
	}
	else = {
		situation:rise_of_persia = {
			remove_variable = third_strongest_strongest_amir_variable
			remove_variable = third_strongest_strongest_amir_total_score_variable
		}
	}
}
