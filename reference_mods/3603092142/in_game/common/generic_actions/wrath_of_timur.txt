
REPLACE:rot_plan_invasion = {
	type = situation	
	potential = {
		scope:actor = {
			tag = TIM
		}
	}

	select_trigger = {
		looking_for_a = situation
		target_flag = recipient
		name = "choose_situation"
		column = {
			data = name
		}
		visible = {
			situation:rise_of_timur = this
			situation_is_active = yes
		}
	}

	select_trigger = {
		looking_for_a = country
		target_flag = target
		name = "rot_plan_invasion_select_country"
		none_available_msg_key = "rot_plan_invasion_no_countries"
		column = {
			data = name
		}
		column = {
			data = expected_army_size
		}
		visible = {
			NOT = { has_truce_with = scope:actor }
			NOT = { is_allied_with = { target = scope:actor } }
			NOT = { is_subject_of = scope:actor }
			within_diplomatic_range = scope:actor
			OR = {
				is_neighbor_of = scope:actor
				any_neighbor_country = {
					is_subject_of = scope:actor
				}
			}
			OR = {
				has_game_rule = ahistorical_timur_conquests
				any_owned_location = {
					OR = {
						this.area = area:badakhshan_area
						this.area = area:western_khorasan_area
						this.area = area:eastern_khorasan_area
						this.region = region:persia_region
						AND = {
							current_year > 1370
							OR = {
								this.area = area:transoxiana_area
								this.area = area:khwarazm_area
								this.region = region:caucasus_region
								this.area = area:iraq_arabi_area
								this.area = area:jazira_area
							}
						}
						AND = {
							current_year > 1385
							OR = {
								this.province_definition = province_definition:sibi_province
								this.province_definition = province_definition:upper_sindh_province
								this.province_definition = province_definition:koh_e_sulayman_province
								this.province_definition = province_definition:derajat_province
								this.province_definition = province_definition:jech_doab
								this.province_definition = province_definition:pothwar_province
								this.province_definition = province_definition:karman_province
								this.province_definition = province_definition:peshawar_province
							}
						}
					}
				}
			}
		}
	}

	cooldown = {
		type = rot_plan_invasion
		months = 6
	}

	automation_tick = never
	automation_tick_frequency = 12

	effect = {
		scope:actor = {
			if = {
				limit = { exists = scope:target }
				add_casus_belli = {
					type = casus_belli:cb_timurs_conquests
                	target = scope:target
				}
				declare_war_with_cb = {
					target = scope:target
					type = casus_belli:cb_timurs_conquests
				}
			}
			else = {
				custom_tooltip = rot_plan_invasion_tt
			}
		}
	}
	ai_will_do = {
		add = 100
		multiply = {
			if = {
				limit = { exists = scope:target }
				value = {
					value = "scope:actor.opinion(scope:target)"
					multiply = -1
				}
			}
			else = {
				value = 1	#We need that else or otherwise the whole chance is multiplied with 0
			}
		}
		add = {
			if = {
				limit = { exists = scope:target }
				value = "scope:actor.antagonism(scope:target)"
				divide = -10
			}
		}
		if = {
			limit = {
				any_location_in_the_world = {
					OR = {
						this.area = area:transoxiana_area
						this.area = area:khwarazm_area
						this.area = area:badakhshan_area
						this.area = area:western_khorasan_area
						this.area = area:eastern_khorasan_area
						this.area = area:iraq_arabi_area
						this.area = area:jazira_area
						this.region = region:caucasus_region
						this.region = region:persia_region
						this.province_definition = province_definition:sibi_province
						this.province_definition = province_definition:upper_sindh_province
						this.province_definition = province_definition:koh_e_sulayman_province
						this.province_definition = province_definition:derajat_province
						this.province_definition = province_definition:jech_doab
						this.province_definition = province_definition:pothwar_province
						this.province_definition = province_definition:karman_province
						this.province_definition = province_definition:peshawar_province
					}
					NOT = { owner ?= c:TIM }
				}
			}
			add = {
				value = 1000
			}
		}
	}
}

rot_submit_to_timur = {
	type = situation

	potential = {
		scope:actor = {
			OR = {
				is_neighbor_of = c:TIM
				any_neighbor_country = {
					is_subject_of = c:TIM
				}
			}
			NOT = { is_subject_of = c:TIM }
			NOT = { tag = TIM }
		}
	}

	select_trigger = {
		looking_for_a = situation
		target_flag = recipient
		name = "choose_situation"
		column = {
			data = name
		}
		visible = {
			situation:rise_of_timur = this
			situation_is_active = yes
		}
	}

	cooldown = {
		type = rot_submit_to_timur
		months = 12
	}

	automation_tick = never
	automation_tick_frequency = 12

	effect = {
		c:TIM = {
			trigger_event_non_silently = { id = wrath_of_timur.8 }
		}
	}

	ai_will_do = {
		add = 100
		add = {
			if = {
				limit = { 
					exists = c:TIM 
					c:TIM = {
						OR = {
							is_neighbor_of = scope:actor
							any_subject_or_below = {
								is_neighbor_of = scope:actor
							}
						}
					}
				}
				value = 200
			}
		}
	}
}
