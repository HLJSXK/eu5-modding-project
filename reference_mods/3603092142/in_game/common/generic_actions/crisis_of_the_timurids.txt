establish_governor = {
    type = disaster
    show_message = no

    potential = {
    }

    allow = {
        scope:actor = {
            in_civil_war = no
            has_reform = government_reform:timurid_diwan_reform
        }
        scope:actor.ruler.dynasty = {
            any_character_in_dynasty = {
                is_eligible_heir = scope:actor
                NOT = { is_ruler = yes}
                count > 0
            }
        }
    }

    automation_tick = never
	automation_tick_frequency = 12
	ai_tick = monthly
	ai_tick_frequency = 3

    cooldown = {
		type = established_governor_cooldown_key
		years = 1
	}

    select_trigger = {
		looking_for_a = disaster
        name = "choose_disaster"
		target_flag = recipient
		column = {
			data = name
		}
		visible = {
			disaster_type = disaster_type:crisis_of_the_timurids
			disaster_is_active = yes
		}
	}

    select_trigger = {
        looking_for_a = area
        target_flag = target
        name = "choose_area"
		none_available_msg_key = "no_areas_available"
        column = { data = owner_flag }
		column = { data = name }
		column = { data = population }

        visible = {
            any_province_in_area = {
                owner ?= scope:actor
            }
            NOR = {
                this = area:eastern_khorasan_area
                this = area:transoxiana_area
                this = area:khwarazm_area
                this = area:western_khorasan_area
                this = area:badakhshan_area
            }
        }
    }
    effect = {
        scope:actor.ruler.dynasty = {
            random_character_in_dynasty = {
                limit = {
                    is_eligible_heir = scope:actor
                    NOT = { is_ruler = yes}
                }
                save_scope_as = governor
            }
        }
        scope:target = {
            random_location_in_area = {
                limit = {
                    owner = scope:actor
                }
                save_scope_as = chosen_capital
            }
        }
        scope:chosen_capital = {
            create_country_from_location = {
                save_scope_as = new_country
                set_new_ruler = scope:governor
                set_capital = prev
                change_government_type = government_type:monarchy
                add_reform = government_reform:soyurghal_governor_reform
                make_subject_of = {
                    target = scope:actor
                    type = subject_type:vassal
                }
            }
        }
        scope:actor = {
            every_owned_location = {
                limit = {
                    this.area ?= scope:target
                }
                add_core = scope:new_country
                change_location_owner = scope:new_country
            }
        }
    }
    ai_will_do = {
        add = 250
    }
}
