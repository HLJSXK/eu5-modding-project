curse_of_stefan_uros_iii = {

	image = "gfx/interface/illustrations/disaster/curse_of_stefan_uros_iii.dds"
	
	monthly_spawn_chance = monthly_spawn_chance_ultimate

	fire_only_once = yes

	can_start = {
		tag = SER
		custom_tooltip = {
			text = had_csd_disaster_variable_tt
			NOT = { has_variable = had_serbian_disaster_variable }
		}
		current_year < 1453
		#country_rank = country_rank:rank_empire
		character:ser_stefan_dusan = {
			is_alive = no
		}
		#ruler ?= { total_abilities <= 250 }
		stability <= 30
		government_power <= 75
	}
	

	can_end = {
		curse_of_stefan_uros_iii_end_trigger = yes
	}
	modifier = {
		monthly_pretender_rebel_growth = 0.05
		monthly_nationalist_rebel_growth = 0.05
		monthly_legitimacy = -0.75
		global_max_control = -0.25
		global_monthly_prosperity = -0.0025
		global_monthly_control = -0.0001
	}
	
	on_start = {
		add_government_power = -15
		trigger_event_non_silently = curse_of_stefan_uros_iii.1

		trigger_event_silently = curse_of_stefan_uros_iii.3
		trigger_event_silently = curse_of_stefan_uros_iii.3
		trigger_event_silently = curse_of_stefan_uros_iii.4
		trigger_event_silently = curse_of_stefan_uros_iii.5
		trigger_event_silently = curse_of_stefan_uros_iii.6
		trigger_event_silently = curse_of_stefan_uros_iii.6

	}
	on_end = {
		trigger_event_non_silently = curse_of_stefan_uros_iii.2
	}
	
	on_monthly = {
		random_list = {
			40 = { custom_tooltip = an_event_occurs_tt }
			60 = { custom_tooltip = no_event_occurs_tt }
		}
		hidden_effect = {
			random_list = {
				600 = {}
				100 = { trigger_event_silently = curse_of_stefan_uros_iii.3 }
				50 = { trigger_event_silently = curse_of_stefan_uros_iii.4 }
				50 = { trigger_event_silently = curse_of_stefan_uros_iii.5 }
				100 = { trigger_event_silently = curse_of_stefan_uros_iii.6 }
				50 = {
					trigger_event_silently = curse_of_stefan_uros_iii.6
					trigger_event_silently = curse_of_stefan_uros_iii.6
				}
				50 = {
					trigger_event_silently = curse_of_stefan_uros_iii.6
					trigger_event_silently = curse_of_stefan_uros_iii.6
					trigger_event_silently = curse_of_stefan_uros_iii.6
				}
			}
		}
	}
}
