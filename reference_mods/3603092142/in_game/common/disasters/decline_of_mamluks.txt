decline_of_mamluks = {

    #image = "gfx/interface/illustrations/disaster/decline_of_mamluks.dds"

    monthly_spawn_chance = 0.25

    fire_only_once = yes

    can_start = {
        tag = MAM
        current_year < 1380
    }

    can_end = {
        OR = {
            current_year > 1520
        }
    }

    modifier = {
		research_speed_modifier = -0.5
        embrace_institution_cost_modifier = 2
        global_institution_growth_modifier = -0.5
        monthly_towards_traditionalist = 0.1
    }

    on_start = {
        trigger_event_silently = flavor_mamluk_stagnation.1
    }

    on_end = {
        trigger_event_silently = flavor_mamluk_stagnation.10
    }

    on_monthly = {
        random_list = {
			10 = { custom_tooltip = an_event_occurs_tt }
			90 = { custom_tooltip = no_event_occurs_tt }
		}
        hidden_effect = {
            random_list = {
                90 = { }
                5 = {
                    trigger = {
                        research_progress >= 10
                    }
                    trigger_event_silently = flavor_mamluk_stagnation.2
                }
                5 = {
                    trigger_event_silently = flavor_mamluk_stagnation.3
                }
            }
        }
    }
}
