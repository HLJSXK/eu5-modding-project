dominate_fars = {
    cost = {
        add = {
			desc = "DIPLOREASON_BASE"
			value = 50
		}
    }

    base_antagonism = 10
    antagonism_type = antagonism_taking_land

    potential = {
        OR = {
            scope:war.attacker_leader ?= c:MZF
            scope:war.attacker_leader ?= c:INJ
        }
        scope:war.defender_leader ?= {
            any_country_in_hierarchy = {
                any_owned_location = {
                    OR = {
                        this.area = area:kerman_area
                        this.area = area:fars_area
                        this.area = area:khuzestan_area
                        this.province_definition = province_definition:ardestan_province
                        this.province_definition = province_definition:isfahan_province
                        this.province_definition = province_definition:qumisha_province
                    }
                    count > 0
                }
            }
        }
    }

    effect = {
        if = {
            limit = {
                OR = {
                    AND = {
                        scope:winner = c:MZF
                        scope:loser = c:INJ
                    }
                    AND = {
                        scope:winner = c:INJ
                        scope:loser = c:MZF
                    }
                }
            }
            scope:winner = {
                trigger_event_non_silently = muzaffarids_injuids_flavor.4
            }
        }
        scope:loser ?= {
            every_country_in_hierarchy = {
                every_owned_location = {
                    limit = {
                        OR = {
                            this.area = area:kerman_area
                            this.area = area:fars_area
                            this.area = area:khuzestan_area
                            this.province_definition = province_definition:ardestan_province
                            this.province_definition = province_definition:isfahan_province
                            this.province_definition = province_definition:qumisha_province
                        }
                    }
                    add_to_temporary_list = fars_region_locations
                }
            }

            if = {
                limit = {
                    list_size = {
                        name = fars_region_locations
                        value >= 1
                    }
                }
                every_in_list = {
                    list = fars_region_locations
                    change_location_owner = scope:winner
                }
            }
        }
    }

    ai_desire = {
        value = 500
    }
}
