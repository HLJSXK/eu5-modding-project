			desc = "DIPLOREASON_BASE"
			value = 50
		}

    potential = {
        scope:war = {
            casus_belli ?= casus_belli:cb_timurs_conquests
        }
        scope:war.attacker_leader ?= scope:winner
    }

    effect = {
        custom_tooltip = {
            text = timur_conquest_peace_tt
            scope:loser = {
                every_country_in_hierarchy = {
                    every_owned_location = {
                        limit = {
                            OR = {
                                this.area = area:transoxiana_area
                                this.area = area:khwarazm_area
                                this.area = area:badakhshan_area
                                this.area = area:western_khorasan_area
                                this.area = area:eastern_khorasan_area
                                this.area = area:iraq_arabi_area
                                this.area = area:jazira_area

                                this.region = region:persia_region
                                this.province_definition = province_definition:sibi_province
                                this.province_definition = province_definition:upper_sindh_province
                                this.province_definition = province_definition:koh_e_sulayman_province
                                this.province_definition = province_definition:derajat_province
                                this.province_definition = province_definition:jech_doab
                                this.province_definition = province_definition:pothwar_province
                                this.province_definition = province_definition:karman_province
                                this.province_definition = province_definition:peshawar_province
                            }
                        }
                        add_to_temporary_list = owned_persian_provinces
                    }
                }

                if = {
                    limit = {
                        list_size = {
                            name = owned_persian_provinces
                            value >= 1
                        }
                    }
                    every_in_list = {
                        list = owned_persian_provinces
                        change_location_owner = scope:winner
                    }
                    add_gold = -200
                    scope:winner = {
                        add_gold = 200
                    }
                }
                else = {
                    add_gold = -500
                    scope:winner = {
                        add_gold = 500
                    }
                }
            }
        }
    }

        value = 500
    }
