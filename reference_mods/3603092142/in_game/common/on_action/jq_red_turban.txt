on_location_occupied = {
	on_actions = {
		jq_red_turban_occupied
	}
}

on_siege_won = {
	on_actions = {
		jq_red_turban_occupied
	}
}

jq_red_turban_occupied = {
	effect = {
		if = {
			limit = {
				scope:target.sub_continent = sub_continent:east_asia
				exists = scope:target.owner
        any_current_war = {
          is_in_war = scope:target.owner
          OR = {
            casus_belli ?= casus_belli:cb_red_turban_rebellions
            casus_belli ?= casus_belli:cb_red_turban_rebellions_rein_in_rebels
            casus_belli ?= casus_belli:cb_red_turban_rebellions_against_other_rebels
            casus_belli ?= casus_belli:cb_chinese_unification
          }
        }
        root = {
          is_subject = no
          OR = {
            has_or_had_tag = CHI
            has_reform = government_reform:warring_state
            has_variable = unlocked_warring_state
          }
        }
        scope:target = {
          owner = {
            is_subject = no
            or = {
              has_or_had_tag = CHI
              has_reform = government_reform:warring_state
              has_variable = unlocked_warring_state
            }
          }
        }
			}
      scope:target = { 
        if = {
          limit = {
            is_province_capital = yes
          }
          scope:target.province = {
            every_location_in_province = {
              change_location_owner ?= root
            }
          }
        }
        else = {
          change_location_owner ?= root
        }
      }
    }
	}
}
