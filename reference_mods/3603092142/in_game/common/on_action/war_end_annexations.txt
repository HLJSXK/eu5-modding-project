on_winning_war = { 
    on_actions = {
        on_generic_end_annex_war
        on_safavid_end_war
    }
}

on_generic_end_annex_war = {
    trigger = {
        scope:winner = {
            this = scope:war.attacker_leader
            is_ai = yes
        }
        scope:loser = {
            is_ai = yes
        }
        scope:war = {
            casus_belli ?= casus_belli:cb_annex
        }
    }

    effect = {
        scope:war.defender_leader = {
            save_scope_as = main_defender
        }

        #white_peace = scope:war

        scope:winner = {
            annex_country = {
                country = scope:main_defender
                reason = MilitaryConquest
            }
        }
    }
}

on_safavid_end_war = {
    trigger = {
        OR = {
            tag_exists = IRA
            tag_exists = ABL
        }
        has_game_rule = historical_safavids
        scope:winner ?= {
            OR = {
                tag = IRA
                tag = ABL
            }
        }
        situation:rise_of_persia = {
            situation_is_active = yes
        }
    }

    effect = {
        scope:war.defender_leader = {
            save_scope_as = main_defender
        }

        #white_peace = scope:war
        
        scope:winner = {
            if = {
                limit = {
                    scope:main_defender = {
                        is_ai = yes
                    }
                }
                scope:main_defender = {
                    every_owned_location = {
                        limit = {
                            OR = {
                                this.region = region:persia_region
                                this.area = area:eastern_khorasan_area
                                this.area = area:western_khorasan_area
                                this.area = area:shirvan_area
                                this.area = area:iraq_arabi_area
                                this.area = area:jazira_area
                                this.area = area:armenian_highlands_area
                            }
                        }
                        change_location_owner = scope:winner
                    }
                }
            }
        }
    }
}
