on_location_changed_owner = {
    on_actions = {
        on_timurid_occupy_ahistorical_borders
        on_safavid_occupy_ahistorical_borders
    }
}

on_timurid_occupy_ahistorical_borders = {
	trigger = {
		tag_exists = TIM
        has_game_rule = historical_timur_conquests
		scope:winner ?= {
			tag = TIM
			modifier:auto_conquer_at_war = yes
		}

        NOR = {
            root.area = area:transoxiana_area
            root.area = area:khwarazm_area
            root.area = area:badakhshan_area
            root.area = area:western_khorasan_area
            root.area = area:eastern_khorasan_area
            root.area = area:iraq_arabi_area
            root.area = area:jazira_area
            root.region = region:caucasus_region
            root.region = region:persia_region
            root.province_definition = province_definition:sibi_province
            root.province_definition = province_definition:upper_sindh_province
            root.province_definition = province_definition:koh_e_sulayman_province
            root.province_definition = province_definition:derajat_province
            root.province_definition = province_definition:jech_doab
            root.province_definition = province_definition:pothwar_province
            root.province_definition = province_definition:karman_province
            root.province_definition = province_definition:peshawar_province
        }
	}
	effect = {
        root = {
            change_location_owner = scope:loser
        }
	}
}

on_safavid_occupy_ahistorical_borders = {
    trigger = {
        OR = {
            tag_exists = IRA
            tag_exists = ABL
        }
        has_game_rule = historical_safavids
        scope:winner ?= {
            OR = {
                tag = IRA
                tag = ABL
            }
            modifier:auto_conquer_at_war = yes
        }
        NOR = {
            root.region = region:persia_region
            root.area = area:eastern_khorasan_area
            root.area = area:western_khorasan_area
            root.area = area:shirvan_area
            root.area = area:iraq_arabi_area
            root.area = area:jazira_area
            root.area = area:armenian_highlands_area
        }
    }
	effect = {
        root = {
            change_location_owner = scope:loser
        }
	}
}
