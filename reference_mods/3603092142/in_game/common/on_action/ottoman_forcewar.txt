four_yearly_country_pulse = {
    on_actions = {
        ottoman_war_helper
        ottoman_war_invasion_helper
    }
}

ottoman_war_helper = {
    trigger = {
        tag = TUR
        is_ai = yes
        at_war = no
        situation:rise_of_the_ottomans = { situation_is_active = yes }
    }

    effect = {
        random_neighbor_country = {
            limit = {
                root = { has_casus_belli_of_type_on = { type = casus_belli:cb_ottoman_core_expansion target = prev } }
            }
            save_scope_as = war_target
        }

        if = {
            limit = { exists = scope:war_target }
            declare_war_with_cb = { type = casus_belli:cb_ottoman_core_expansion target = scope:war_target }
        }
    }
}

ottoman_war_invasion_helper = {
    trigger = {
        tag = TUR
        is_ai = yes
        at_war = no
        situation:wot_turkish_expansion = { situation_is_active = yes }
    }

    effect = {
        random_neighbor_country = {
            limit = {
                root = { has_casus_belli_of_type_on = { type = casus_belli:cb_ottoman_invasion target = prev } }
            }
            save_scope_as = war_target
        }

        if = {
            limit = { exists = scope:war_target }
            declare_war_with_cb = { type = casus_belli:cb_ottoman_invasion target = scope:war_target }
        }
    }
}

yearly_country_pulse = {
    on_actions = {
        ottoman_building_builder
    }
}

ottoman_building_builder = {
    trigger = {
        is_ai = yes
        has_advance = armory_advance
        current_year < 1470
        tag = TUR
    }

    effect = {
        every_owned_location = {
            limit = { can_build_building = building_type:armory }

            construct_building = {
                building_type = building_type:armory
                cost_multiplier = 0
                cost_multiplier_reason = "ottoman_discount"
            }
        }
    }
}
