
on_game_start = {
	on_actions = {
		on_game_start_historical_tweaks
	}
}

on_game_start_historical_tweaks = {
	effect = {
		c:HUN = {
			add_reform = government_reform:ht_hungarian_malaise
		}
		c:BOH = {
			add_reform = government_reform:ht_bohemian_borders
		}
	}
}

yearly_country_pulse = {
	on_actions = {
		historical_tweaks_monthly_tick
		ht_russian_unification_yearly_tick
	}
}

historical_tweaks_yearly_tick = {
	random_events = {

		50 = historical_tweaks_union.2 #Flip to habsburg fast law

		10 = historical_tweaks_union.1 #Opinion boost

		10 = 0

		chance_to_happen = 75
	}
}

ht_russian_unification_yearly_tick = {
	random_events = {
		90 = historical_tweaks.4
		10 = 0
		chance_to_happen = 75
	}
}

on_diplomatic_annex = {
	events = {
		historical_tweaks_union.3
	}
}
