on_annexed = {
    on_actions = {
        on_rise_of_persia_annexed
    }
}

on_rise_of_persia_annexed = {
    effect = {
        if = {
			limit = {
				is_situation_active = situation:rise_of_persia
				any_in_global_list = {
					variable = eligible_amirs_list
					OR = {
						this = root
						this = scope:target
					}
				}
			}
			rise_of_persia_recalculate_top_3 = yes
			if = {
				limit = {
					any_in_global_list = {
						variable = eligible_amirs_list
						this = scope:target
					}
				}
				remove_list_global_variable = { 
					name = eligible_amirs_list 
					target = scope:target 
				}
			}
		}
    }
}
