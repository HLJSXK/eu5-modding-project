cb_ottoman_invasion = {
    allow_creation = {
        tag = TUR
    }

    visible = {
        situation:rise_of_the_ottomans ?= { situation_is_active = no }
        scope:target = {
            any_owned_location = {
                OR = {
                    this.region = region:anatolia_region
                    this.region = region:crescent_region
                    this.region = region:egypt_region
                    this.region = region:balkan_region
                    this.region = region:carpathia_region
                    this.area = area:tripolitania_area
                    this.area = area:tunis_area
                    this.area = area:algiers_area
                    this.area = area:atlas_high_plateau
                    this.area = area:hedjaz_area
                    this.area = area:north_yemen_area
                    this.area = area:south_yemen_area
                    this.area = area:yedisan_area
                    this.area = area:pryazovia_area
                    this.province_definition = province_definition:zaporizhzhia_province
                    this.area = area:crimea_area
                    this.area = area:azov_area
                    this.province_definition = province_definition:akhtar_province
                    this.province_definition = province_definition:privolnoye_province
                    this.province_definition = province_definition:taman_province
                    this.area = area:armenian_highlands_area
                }
            }
        }
    }


    ai_subjugation_desire = -50
    ai_cede_location_desire = {
		add = {
			if = {
				limit = {
                    OR = {
                        AND = {
                            scope:location.region = region:balkan_region
                            scope:location.region = region:carpathia_region
                            NOR = {
                                scope:location.area = area:croatia_area
                                scope:location.area = area:slovenia_area
                            }
                        }
                    }
				}
                add = 1250
			}
		}
        add = {
            if = {
                limit = {
                    current_year > 1400
                    OR = {
                        scope:location.region = region:anatolia_region
                        scope:location.area = area:armenian_highlands_area
                    }
                }
                add = 500
            }
        }
        add = {
            if = {
                limit = {
                    current_year > 1480
                    OR = {
                        scope:location.area = area:levant_area
                        scope:location.region = region:egypt_region
                    }
                }
                add = 750
            }
        }
	}

    war_goal_type = ottoman_invasion_wargoal
}

cb_ottoman_core_expansion = { #CB to help with core expansion, i.e balkans and anatolia
    allow_creation = { always = no }
    visible = { always = no}

    province = {
        any_location_in_province = {
            always = yes
        }
    }

    ai_subjugation_desire = {
        add = {
            if = {
                limit = {
                    scope:recipient.capital.region = region:carpathia_region
                }
                add = 500
            }
        }
    }

    ai_cede_location_desire = {
        add = {
			if = {
				limit = {
                    AND = {
                        scope:location.region = region:balkan_region
                        NOR = {
                            scope:location.area = area:croatia_area
                            scope:location.area = area:slovenia_area
                        }
                    }
				}
				add = 1350
			}
		}
        add = {
            if = {
                limit = {
                    current_year < 1400
                    scope:location.province_definition = province_definition:constantinople_province
                }
                add = -1350
            }
        }
        add = {
            if = {
                limit = {
                    current_year > 1400
                    scope:location.region = region:anatolia_region
                }
                add = 500
            }
        }
        subtract = {
            if = {
                limit = {
                    OR = {
                        scope:location.region != region:anatolia_region
                        scope:location.region != region:anatolia_region
                    }
                }
                subtract = 1000
            }
        }
    }

    war_goal_type = ottoman_core_expansion_wargoal
}
