# Small countries can mobilize a larger proportion of their population into levies
# Used to give coalitions of small countries a fighting chance against larger nations
# 1 location gives +200% levy size and + 100% recovery speed
# Scales to 0 at 10 locations
# Exponential scaling to mostly benefit the smallest countries
small_country_levy_buff = {
	type = country
	category = country

	potential_trigger = {
		num_locations <= 10
	}

	scales_with = {
		value = 10
		subtract = num_locations
		divide = 9
		power = 3
	}

	global_levy_size_modifier = 2 # Up to +200% global levy size
	levy_recovery_modifier = 1 # Up to +100% levy recovery speed

}
