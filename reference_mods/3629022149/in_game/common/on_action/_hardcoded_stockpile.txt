on_game_start = {
on_actions = {
		initialize_stockpile
	}
}

initialize_stockpile = {
	effect = {
		every_country = {
			limit = { has_markets= yes }

			every_market_center_in_country = {
			save_scope_as = full_market
			every_goods= {
				save_scope_as = abundant_good
				if = {
				#limit = { "scope:full_market.goods_supply_in_market(scope:abundant_good)" > "scope:full_market.goods_demand_in_market(scope:abundant_good)" }
				limit = { OR = {	"scope:full_market.goods_supply_in_market(scope:abundant_good)" > 0.2
									scope:abundant_good = goods:silver
									scope:abundant_good = goods:goods_gold} }
					scope:full_market={
						add_goods_supply = {
							goods = scope:abundant_good
							amount = {value =  "scope:full_market.goods_supply_in_market(scope:abundant_good)"
										subtract = "scope:full_market.goods_demand_in_market(scope:abundant_good)"
										multiply=12
										min= {	value = location.local_stockpile_capacity 
												multiply = 0.05}
										if = { limit = { scope:abundant_good = goods:goods_gold } min=10}
										if = { limit = { scope:abundant_good = goods:silver } min=10}
								
						}
						
						}	
						
					}
				}
			}	
		}
			}
		}
	}
