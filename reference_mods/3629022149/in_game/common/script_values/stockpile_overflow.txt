
local_stockpile_capacity={  #location scope
value = "modifier:maximum_stockpile_capacity"
}

upper_value = { #market scope
	value = location.local_stockpile_capacity 
	multiply = 0.05
 }
 
lower_value = { #market scope
	value = location.local_stockpile_capacity 
	multiply = 0.05
 }
 
 warehouse_value = {
	value = location.local_stockpile_capacity 
	multiply = 0.04
 }
 
 1cutoff_value = {
	value = location.local_stockpile_capacity 
	multiply = 0.5
 }
 
 2cutoff_value = {
	value = location.local_stockpile_capacity 
	multiply = 0.75
 }
 
 3cutoff_value = {
	value = location.local_stockpile_capacity 
	multiply = 0.95
 }
 
lower_value_overflow = {
		value = "scope:full_market.stockpile_in_market(scope:abundant_good)"
		divide = "location.local_stockpile_capacity"
		subtract = 0.05
		multiply = -100
 }
 

 
 
 upper_value_overflow = {
		value = "scope:full_market.stockpile_in_market(scope:abundant_good)"
		divide = "location.local_stockpile_capacity"
		#divide = "scope:full_market.full_market_capacity" 
		subtract = 0.05
		multiply = 100
 }
 


 supply_effect = {
 
	#value = "scope:full_market.goods_demand_in_market(scope:abundant_good)"
	value =  	"scope:full_market.stockpile_in_market(scope:abundant_good)"
	subtract = upper_value
	#save_temporary_value_as=extra_stock
	divide = 18
	min = 1
	#value = "scope:full_market.goods_supply_in_market(scope:abundant_good)"
	
	divide=95
	
	multiply = upper_value_overflow
	
	if = { limit = { 	"scope:full_market.stockpile_in_market(scope:abundant_good)" > 	scope:full_market.1cutoff_value }
	add = { value = "scope:full_market.stockpile_in_market(scope:abundant_good)"
			subtract = scope:full_market.1cutoff_value
			divide =24
			#multiply=0
			if = { limit = { 	"scope:full_market.stockpile_in_market(scope:abundant_good)" > 	2cutoff_value }
				multiply = 1.5 #to cutoff to not prelong that 
				if = { limit = { 	"scope:full_market.stockpile_in_market(scope:abundant_good)" > 	3cutoff_value }
				add = 1000 #to cutoff to not prelong that
			}}
		}
	}
	#min=0
	#value = scope:full_market.1cutoff_value
}

 stockpile_effect = {
	value = supply_effect
	multiply = 0

}

  low_supply_effect = {
  #value =  "scope:full_market.goods_supply_in_market(scope:abundant_good)"
	value = -0.1
	
	multiply = lower_value_overflow
	min={value =  "scope:full_market.goods_supply_in_market(scope:abundant_good)"
		multiply=-1}
	max=-0.1
							
							
	#multiply = -1
}

low_stockpile_effect = {
	value= "scope:full_market.stockpile_in_market(scope:abundant_good)"
	#min = 0.1
	
	max = { value =  low_supply_effect
			multiply = -1 
			
			}
	
}

total_production_efficiency={
value = 1 
add={value = location.modifier:local_production_efficiency}
		add={value = location.owner.modifier:global_production_efficiency}
		if = {
				limit = {	location.owner= { "modifier:raw_material_in_province_impact" >0 }
							location.province = { any_location_in_province = {raw_material=scope:abundant_good} }
				}
				add= location.owner.modifier:raw_material_in_province_impact    #0.1
				}
}
