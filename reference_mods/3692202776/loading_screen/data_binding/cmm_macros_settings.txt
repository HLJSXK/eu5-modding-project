# CMM GUI Macros: Settings

# Args: SettingKey (string)
# Returns: boolean
macro = {
    description = "True when a setting key has been registered (type > 0)"
    definition = "CMMSettingIsRegistered(SettingKey)"
    replace_with = "GreaterThan_CFixedPoint(CMMMapValue('cmm_type', SettingKey), '(CFixedPoint)0')"
}

# Args: SettingKey (string)
# Returns: boolean
macro = {
    description = "True when a setting key is registered as global"
    definition = "CMMSettingIsGlobal(SettingKey)"
    replace_with = "CMMValueEqualsOne(CMMMapValue('cmm_is_global', SettingKey))"
}

# Args: SettingKey (string)
# Returns: boolean
macro = {
    description = "True when local player can edit this setting"
    definition = "CMMCanEditSetting(SettingKey)"
    replace_with = "Or(Not(CMMSettingIsGlobal(SettingKey)), And(Or(EqualTo_string(SettingKey, 'cmm_core__enable_host_only_tools'), And(CMMSettingIsRegistered('cmm_core__enable_host_only_tools'), CMMValueEqualsOne(CMMSettingValue('cmm_core__enable_host_only_tools')))), Or(Not(GameIsMultiplayer), IsHost)))"
}

# Args: SettingKey (string)
# Returns: value
macro = {
    description = "Read current setting value from country or global scope"
    definition = "CMMSettingValue(SettingKey)"
    replace_with = "Select_CFixedPoint(CMMSettingIsGlobal(SettingKey), GetVariableFromGlobalVariableMap('cmm', MakeScopeFlag(SettingKey)).GetValue, Player.MakeScope.GetVariableFromVariableMap('cmm', MakeScopeFlag(SettingKey)).GetValue)"
}

# Args: SettingKey (string)
# Returns: boolean
macro = {
    description = "Visibility rule for one setting row"
    definition = "CMMSettingRowVisible(SettingKey)"
    replace_with = "And(CMMMatchesSelectedModAndTab(CMMMetadataFlag('cmm_setting_owner_mod_id_', SettingKey), CMMMetadataFlag('cmm_setting_owner_tab_key_', SettingKey)), And(CMMSettingIsRegistered(SettingKey), Or(Not(CMMHasScriptedGui(SettingKey)), CMMGuiIsShown(Concatenate(SettingKey, '_on_changed')))))"
}

}

# Args: SettingKey (string)
# Returns: int32
macro = {
    description = "Slider thumb visual index on 0-100 scale for one setting"
    definition = "CMMSliderThumbIndex(SettingKey)"
    replace_with = "FixedPointToInt(Add_CFixedPoint(Multiply_CFixedPoint(Divide_CFixedPoint(CMMSliderNormalizedValue(SettingKey), CMMSliderNormalizedMax(SettingKey)), '(CFixedPoint)100'), '(CFixedPoint)0.500'))"
}

# Args: ThumbIndex (int32), Offset (int32 literal)
# Returns: boolean
macro = {
    description = "True when hovered slider index equals thumb index plus offset"
    definition = "CMMSliderHoverMatch(ThumbIndex,Offset)"
    replace_with = "GetVariableSystem.HasValue('cmm_slider_hover_idx', ToString_int32(Add_int32(ThumbIndex, Offset)))"
}

# Args: ThumbIndex (int32), ScopeKey (string)
# Returns: boolean
macro = {
    description = "True when the hovered overlay index is within 5 of the thumb and the scope key matches"
    definition = "CMMSliderThumbHover(ThumbIndex,ScopeKey)"
    replace_with = "And(GetVariableSystem.HasValue('cmm_slider_hover_key', ScopeKey), Or(CMMSliderHoverMatch(ThumbIndex, '(int32)-5'), Or(CMMSliderHoverMatch(ThumbIndex, '(int32)-4'), Or(CMMSliderHoverMatch(ThumbIndex, '(int32)-3'), Or(CMMSliderHoverMatch(ThumbIndex, '(int32)-2'), Or(CMMSliderHoverMatch(ThumbIndex, '(int32)-1'), Or(CMMSliderHoverMatch(ThumbIndex, '(int32)0'), Or(CMMSliderHoverMatch(ThumbIndex, '(int32)1'), Or(CMMSliderHoverMatch(ThumbIndex, '(int32)2'), Or(CMMSliderHoverMatch(ThumbIndex, '(int32)3'), Or(CMMSliderHoverMatch(ThumbIndex, '(int32)4'), CMMSliderHoverMatch(ThumbIndex, '(int32)5'))))))))))))"
}
