#noinspection ALL
# True when host-only tools are enabled and this country counts as host.
# Scope: country
# Args: none
is_host = {
	AND = {
		has_global_variable_map = cmm
		is_key_in_global_variable_map = { name = cmm target = flag:cmm_core__enable_host_only_tools }
		"global_variable_map(cmm|flag:cmm_core__enable_host_only_tools)" >= 1
		OR = {
			AND = {
				has_multiple_players = no
				is_human = yes
			}
			AND = {
				has_multiple_players = yes
				has_global_variable = cmm_core_host_country
				this ?= global_var:cmm_core_host_country
			}
		}
	}
}
