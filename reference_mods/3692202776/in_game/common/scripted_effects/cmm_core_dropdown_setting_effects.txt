# Register a dropdown setting for a mod (idempotent by mod_id + setting_id).
# Scope: country
# Args:
# - mod_id (key): Owner mod id.
# - setting_id (key): Setting id unique within mod.
# - tab_id (key): Owner tab id within mod.
# - group_id (key): Owner group id within mod.
# - default_index (number): Initial selected option ordinal (`1..option_count`).
# - option_count (number): Number of options (`>= 1`; numbered `1..option_count`).
cmm_register_dropdown_setting = {
	_cmm_register_dropdown_setting_internal = {
		mod_id = $mod_id$ setting_id = $setting_id$ tab_id = $tab_id$ group_id = $group_id$
		default_index = $default_index$ option_count = $option_count$
		is_global = 0 vt = variable
	}
}

# Register a global dropdown setting for a mod.
# Scope: country
# Args:
# - mod_id (key): Owner mod id.
# - setting_id (key): Setting id unique within mod.
# - tab_id (key): Owner tab id within mod.
# - group_id (key): Owner group id within mod.
# - default_index (number): Initial global selected option ordinal (`1..option_count`).
# - option_count (number): Number of options (`>= 1`; numbered `1..option_count`).
cmm_register_global_dropdown_setting = {
	_cmm_register_dropdown_setting_internal = {
		mod_id = $mod_id$ setting_id = $setting_id$ tab_id = $tab_id$ group_id = $group_id$
		default_index = $default_index$ option_count = $option_count$
		is_global = 1 vt = global_variable
	}
}

# Register shared dropdown-setting metadata and storage for a mod (idempotent by mod_id + setting_id).
# Scope: country
# Args: mod_id (key), setting_id (key), tab_id (key), group_id (key), default_index (number),
#   option_count (number), is_global (1/0), vt (variable/global_variable)
_cmm_register_dropdown_setting_internal = {
	add_to_variable_map = { name = cmm_type key = flag:$mod_id$__$setting_id$ value = 2 }
	add_to_variable_map = { name = cmm_is_global key = flag:$mod_id$__$setting_id$ value = $is_global$ }
	add_to_variable_map = { name = cmm_default key = flag:$mod_id$__$setting_id$ value = $default_index$ }

	set_variable = { name = cmm_tmp_dd_count value = $option_count$ }
	if = {
		limit = { var:cmm_tmp_dd_count < 1 }
		set_variable = { name = cmm_tmp_dd_count value = 1 }
	}
	add_to_variable_map = { name = cmm_dropdown_count key = flag:$mod_id$__$setting_id$ value = var:cmm_tmp_dd_count }
	add_to_variable_map = { name = cmm_dropdown_last_index key = flag:$mod_id$__$setting_id$ value = var:cmm_tmp_dd_count }
	remove_variable = cmm_tmp_dd_count

	_cmm_register_setting_metadata = {
		mod_id = $mod_id$
		setting_id = $setting_id$
		tab_id = $tab_id$
		group_id = $group_id$
	}

	if = {
		limit = { NOT = { is_key_in_$vt$_map = { name = cmm target = flag:$mod_id$__$setting_id$ } } }
		add_to_$vt$_map = { name = cmm key = flag:$mod_id$__$setting_id$ value = $default_index$ }
	}

	set_local_variable = { name = cmm_tmp_flag value = flag:$mod_id$__$setting_id$ }
	if = {
		limit = { $is_global$ >= 1 }
		set_local_variable = { name = cmm_tmp_val value = "global_variable_map(cmm|local_var:cmm_tmp_flag)" }
	}
	else = {
		set_local_variable = { name = cmm_tmp_val value = "variable_map(cmm|local_var:cmm_tmp_flag)" }
	}
	set_local_variable = { name = cmm_tmp_last value = "variable_map(cmm_dropdown_last_index|local_var:cmm_tmp_flag)" }

	if = {
		limit = { local_var:cmm_tmp_val < 1 }
		remove_from_$vt$_map = { name = cmm key = flag:$mod_id$__$setting_id$ }
		add_to_$vt$_map = { name = cmm key = flag:$mod_id$__$setting_id$ value = 1 }
	}

	if = {
		limit = { local_var:cmm_tmp_val > local_var:cmm_tmp_last }
		remove_from_$vt$_map = { name = cmm key = flag:$mod_id$__$setting_id$ }
		add_to_$vt$_map = { name = cmm key = flag:$mod_id$__$setting_id$ value = local_var:cmm_tmp_last }
	}

	# Benign warning suppression
	if = {
		limit = { always = no }
		cmf_suppress = { v = cmm_tmp_dd_count }
		cmf_suppress = { v = cmm_tmp_flag }
		cmf_suppress = { v = cmm_tmp_val }
		cmf_suppress = { v = cmm_tmp_last }
	}
}

# Apply the pending dropdown selected option ordinal to a setting and clear the marker.
# Scope: country
# Args:
# - setting (key): Composite setting key `<mod_id>__<setting_id>`.
_cmm_apply_dropdown_change = {
	if = {
		limit = { has_variable = cmm_dropdown_index_to_apply }

		set_local_variable = { name = cmm_tmp_flag value = flag:$setting$ }
		set_local_variable = { name = cmm_tmp_last value = "variable_map(cmm_dropdown_last_index|local_var:cmm_tmp_flag)" }

		set_variable = { name = cmm_tmp_new_val value = var:cmm_dropdown_index_to_apply }
		if = {
			limit = { var:cmm_tmp_new_val < 1 }
			set_variable = { name = cmm_tmp_new_val value = 1 }
		}
		if = {
			limit = { var:cmm_tmp_new_val > local_var:cmm_tmp_last }
			set_variable = { name = cmm_tmp_new_val value = local_var:cmm_tmp_last }
		}

		if = {
			limit = { "variable_map(cmm_is_global|local_var:cmm_tmp_flag)" >= 1 }
			remove_from_global_variable_map = { name = cmm key = flag:$setting$ }
			add_to_global_variable_map = { name = cmm key = flag:$setting$ value = var:cmm_tmp_new_val }
		}
		else = {
			remove_from_variable_map = { name = cmm key = flag:$setting$ }
			add_to_variable_map = { name = cmm key = flag:$setting$ value = var:cmm_tmp_new_val }
		}

		remove_variable = cmm_tmp_new_val
		remove_variable = cmm_dropdown_index_to_apply
	}

	# Benign warning suppression
	if = {
		limit = { always = no }
		cmf_suppress = { v = cmm_dropdown_index_to_apply }
		cmf_suppress = { v = cmm_tmp_new_val }
		cmf_suppress = { v = cmm_tmp_flag }
		cmf_suppress = { v = cmm_tmp_last }
	}
}
