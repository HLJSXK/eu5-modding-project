# Pre-initialize all CMM variable maps so quoted scope links never access a nonexistent map.
# Idempotent: add_to_variable_map is a no-op for existing keys.
# Scope: country
# Args: none
_cmm_init_variable_maps = {
	add_to_variable_map = { name = cmm key = flag:cmm_init value = 0 }
	add_to_variable_map = { name = cmm_type key = flag:cmm_init value = 0 }
	add_to_variable_map = { name = cmm_is_global key = flag:cmm_init value = 0 }
	add_to_variable_map = { name = cmm_default key = flag:cmm_init value = 0 }
	add_to_variable_map = { name = cmm_min key = flag:cmm_init value = 0 }
	add_to_variable_map = { name = cmm_max key = flag:cmm_init value = 0 }
	add_to_variable_map = { name = cmm_step key = flag:cmm_init value = 0 }
	add_to_variable_map = { name = cmm_dropdown_count key = flag:cmm_init value = 0 }
	add_to_variable_map = { name = cmm_dropdown_last_index key = flag:cmm_init value = 0 }
	add_to_variable_map = { name = cmm_slider_actual_step_count key = flag:cmm_init value = 0 }
	add_to_variable_map = { name = cmm_slider_actual_last_index key = flag:cmm_init value = 0 }
	add_to_variable_map = { name = cmm_text_char_limit key = flag:cmm_init value = 0 }
	add_to_variable_map = { name = cmm_text_quote key = flag:cmm_init value = 0 }
	add_to_variable_map = { name = cmm_list_count key = flag:cmm_init value = 0 }
	add_to_variable_map = { name = cmm_list_last_index key = flag:cmm_init value = 0 }
	add_to_variable_map = { name = cmm_list_is_ordered key = flag:cmm_init value = 0 }
	add_to_variable_map = { name = cmm_has_sgui key = flag:cmm_init value = 0 }
	add_to_variable_map = { name = cmm_no_reset key = flag:cmm_init value = 0 }
	add_to_variable_map = { name = cmm_list_field_disabled key = flag:cmm_init value = 0 }
	add_to_global_variable_map = { name = cmm key = flag:cmm_init value = 0 }

	# Benign warning suppression
	if = {
		limit = {
			always = no
			has_variable_map = cmm
			has_variable_map = cmm_type
			has_variable_map = cmm_is_global
			has_variable_map = cmm_default
			has_variable_map = cmm_min
			has_variable_map = cmm_max
			has_variable_map = cmm_step
			has_variable_map = cmm_dropdown_count
			has_variable_map = cmm_dropdown_last_index
			has_variable_map = cmm_slider_actual_step_count
			has_variable_map = cmm_slider_actual_last_index
			has_variable_map = cmm_text_char_limit
			has_variable_map = cmm_text_quote
			has_variable_map = cmm_list_count
			has_variable_map = cmm_list_last_index
			has_variable_map = cmm_list_is_ordered
			has_variable_map = cmm_has_sgui
			has_variable_map = cmm_no_reset
			has_variable_map = cmm_list_field_disabled
			has_global_variable_map = cmm
		}
		cmf_suppress = { v = cmm_init }
	}
}

# Register a mod (idempotent by mod_id).
# Scope: country
# Args:
# - mod_id (key): Mod identifier; derives `<mod_id>_name` and `<mod_id>_desc`.
_cmm_register_mod = {
	_cmm_init_variable_maps = yes
	set_variable = { name = $mod_id$_name value = flag:$mod_id$_name }
	set_variable = { name = $mod_id$_desc value = flag:$mod_id$_desc }

	# First-time registration path for this mod id.
	if = {
		limit = { NOT = { has_variable = cmm_registered_mod_$mod_id$ } }
		set_variable = { name = cmm_registered_mod_$mod_id$ value = yes }
		# Global mod id list; used by left-panel datamodel iteration.
		add_to_variable_list = {
			name = cmm_registered_mod_ids
			target = flag:$mod_id$
		}
	}

	# Benign warning suppression
	if = {
		limit = { always = no }
		cmf_suppress = { v = $mod_id$ }
		cmf_suppress = { v = $mod_id$_name }
		cmf_suppress = { v = $mod_id$_desc }
		cmf_suppress = { v = cmm_registered_mod_ids }
		cmf_suppress = { v = cmm_registered_mod_$mod_id$ }
	}
}

# Register a tab for a mod and implicitly register the owner mod (idempotent by mod_id + tab_id).
# Scope: country
# Args:
# - mod_id (key): Owner mod id.
# - tab_id (key): Tab id unique within mod.
_cmm_register_tab = {
	_cmm_register_mod = {
		mod_id = $mod_id$
	}

	# First-time registration path for this tab key.
	if = {
		limit = { NOT = { has_variable = cmm_registered_tab_$mod_id$__$tab_id$ } }
		set_variable = { name = cmm_registered_tab_$mod_id$__$tab_id$ value = yes }
		set_variable = { name = cmm_tab_owner_mod_id_$mod_id$__$tab_id$ value = flag:$mod_id$ }
		set_variable = { name = $mod_id$__$tab_id$_name value = flag:$mod_id$__$tab_id$_name }

		# Global tab key list; used by right-panel tab datamodel iteration.
		add_to_variable_list = {
			name = cmm_registered_tab_keys
			target = flag:$mod_id$__$tab_id$
		}

		# First registered tab becomes default selected tab for this mod.
		if = {
			limit = { NOT = { has_variable = cmm_mod_default_tab_$mod_id$ } }
			set_variable = { name = cmm_mod_default_tab_$mod_id$ value = flag:$mod_id$__$tab_id$ }
		}
	}

	# Benign warning suppression
	if = {
		limit = { always = no }
		cmf_suppress = { v = $mod_id$__$tab_id$ }
		cmf_suppress = { v = $mod_id$__$tab_id$_name }
		cmf_suppress = { v = cmm_registered_tab_keys }
		cmf_suppress = { v = cmm_registered_tab_$mod_id$__$tab_id$ }
		cmf_suppress = { v = cmm_mod_default_tab_$mod_id$ }
		cmf_suppress = { v = cmm_tab_owner_mod_id_$mod_id$__$tab_id$ }
	}
}

		set_variable = { name = cmm_group_owner_mod_id_$mod_id$__$tab_id$__$group_id$ value = flag:$mod_id$ }
		set_variable = { name = cmm_group_owner_tab_key_$mod_id$__$tab_id$__$group_id$ value = flag:$mod_id$__$tab_id$ }
		set_variable = { name = $mod_id$__$tab_id$__$group_id$_name value = flag:$mod_id$__$tab_id$__$group_id$_name }
		set_variable = { name = $mod_id$__$tab_id$__$group_id$_desc value = flag:$mod_id$__$tab_id$__$group_id$_desc }

		# Global group key list; used by settings-panel group datamodel iteration.
		add_to_variable_list = {
			name = cmm_registered_group_keys

		cmf_suppress = { v = $mod_id$__$tab_id$ }
		cmf_suppress = { v = cmm_registered_group_keys }
		cmf_suppress = { v = cmm_registered_group_$mod_id$__$tab_id$__$group_id$ }
		cmf_suppress = { v = cmm_group_owner_mod_id_$mod_id$__$tab_id$__$group_id$ }
		cmf_suppress = { v = cmm_group_owner_tab_key_$mod_id$__$tab_id$__$group_id$ }

}

# Register shared metadata for a setting key and implicitly register the owner group/tab/mod (idempotent by mod_id + setting_id).
# Scope: country
# Args:
# - mod_id (key): Owner mod id.
# - setting_id (key): Setting id unique within mod.
# - tab_id (key): Owner tab id within mod.
# - group_id (key): Owner group id within tab.
_cmm_register_setting_metadata = {
	_cmm_register_group = {
		mod_id = $mod_id$
		tab_id = $tab_id$
		group_id = $group_id$
	}

	# First-time registration path for this setting key.
	if = {
		limit = { NOT = { has_variable = cmm_registered_setting_$mod_id$__$setting_id$ } }
		set_variable = { name = cmm_registered_setting_$mod_id$__$setting_id$ value = yes }
		set_variable = { name = cmm_setting_owner_mod_id_$mod_id$__$setting_id$ value = flag:$mod_id$ }
		set_variable = { name = cmm_setting_owner_tab_key_$mod_id$__$setting_id$ value = flag:$mod_id$__$tab_id$ }
		set_variable = { name = cmm_setting_owner_group_key_$mod_id$__$setting_id$ value = flag:$mod_id$__$tab_id$__$group_id$ }
		set_variable = { name = $mod_id$__$setting_id$_name value = flag:$mod_id$__$setting_id$_name }
		set_variable = { name = $mod_id$__$setting_id$_desc value = flag:$mod_id$__$setting_id$_desc }

		# Global setting key list; used by right-panel datamodel iteration.
		add_to_variable_list = {
			name = cmm_registered_setting_keys
			target = flag:$mod_id$__$setting_id$
		}

		# Per-group setting key list; used by group-scoped setting rendering.
		add_to_variable_list = {
			name = cmm_group_setting_keys_$mod_id$__$tab_id$__$group_id$
			target = flag:$mod_id$__$setting_id$
		}

		# Per-tab count; used for "no settings in this tab" UI state.
		if = {
			limit = { NOT = { has_variable = cmm_tab_setting_count_$mod_id$__$tab_id$ } }
			set_variable = { name = cmm_tab_setting_count_$mod_id$__$tab_id$ value = 0 }
		}
		change_variable = { name = cmm_tab_setting_count_$mod_id$__$tab_id$ add = 1 }
	}

	# Benign warning suppression
	if = {
		limit = { always = no }
		cmf_suppress = { v = $mod_id$__$setting_id$ }
		cmf_suppress = { v = $mod_id$__$tab_id$ }
		cmf_suppress = { v = $mod_id$__$tab_id$__$group_id$ }
		cmf_suppress = { v = $mod_id$__$setting_id$_name }
		cmf_suppress = { v = $mod_id$__$setting_id$_desc }
		cmf_suppress = { v = $mod_id$__$tab_id$__$group_id$_name }
		cmf_suppress = { v = cmm_registered_setting_keys }
		cmf_suppress = { v = cmm_registered_setting_$mod_id$__$setting_id$ }
		cmf_suppress = { v = cmm_setting_owner_mod_id_$mod_id$__$setting_id$ }
		cmf_suppress = { v = cmm_setting_owner_tab_key_$mod_id$__$setting_id$ }
		cmf_suppress = { v = cmm_setting_owner_group_key_$mod_id$__$setting_id$ }
		cmf_suppress = { v = cmm_group_setting_keys_$mod_id$__$tab_id$__$group_id$ }
		cmf_suppress = { v = cmm_tab_setting_count_$mod_id$__$tab_id$ }
	}
}

# Register a scripted GUI for a setting (enables is_shown/is_valid checks).
# The scripted GUI must be named `<mod_id>__<setting_id>_on_changed`.
# Call after the setting registration effect.
# Scope: country
# Args:
# - mod_id (key): Owner mod id.
# - setting_id (key): Setting id.
cmm_add_scripted_gui = {
	add_to_variable_map = { name = cmm_has_sgui key = flag:$mod_id$__$setting_id$ value = 1 }
}
