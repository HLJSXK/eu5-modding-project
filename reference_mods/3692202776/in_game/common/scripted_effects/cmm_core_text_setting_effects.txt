# Register a text setting for a mod (idempotent by mod_id + setting_id).
# Scope: country
# Args:
# - mod_id (key): Owner mod id.
# - setting_id (key): Setting id unique within mod.
# - tab_id (key): Owner tab id within mod.
# - group_id (key): Owner group id within mod.
# - character_limit (number): Maximum editbox character count; clamped to at least `1`.
# - quote_text (1/0): Wrap submitted text in double quotes before passing it to the effect callback.
cmm_register_text_setting = {
	add_to_variable_map = { name = cmm_type key = flag:$mod_id$__$setting_id$ value = 5 }
	add_to_variable_map = { name = cmm_is_global key = flag:$mod_id$__$setting_id$ value = 0 }
	add_to_variable_map = { name = cmm_text_quote key = flag:$mod_id$__$setting_id$ value = $quote_text$ }

	set_variable = { name = cmm_tmp_char_limit value = $character_limit$ }
	if = {
		limit = { var:cmm_tmp_char_limit < 1 }
		set_variable = { name = cmm_tmp_char_limit value = 1 }
	}
	add_to_variable_map = { name = cmm_text_char_limit key = flag:$mod_id$__$setting_id$ value = var:cmm_tmp_char_limit }
	remove_variable = cmm_tmp_char_limit

	_cmm_register_setting_metadata = {
		mod_id = $mod_id$
		setting_id = $setting_id$
		tab_id = $tab_id$
		group_id = $group_id$
	}

	# Benign warning suppression
	if = {
		limit = { always = no }
		cmf_suppress = { v = cmm_tmp_char_limit }
	}
}
