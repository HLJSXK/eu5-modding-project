# Register a slider setting for a mod (idempotent by mod_id + setting_id).
# Scope: country
# Args:
# - mod_id (key): Owner mod id.
# - setting_id (key): Setting id unique within mod.
# - tab_id (key): Owner tab id within mod.
# - group_id (key): Owner group id within mod.
# - default_value (number): Initial value; applied when the setting does not exist yet.
# - min_value (number): Lower clamp bound.
# - max_value (number): Upper clamp bound.
# - step_value (number): Logical slider step size.
cmm_register_slider_setting = {
	_cmm_register_slider_setting_internal = {
		mod_id = $mod_id$
		setting_id = $setting_id$
		tab_id = $tab_id$
		group_id = $group_id$
		default_value = $default_value$
		min_value = $min_value$
		max_value = $max_value$
		step_value = $step_value$
		is_global = 0
		vt = variable
	}
}

# Register a global slider setting for a mod.
# Scope: country
# Args:
# - mod_id (key): Owner mod id.
# - setting_id (key): Setting id unique within mod.
# - tab_id (key): Owner tab id within mod.
# - group_id (key): Owner group id within mod.
# - default_value (number): Initial global value when missing.
# - min_value (number): Lower clamp bound.
# - max_value (number): Upper clamp bound.
# - step_value (number): Logical slider step size.
cmm_register_global_slider_setting = {
	_cmm_register_slider_setting_internal = {
		mod_id = $mod_id$
		setting_id = $setting_id$
		tab_id = $tab_id$
		group_id = $group_id$
		default_value = $default_value$
		min_value = $min_value$
		max_value = $max_value$
		step_value = $step_value$
		is_global = 1
		vt = global_variable
	}
}

# Internal: shared registration for slider settings.
# Scope: country
# Args: mod_id (key), setting_id (key), tab_id (key), group_id (key), default_value (number), min_value (number), max_value (number), step_value (number), is_global (1/0), vt (variable/global_variable)
_cmm_register_slider_setting_internal = {
	add_to_variable_map = { name = cmm_type key = flag:$mod_id$__$setting_id$ value = 4 }
	add_to_variable_map = { name = cmm_is_global key = flag:$mod_id$__$setting_id$ value = $is_global$ }
	add_to_variable_map = { name = cmm_default key = flag:$mod_id$__$setting_id$ value = $default_value$ }
	add_to_variable_map = { name = cmm_min key = flag:$mod_id$__$setting_id$ value = $min_value$ }
	add_to_variable_map = { name = cmm_max key = flag:$mod_id$__$setting_id$ value = $max_value$ }
	add_to_variable_map = { name = cmm_step key = flag:$mod_id$__$setting_id$ value = $step_value$ }

	set_variable = { name = cmm_tmp_step_count value = $max_value$ }
	change_variable = { name = cmm_tmp_step_count subtract = $min_value$ }
	change_variable = { name = cmm_tmp_step_count divide = $step_value$ }
	round_variable = {
		name = cmm_tmp_step_count
		nearest = 1
	}
	change_variable = { name = cmm_tmp_step_count add = 1 }

	if = {
		limit = { var:cmm_tmp_step_count < 2 }
		set_variable = { name = cmm_tmp_step_count value = 2 }
	}

	add_to_variable_map = { name = cmm_slider_actual_step_count key = flag:$mod_id$__$setting_id$ value = var:cmm_tmp_step_count }

	set_variable = { name = cmm_tmp_last_index value = var:cmm_tmp_step_count }
	change_variable = { name = cmm_tmp_last_index add = -1 }
	add_to_variable_map = { name = cmm_slider_actual_last_index key = flag:$mod_id$__$setting_id$ value = var:cmm_tmp_last_index }

	remove_variable = cmm_tmp_step_count
	remove_variable = cmm_tmp_last_index

	_cmm_register_setting_metadata = {
		mod_id = $mod_id$
		setting_id = $setting_id$
		tab_id = $tab_id$
		group_id = $group_id$
	}

	if = {
		limit = { NOT = { is_key_in_$vt$_map = { name = cmm target = flag:$mod_id$__$setting_id$ } } }
		add_to_$vt$_map = { name = cmm key = flag:$mod_id$__$setting_id$ value = $default_value$ }
	}

	set_local_variable = { name = cmm_tmp_flag value = flag:$mod_id$__$setting_id$ }

	if = {
		limit = { "variable_map(cmm_is_global|local_var:cmm_tmp_flag)" >= 1 }
		set_local_variable = { name = cmm_tmp_val value = "global_variable_map(cmm|local_var:cmm_tmp_flag)" }
	}
	else = {
		set_local_variable = { name = cmm_tmp_val value = "variable_map(cmm|local_var:cmm_tmp_flag)" }
	}

	if = {
		limit = { local_var:cmm_tmp_val < $min_value$ }
		remove_from_$vt$_map = { name = cmm key = flag:$mod_id$__$setting_id$ }
		add_to_$vt$_map = { name = cmm key = flag:$mod_id$__$setting_id$ value = $min_value$ }
	}

	if = {
		limit = { local_var:cmm_tmp_val > $max_value$ }
		remove_from_$vt$_map = { name = cmm key = flag:$mod_id$__$setting_id$ }
		add_to_$vt$_map = { name = cmm key = flag:$mod_id$__$setting_id$ value = $max_value$ }
	}

	_cmm_quantize_slider_setting = {
		setting = $mod_id$__$setting_id$
	}

	# Benign warning suppression
	if = {
		limit = { always = no }
		cmf_suppress = { v = cmm_tmp_step_count }
		cmf_suppress = { v = cmm_tmp_last_index }
		cmf_suppress = { v = cmm_tmp_flag }
		cmf_suppress = { v = cmm_tmp_val }
	}
}

# Quantize a slider setting value to the nearest configured step.
# Scope: country
# Args:
# - setting (key): Composite setting key `<mod_id>__<setting_id>`.
_cmm_quantize_slider_setting = {
	set_local_variable = { name = cmm_tmp_flag value = flag:$setting$ }
	if = {
		limit = { "variable_map(cmm_is_global|local_var:cmm_tmp_flag)" >= 1 }
		_cmm_quantize_slider_value = { setting = $setting$ vt = global_variable }
	}
	else = {
		_cmm_quantize_slider_value = { setting = $setting$ vt = variable }
	}

	# Benign warning suppression
	if = {
		limit = { always = no }
		cmf_suppress = { v = cmm_tmp_flag }
	}
}

# Internal: quantize a slider value using the appropriate variable type.
# Scope: country
# Args: setting (key), vt (variable/global_variable)
_cmm_quantize_slider_value = {
	set_local_variable = { name = cmm_tmp_min value = "variable_map(cmm_min|local_var:cmm_tmp_flag)" }
	set_local_variable = { name = cmm_tmp_max value = "variable_map(cmm_max|local_var:cmm_tmp_flag)" }
	set_local_variable = { name = cmm_tmp_step value = "variable_map(cmm_step|local_var:cmm_tmp_flag)" }

	if = {
		limit = { "variable_map(cmm_is_global|local_var:cmm_tmp_flag)" >= 1 }
		set_variable = { name = cmm_tmp_slider_value value = "global_variable_map(cmm|local_var:cmm_tmp_flag)" }
	}
	else = {
		set_variable = { name = cmm_tmp_slider_value value = "variable_map(cmm|local_var:cmm_tmp_flag)" }
	}
	change_variable = { name = cmm_tmp_slider_value subtract = local_var:cmm_tmp_min }
	change_variable = { name = cmm_tmp_slider_value divide = local_var:cmm_tmp_step }
	round_variable = {
		name = cmm_tmp_slider_value
		nearest = 1
	}
	change_variable = { name = cmm_tmp_slider_value multiply = local_var:cmm_tmp_step }
	change_variable = { name = cmm_tmp_slider_value add = local_var:cmm_tmp_min }

	if = {
		limit = { var:cmm_tmp_slider_value < local_var:cmm_tmp_min }
		set_variable = { name = cmm_tmp_slider_value value = local_var:cmm_tmp_min }
	}

	if = {
		limit = { var:cmm_tmp_slider_value > local_var:cmm_tmp_max }
		set_variable = { name = cmm_tmp_slider_value value = local_var:cmm_tmp_max }
	}

	remove_from_$vt$_map = { name = cmm key = flag:$setting$ }
	add_to_$vt$_map = { name = cmm key = flag:$setting$ value = var:cmm_tmp_slider_value }
	remove_variable = cmm_tmp_slider_value

	# Benign warning suppression
	if = {
		limit = { always = no }
		cmf_suppress = { v = cmm_tmp_min }
		cmf_suppress = { v = cmm_tmp_max }
		cmf_suppress = { v = cmm_tmp_step }
	}
}

# Set a slider setting directly from the pending visual-track index and clamp to range.
# Scope: country
# Args:
# - setting (key): Composite setting key `<mod_id>__<setting_id>`.
_cmm_set_slider_setting_from_visual_index = {
	set_local_variable = { name = cmm_tmp_flag value = flag:$setting$ }
	set_local_variable = { name = cmm_tmp_actual_last value = "variable_map(cmm_slider_actual_last_index|local_var:cmm_tmp_flag)" }
	set_local_variable = { name = cmm_tmp_step value = "variable_map(cmm_step|local_var:cmm_tmp_flag)" }
	set_local_variable = { name = cmm_tmp_min value = "variable_map(cmm_min|local_var:cmm_tmp_flag)" }
	set_local_variable = { name = cmm_tmp_max value = "variable_map(cmm_max|local_var:cmm_tmp_flag)" }

	set_variable = { name = cmm_tmp_slider_actual_index value = var:cmm_slider_index_to_apply }
	change_variable = { name = cmm_tmp_slider_actual_index multiply = local_var:cmm_tmp_actual_last }
	change_variable = { name = cmm_tmp_slider_actual_index divide = 100 }
	round_variable = {
		name = cmm_tmp_slider_actual_index
		nearest = 1
	}

	set_variable = { name = cmm_tmp_slider_value value = local_var:cmm_tmp_step }
	change_variable = { name = cmm_tmp_slider_value multiply = var:cmm_tmp_slider_actual_index }
	change_variable = { name = cmm_tmp_slider_value add = local_var:cmm_tmp_min }

	if = {
		limit = { var:cmm_tmp_slider_value < local_var:cmm_tmp_min }
		set_variable = { name = cmm_tmp_slider_value value = local_var:cmm_tmp_min }
	}

	if = {
		limit = { var:cmm_tmp_slider_value > local_var:cmm_tmp_max }
		set_variable = { name = cmm_tmp_slider_value value = local_var:cmm_tmp_max }
	}

	if = {
		limit = { "variable_map(cmm_is_global|local_var:cmm_tmp_flag)" >= 1 }
		remove_from_global_variable_map = { name = cmm key = flag:$setting$ }
		add_to_global_variable_map = { name = cmm key = flag:$setting$ value = var:cmm_tmp_slider_value }
	}
	else = {
		remove_from_variable_map = { name = cmm key = flag:$setting$ }
		add_to_variable_map = { name = cmm key = flag:$setting$ value = var:cmm_tmp_slider_value }
	}

	remove_variable = cmm_tmp_slider_actual_index
	remove_variable = cmm_tmp_slider_value

	# Benign warning suppression
	if = {
		limit = { always = no }
		cmf_suppress = { v = cmm_tmp_flag }
		cmf_suppress = { v = cmm_tmp_actual_last }
		cmf_suppress = { v = cmm_tmp_step }
		cmf_suppress = { v = cmm_tmp_min }
		cmf_suppress = { v = cmm_tmp_max }
	}
}

# Apply the pending slider track selection or queued slider button step change.
# Scope: country
# Args:
# - setting (key): Composite setting key `<mod_id>__<setting_id>`.
_cmm_apply_slider_change = {
	if = {
		limit = { has_variable = cmm_slider_index_to_apply }

		_cmm_set_slider_setting_from_visual_index = {
			setting = $setting$
		}

		remove_variable = cmm_slider_index_to_apply

		if = {
			limit = { has_variable = cmm_numeric_step_delta_to_apply }
			remove_variable = cmm_numeric_step_delta_to_apply
		}

		if = {
			limit = { has_variable = cmm_numeric_change_to_apply }
			remove_variable = cmm_numeric_change_to_apply
		}
	}
	else = {
		_cmm_apply_numeric_change = {
			setting = $setting$
		}
	}
}
