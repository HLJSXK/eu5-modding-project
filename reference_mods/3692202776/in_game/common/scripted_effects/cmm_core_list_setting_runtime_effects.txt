# Rebuild one ordered list after moving the marked row to `local_var:cmm_list_target_index`.
# Scope: country
# Args:
# - setting (key): Composite setting key `<mod_id>__<setting_id>`.
# - vt (variable/global_variable): Variable type token.
# - lt (in/in_global): List iteration token.
_cmm_reorder_list_setting = {
	clear_variable_list = cmm_list_rebuild_temp
	set_local_variable = { name = cmm_list_source_index value = 0 }
	set_local_variable = { name = cmm_list_found_moving_item value = 0 }
	set_local_variable = { name = cmm_list_marked_position value = var:cmm_list_position_to_apply }

	every_$lt$_list = {
		variable = cmm_list_items_$setting$
		save_scope_as = cmm_list_current_item

		root = {
			add_to_variable_list = {
				name = cmm_list_rebuild_temp
				target = scope:cmm_list_current_item
			}
		}

		if = {
			limit = { local_var:cmm_list_source_index = local_var:cmm_list_marked_position }
			save_scope_as = cmm_list_moving_item
			set_local_variable = { name = cmm_list_found_moving_item value = 1 }
		}

		change_local_variable = { name = cmm_list_source_index add = 1 }
	}

	if = {
		limit = { local_var:cmm_list_found_moving_item = 1 }

		clear_$vt$_list = cmm_list_items_$setting$

		set_local_variable = { name = cmm_list_reinsert_index value = 0 }
		set_local_variable = { name = cmm_list_inserted_moving_item value = 0 }

		every_in_list = {
			variable = cmm_list_rebuild_temp
			save_scope_as = cmm_list_current_item

			if = {
				limit = { NOT = { this = scope:cmm_list_moving_item } }
				if = {
					limit = {
						AND = {
							local_var:cmm_list_reinsert_index = local_var:cmm_list_target_index
							local_var:cmm_list_inserted_moving_item = 0
						}
					}
					root = {
						add_to_$vt$_list = {
							name = cmm_list_items_$setting$
							target = scope:cmm_list_moving_item
						}
					}
					set_local_variable = { name = cmm_list_inserted_moving_item value = 1 }
				}

				root = {
					add_to_$vt$_list = {
						name = cmm_list_items_$setting$
						target = scope:cmm_list_current_item
					}
				}
				change_local_variable = { name = cmm_list_reinsert_index add = 1 }
			}
		}

		if = {
			limit = { local_var:cmm_list_inserted_moving_item = 0 }
			add_to_$vt$_list = {
				name = cmm_list_items_$setting$
				target = scope:cmm_list_moving_item
			}
		}
	}

	clear_variable_list = cmm_list_rebuild_temp
}

# Apply the queued numeric markers to one ordered list numeric field slot for one item identity.
# Scope: country
# Args:
# - setting (key): Composite setting key `<mod_id>__<setting_id>`.
# - item (number): Stable item identity ordinal (`1..20`).
# - slot (number): Field slot index (`1..5`).
# - vt (variable/global_variable): Variable type token.
_cmm_list_apply_numeric_field_slot = {
	set_local_variable = { name = cmm_tmp_disabled_fk value = flag:$setting$_i$item$_f$slot$ }
	if = {
		limit = { NOT = { is_key_in_variable_map = { name = cmm_list_field_disabled target = local_var:cmm_tmp_disabled_fk } } }
		if = {
			limit = { AND = { OR = { var:cmm_list_field_$slot$_type_$setting$ = 3 var:cmm_list_field_$slot$_type_$setting$ = 4 } has_variable = cmm_numeric_step_delta_to_apply } }
			set_local_variable = { name = cmm_tmp_fk value = flag:$setting$_i$item$_f$slot$ }
			if = {
				limit = { var:cmm_tmp_is_global_list >= 1 }
				set_variable = { name = cmm_tmp_numeric_value value = "global_variable_map(cmm|local_var:cmm_tmp_fk)" }
			}
			else = {
				set_variable = { name = cmm_tmp_numeric_value value = "variable_map(cmm|local_var:cmm_tmp_fk)" }
			}
			set_variable = { name = cmm_tmp_numeric_delta value = var:cmm_list_field_$slot$_step_$setting$ }
			change_variable = { name = cmm_tmp_numeric_delta multiply = var:cmm_numeric_step_delta_to_apply }
			change_variable = { name = cmm_tmp_numeric_value add = var:cmm_tmp_numeric_delta }

			if = {
				limit = { var:cmm_tmp_numeric_value > var:cmm_list_field_$slot$_max_$setting$ }
				set_variable = { name = cmm_tmp_numeric_value value = var:cmm_list_field_$slot$_max_$setting$ }
			}

			if = {
				limit = { var:cmm_tmp_numeric_value < var:cmm_list_field_$slot$_min_$setting$ }
				set_variable = { name = cmm_tmp_numeric_value value = var:cmm_list_field_$slot$_min_$setting$ }
			}

			remove_from_$vt$_map = { name = cmm key = flag:$setting$_i$item$_f$slot$ }
			add_to_$vt$_map = { name = cmm key = flag:$setting$_i$item$_f$slot$ value = var:cmm_tmp_numeric_value }
			remove_variable = cmm_tmp_numeric_value
			remove_variable = cmm_tmp_numeric_delta
		}
		else_if = {
			limit = { AND = { OR = { var:cmm_list_field_$slot$_type_$setting$ = 3 var:cmm_list_field_$slot$_type_$setting$ = 4 } has_variable = cmm_numeric_change_to_apply } }
			if = {
				limit = { var:cmm_numeric_change_to_apply > 0 }
				remove_from_$vt$_map = { name = cmm key = flag:$setting$_i$item$_f$slot$ }
				add_to_$vt$_map = { name = cmm key = flag:$setting$_i$item$_f$slot$ value = var:cmm_list_field_$slot$_max_$setting$ }
			}
			else_if = {
				limit = { var:cmm_numeric_change_to_apply < 0 }
				remove_from_$vt$_map = { name = cmm key = flag:$setting$_i$item$_f$slot$ }
				add_to_$vt$_map = { name = cmm key = flag:$setting$_i$item$_f$slot$ value = var:cmm_list_field_$slot$_min_$setting$ }
			}
		}
	}

	# Benign warning suppression
	if = {
		limit = { always = no }
		cmf_suppress = { v = cmm_tmp_fk }
		cmf_suppress = { v = cmm_tmp_disabled_fk }
		cmf_suppress = { v = cmm_list_field_disabled }
	}

# Internal: apply the pending list change using parameterized variable types.
# Called by cmm_apply_list_change after resolving the global/local branch once.
# Scope: country
# Args:
# - setting (key): Composite setting key `<mod_id>__<setting_id>`.
# - vt (variable/global_variable): Variable type token.
# - lt (in/in_global): List iteration token.
_cmm_apply_list_change_internal = {
	if = {
		limit = {
			AND = {
				has_variable = cmm_list_position_to_apply
				OR = {
					has_variable = cmm_list_action_to_apply
					has_variable = cmm_numeric_step_delta_to_apply
					has_variable = cmm_numeric_change_to_apply
					has_variable = cmm_slider_index_to_apply
				}
			}
		}
		if = {
			limit = { AND = { has_variable = cmm_list_action_to_apply var:cmm_setting_list_is_ordered_$setting$ = 1 var:cmm_list_action_to_apply = 1 var:cmm_list_position_to_apply > 0 } }
			set_local_variable = { name = cmm_list_target_index value = { value = var:cmm_list_position_to_apply subtract = 1 } }
			_cmm_reorder_list_setting = { setting = $setting$ vt = $vt$ lt = $lt$ }
		}
		else_if = {
			limit = { AND = { has_variable = cmm_list_action_to_apply var:cmm_setting_list_is_ordered_$setting$ = 1 var:cmm_list_action_to_apply = 2 var:cmm_list_position_to_apply < var:cmm_setting_list_last_index_$setting$ } }
			set_local_variable = { name = cmm_list_target_index value = { value = var:cmm_list_position_to_apply add = 1 } }
			_cmm_reorder_list_setting = { setting = $setting$ vt = $vt$ lt = $lt$ }
		}
		else_if = {
			limit = { AND = { has_variable = cmm_list_action_to_apply var:cmm_setting_list_is_ordered_$setting$ = 1 var:cmm_list_action_to_apply = 3 var:cmm_list_position_to_apply > 0 } }
			set_local_variable = { name = cmm_list_target_index value = 0 }
			_cmm_reorder_list_setting = { setting = $setting$ vt = $vt$ lt = $lt$ }
		}
		else_if = {
			limit = { AND = { has_variable = cmm_list_action_to_apply var:cmm_setting_list_is_ordered_$setting$ = 1 var:cmm_list_action_to_apply = 4 var:cmm_list_position_to_apply < var:cmm_setting_list_last_index_$setting$ } }
			set_local_variable = { name = cmm_list_target_index value = var:cmm_setting_list_last_index_$setting$ }
			_cmm_reorder_list_setting = { setting = $setting$ vt = $vt$ lt = $lt$ }
		}
		else_if = {
			limit = { AND = { has_variable = cmm_list_action_to_apply var:cmm_list_action_to_apply = 9 has_variable = cmm_list_field_slot_to_apply } }
			set_variable = { name = cmm_list_dispatch_action value = 1 }
			_cmm_list_apply_dispatch_to_marked_position = { setting = $setting$ lt = $lt$ vt = $vt$ }

			if = {
				limit = { has_variable = cmm_list_field_value_to_apply }
				_cmm_list_apply_field_value_to_all_items = { setting = $setting$ lt = $lt$ vt = $vt$ }
			}
		}
		else_if = {
			limit = { AND = { has_variable = cmm_list_action_to_apply var:cmm_list_action_to_apply = 10 has_variable = cmm_list_field_slot_to_apply has_variable = cmm_dropdown_index_to_apply } }
			set_variable = { name = cmm_list_field_value_to_apply value = var:cmm_dropdown_index_to_apply }
			_cmm_list_apply_field_value_to_all_items = { setting = $setting$ lt = $lt$ vt = $vt$ }
		}
		else_if = {
			limit = { AND = { has_variable = cmm_list_action_to_apply var:cmm_list_action_to_apply = 11 has_variable = cmm_list_field_slot_to_apply } }
			set_variable = { name = cmm_list_dispatch_action value = 2 }
			_cmm_list_apply_dispatch_to_marked_position = { setting = $setting$ lt = $lt$ vt = $vt$ }

			if = {
				limit = { has_variable = cmm_list_field_value_to_apply }
				_cmm_list_apply_field_value_to_all_items = { setting = $setting$ lt = $lt$ vt = $vt$ }
			}
		}
		else_if = {
			limit = { AND = { has_variable = cmm_list_action_to_apply var:cmm_list_action_to_apply = 12 has_variable = cmm_list_field_slot_to_apply } }
			set_variable = { name = cmm_list_dispatch_action value = 2 }
			_cmm_list_apply_dispatch_to_marked_position = { setting = $setting$ lt = $lt$ vt = $vt$ }

			if = {
				limit = { has_variable = cmm_list_field_value_to_apply }
				_cmm_list_apply_field_value_to_all_items = { setting = $setting$ lt = $lt$ vt = $vt$ }
			}
		}
		else_if = {
			limit = {
				AND = {
					has_variable = cmm_list_field_slot_to_apply
					has_variable = cmm_slider_index_to_apply
				}
			}
			set_variable = { name = cmm_list_dispatch_action value = 5 }
			_cmm_list_apply_dispatch_to_marked_position = { setting = $setting$ lt = $lt$ vt = $vt$ }
		}
		else_if = {
			limit = {
				AND = {
					has_variable = cmm_list_field_slot_to_apply
					OR = {
						has_variable = cmm_numeric_step_delta_to_apply
						has_variable = cmm_numeric_change_to_apply
					}
				}
			}
			set_variable = { name = cmm_list_dispatch_action value = 3 }
			_cmm_list_apply_dispatch_to_marked_position = { setting = $setting$ lt = $lt$ vt = $vt$ }
		}
		else_if = {
			limit = {
				AND = {
					has_variable = cmm_list_action_to_apply
					OR = { var:cmm_list_action_to_apply = 5 var:cmm_list_action_to_apply = 6 var:cmm_list_action_to_apply = 7 var:cmm_list_action_to_apply = 8 }
					has_variable = cmm_list_field_slot_to_apply
				}
			}
			set_variable = { name = cmm_list_dispatch_action value = 4 }
			_cmm_list_apply_dispatch_to_marked_position = { setting = $setting$ lt = $lt$ vt = $vt$ }
		}
	}
}

# Apply the pending list reorder (ordered lists only) or field action and clear its markers.
# Scope: country
# Args:
# - setting (key): Composite setting key `<mod_id>__<setting_id>`.
cmm_apply_list_change = {
	set_local_variable = { name = cmm_tmp_flag value = flag:$setting$ }
	set_variable = { name = cmm_tmp_is_global_list value = "variable_map(cmm_is_global|local_var:cmm_tmp_flag)" }

	if = {
		limit = { var:cmm_tmp_is_global_list >= 1 }
		_cmm_apply_list_change_internal = { setting = $setting$ vt = global_variable lt = in_global }
	}
	else = {
		_cmm_apply_list_change_internal = { setting = $setting$ vt = variable lt = in }
	}

	if = {
		limit = { has_variable = cmm_list_position_to_apply }
		remove_variable = cmm_list_position_to_apply
	}

	if = {
		limit = { has_variable = cmm_list_action_to_apply }
		remove_variable = cmm_list_action_to_apply
	}

	if = {
		limit = { has_variable = cmm_list_field_slot_to_apply }
		remove_variable = cmm_list_field_slot_to_apply
	}

	if = {
		limit = { has_variable = cmm_dropdown_index_to_apply }
		remove_variable = cmm_dropdown_index_to_apply
	}

	if = {
		limit = { has_variable = cmm_list_field_value_to_apply }
		remove_variable = cmm_list_field_value_to_apply
	}

	if = {
		limit = { has_variable = cmm_numeric_step_delta_to_apply }
		remove_variable = cmm_numeric_step_delta_to_apply
	}

	if = {
		limit = { has_variable = cmm_numeric_change_to_apply }
		remove_variable = cmm_numeric_change_to_apply
	}

	if = {
		limit = { has_variable = cmm_slider_index_to_apply }
		remove_variable = cmm_slider_index_to_apply
	}

	if = {
		limit = { has_variable = cmm_list_dispatch_action }
		remove_variable = cmm_list_dispatch_action
	}

	if = {
		limit = { has_variable = cmm_tmp_is_global_list }
		remove_variable = cmm_tmp_is_global_list
	}

	# Benign warning suppression
	if = {
		limit = { always = no }
		cmf_suppress = { v = cmm_tmp_flag }
		cmf_suppress = { v = cmm_tmp_is_global_list }
	}
}

# Iterate over every item in a list setting and call a user-provided effect for each.
# The effect receives `$i$` as the resolved item number (1-20) and `$setting$` as the
# composite key, so the caller can read field values from the cmm variable map like
# `"variable_map(cmm|flag:$setting$_i$i$_f1)"`.
# If the item has a value (set via cmm_set_list_item_value or cmm_add_settings_list_item),
# it is available as `scope:cmm_list_current_item_value` inside the called effect.
# Scope: country
# Args:
# - setting (key): Composite setting key `<mod_id>__<setting_id>`.
# - effect (key): Name of the scripted effect to call per item.
cmm_for_each_list_item = {
	set_local_variable = { name = cmm_tmp_flag value = flag:$setting$ }
	if = {
		limit = { "variable_map(cmm_is_global|local_var:cmm_tmp_flag)" >= 1 }
		every_in_global_list = {
			variable = cmm_list_items_$setting$
			save_scope_as = cmm_list_current_item

			root = {
				_cmm_resolve_list_item_and_call = { setting = $setting$ effect = $effect$ }
			}
		}
	}
	else = {
		every_in_list = {
			variable = cmm_list_items_$setting$
			save_scope_as = cmm_list_current_item

			root = {
				_cmm_resolve_list_item_and_call = { setting = $setting$ effect = $effect$ }
			}
		}
	}

	# Benign warning suppression
	if = {
		limit = { always = no }
		cmf_suppress = { v = cmm_tmp_flag }
	}
}

# Internal: save the item's value scope (if set) and invoke the user-provided effect.
# The value is read from the single-element variable list `<setting>_item_<item>_value`
# and exposed as `scope:cmm_list_current_item_value` inside the called effect.
# If no value was set for the item, the scope is simply not available.
# Scope: country
# Args: setting (key), item (number), effect (key)
_cmm_save_list_item_value_and_call = {
	every_in_list = {
		variable = $setting$_i$item$_value
		save_scope_as = cmm_list_current_item_value
	}
	$effect$ = { setting = $setting$ i = $item$ }
}
