# Register a settings list for a mod (idempotent by mod_id + setting_id).
# Scope: country
# Args:
# - mod_id (key): Owner mod id.
# - setting_id (key): Setting id unique within mod.
# - tab_id (key): Owner tab id within mod.
# - item_count (number): Number of list items (`1..20`).
# - is_ordered (1/0): `1` shows row move controls, `0` hides them.
cmm_register_settings_list = {
	_cmm_register_list_setting_internal = {
		mod_id = $mod_id$
		setting_id = $setting_id$
		tab_id = $tab_id$
		item_count = $item_count$
		is_ordered = $is_ordered$
		is_global = 0
		vt = variable
	}
}

# Register shared list-setting metadata and storage for a mod (idempotent by mod_id + setting_id).
# Scope: country
# Args:
# - mod_id (key): Owner mod id.
# - setting_id (key): Setting id unique within mod.
# - tab_id (key): Owner tab id within mod.
# - item_count (number): Number of list items (`1..20`).
# - is_ordered (1/0): `1` shows row move controls, `0` hides them.
# - is_global (1/0): `1` for global variable storage, `0` for country variable storage.
# - vt (variable/global_variable): Variable type token for macro composition.
_cmm_register_list_setting_internal = {
	add_to_variable_map = { name = cmm_type key = flag:$mod_id$__$setting_id$ value = 7 }
	add_to_variable_map = { name = cmm_is_global key = flag:$mod_id$__$setting_id$ value = $is_global$ }
	set_variable = { name = cmm_setting_list_is_ordered_$mod_id$__$setting_id$ value = $is_ordered$ }
	set_variable = { name = cmm_setting_list_count_$mod_id$__$setting_id$ value = $item_count$ }

	if = {
		limit = { var:cmm_setting_list_is_ordered_$mod_id$__$setting_id$ > 0 }
		set_variable = { name = cmm_setting_list_is_ordered_$mod_id$__$setting_id$ value = 1 }
	}
	else = {
		set_variable = { name = cmm_setting_list_is_ordered_$mod_id$__$setting_id$ value = 0 }
	}

	if = {
		limit = { var:cmm_setting_list_count_$mod_id$__$setting_id$ < 1 }
		set_variable = { name = cmm_setting_list_count_$mod_id$__$setting_id$ value = 1 }
	}

	if = {
		limit = { var:cmm_setting_list_count_$mod_id$__$setting_id$ > 20 }
		set_variable = { name = cmm_setting_list_count_$mod_id$__$setting_id$ value = 20 }
	}

	set_variable = {
		name = cmm_setting_list_last_index_$mod_id$__$setting_id$
		value = var:cmm_setting_list_count_$mod_id$__$setting_id$
	}
	change_variable = { name = cmm_setting_list_last_index_$mod_id$__$setting_id$ add = -1 }

	add_to_variable_map = { name = cmm_list_count key = flag:$mod_id$__$setting_id$ value = var:cmm_setting_list_count_$mod_id$__$setting_id$ }
	add_to_variable_map = { name = cmm_list_is_ordered key = flag:$mod_id$__$setting_id$ value = var:cmm_setting_list_is_ordered_$mod_id$__$setting_id$ }
	add_to_variable_map = { name = cmm_list_last_index key = flag:$mod_id$__$setting_id$ value = var:cmm_setting_list_last_index_$mod_id$__$setting_id$ }
	add_to_variable_map = { name = cmm_has_sgui key = flag:$mod_id$__$setting_id$ value = 1 }

	if = {
		limit = { NOT = { has_variable = cmm_list_field_count_$mod_id$__$setting_id$ } }
		set_variable = { name = cmm_list_field_count_$mod_id$__$setting_id$ value = 0 }
	}

	_cmm_register_setting_metadata = {
		mod_id = $mod_id$
		setting_id = $setting_id$
		tab_id = $tab_id$
		group_id = $setting_id$
	}

	if = {
		limit = { NOT = { has_variable = cmm_list_initialized_$mod_id$__$setting_id$ } }

		set_variable = {
			name = cmm_list_item_column_name_$mod_id$__$setting_id$
			value = flag:$mod_id$__$setting_id$_item_column_name
		}

		_cmm_initialize_list_setting_item_metadata = {
			mod_id = $mod_id$
			setting_id = $setting_id$
		}

		_cmm_initialize_list_setting_item_list = {
			mod_id = $mod_id$
			setting_id = $setting_id$
			vt = $vt$
		}

		set_variable = { name = cmm_list_initialized_$mod_id$__$setting_id$ value = 1 }
	}

	_cmm_suppress_list_setting_warnings = {
		mod_id = $mod_id$
		setting_id = $setting_id$
	}
}

# Internal: dispatch bool field init at a given slot with global/local branching.
# Scope: country
# Args: mod_id (key), setting_id (key), field_id (key), slot (number), default_value (1/0)
_cmm_init_list_bool_field_at_slot = {
	set_local_variable = { name = cmm_tmp_flag value = flag:$mod_id$__$setting_id$ }
	if = {
		limit = { "variable_map(cmm_is_global|local_var:cmm_tmp_flag)" >= 1 }
		_cmm_initialize_list_bool_field_slot = { mod_id = $mod_id$ setting_id = $setting_id$ field_id = $field_id$ slot = $slot$ default_value = $default_value$ vt = global_variable is_global = 1 }
	}
	else = {
		_cmm_initialize_list_bool_field_slot = { mod_id = $mod_id$ setting_id = $setting_id$ field_id = $field_id$ slot = $slot$ default_value = $default_value$ vt = variable is_global = 0 }
	}
	# Benign warning suppression
	if = {
		limit = { always = no }
		cmf_suppress = { v = cmm_tmp_flag }
	}
}

# Register a bool field shared by every item in one list setting.
# Scope: country
# Args:
# - mod_id (key): Owner mod id.
# - setting_id (key): Target list setting id within the mod.
# - field_id (key): Field id unique within the list setting.
# - default_value (1/0): Default checkbox state for new saves.
cmm_register_list_bool_field = {
	_cmm_register_list_field_slot = {
		mod_id = $mod_id$
		setting_id = $setting_id$
		field_id = $field_id$
	}

	if = {
		limit = {
			AND = {
				has_variable = cmm_setting_list_count_$mod_id$__$setting_id$
				has_variable = cmm_list_field_slot_$mod_id$__$setting_id$__$field_id$
			}
		}
		switch = {
			trigger = var:cmm_list_field_slot_$mod_id$__$setting_id$__$field_id$
			1 = { _cmm_init_list_bool_field_at_slot = { mod_id = $mod_id$ setting_id = $setting_id$ field_id = $field_id$ slot = 1 default_value = $default_value$ } }
			2 = { _cmm_init_list_bool_field_at_slot = { mod_id = $mod_id$ setting_id = $setting_id$ field_id = $field_id$ slot = 2 default_value = $default_value$ } }
			3 = { _cmm_init_list_bool_field_at_slot = { mod_id = $mod_id$ setting_id = $setting_id$ field_id = $field_id$ slot = 3 default_value = $default_value$ } }
			4 = { _cmm_init_list_bool_field_at_slot = { mod_id = $mod_id$ setting_id = $setting_id$ field_id = $field_id$ slot = 4 default_value = $default_value$ } }
			5 = { _cmm_init_list_bool_field_at_slot = { mod_id = $mod_id$ setting_id = $setting_id$ field_id = $field_id$ slot = 5 default_value = $default_value$ } }
		}
	}

	_cmm_suppress_list_bool_field_warnings = {
		mod_id = $mod_id$
		setting_id = $setting_id$
		field_id = $field_id$
	}
}

# Disable one field column for one specific item in a list setting.
# The field cell will render as empty space, preserving column alignment.
# Call after the field has been registered (e.g. cmm_register_list_bool_field).
# Scope: country
# Args:
# - mod_id (key): Owner mod id.
# - setting_id (key): Target list setting id within the mod.
# - field_id (key): Field id to disable for the item.
# - item (number): Item ordinal to disable (`1..item_count`).
cmm_disable_list_field_for_item = {
	if = {
		limit = { has_variable = cmm_list_field_slot_$mod_id$__$setting_id$__$field_id$ }
		switch = {
			trigger = var:cmm_list_field_slot_$mod_id$__$setting_id$__$field_id$
			1 = { add_to_variable_map = { name = cmm_list_field_disabled key = flag:$mod_id$__$setting_id$_i$item$_f1 value = 1 } }
			2 = { add_to_variable_map = { name = cmm_list_field_disabled key = flag:$mod_id$__$setting_id$_i$item$_f2 value = 1 } }
			3 = { add_to_variable_map = { name = cmm_list_field_disabled key = flag:$mod_id$__$setting_id$_i$item$_f3 value = 1 } }
			4 = { add_to_variable_map = { name = cmm_list_field_disabled key = flag:$mod_id$__$setting_id$_i$item$_f4 value = 1 } }
			5 = { add_to_variable_map = { name = cmm_list_field_disabled key = flag:$mod_id$__$setting_id$_i$item$_f5 value = 1 } }
		}
	}
}

# Attach a game object scope as the value of one list item (call after cmm_register_settings_list).
# Stores the scope in a single-element variable list per item.
# When iterating with cmm_for_each_list_item, the scope is available as
# `scope:cmm_list_current_item_value` inside your per-item effect.
# Scope: country
# Args:
# - mod_id (key): Owner mod id.
# - setting_id (key): Target list setting id within the mod.
# - item (number): Item ordinal to tag (`1..item_count`).
# - value (scope): Game scope to attach (e.g. `building_type:fine_cloth_guild`).
cmm_set_list_item_value = {
	clear_variable_list = $mod_id$__$setting_id$_i$item$_value
	add_to_variable_list = { name = $mod_id$__$setting_id$_i$item$_value target = $value$ }
}

# Finalize a dynamic settings list after all item-add calls.
# Sets the item count from the accumulated add counter and marks the list initialized.
# Register fields (cmm_register_list_bool_field, etc.) after this call.
# Idempotent: subsequent calls are no-ops.
# Scope: country
# Args:
# - mod_id (key): Owner mod id.
# - setting_id (key): Target list setting id within the mod.
cmm_finish_settings_list = {
	if = {
		limit = { NOT = { has_variable = cmm_list_initialized_$mod_id$__$setting_id$ } }

		if = {
			limit = { var:cmm_list_add_counter_$mod_id$__$setting_id$ < 1 }
			set_variable = { name = cmm_list_add_counter_$mod_id$__$setting_id$ value = 1 }
		}

		set_variable = {
			name = cmm_setting_list_count_$mod_id$__$setting_id$
			value = var:cmm_list_add_counter_$mod_id$__$setting_id$
		}

		set_variable = {
			name = cmm_setting_list_last_index_$mod_id$__$setting_id$
			value = var:cmm_list_add_counter_$mod_id$__$setting_id$
		}
		change_variable = { name = cmm_setting_list_last_index_$mod_id$__$setting_id$ add = -1 }

		add_to_variable_map = { name = cmm_list_count key = flag:$mod_id$__$setting_id$ value = var:cmm_list_add_counter_$mod_id$__$setting_id$ }
		add_to_variable_map = { name = cmm_list_last_index key = flag:$mod_id$__$setting_id$ value = var:cmm_setting_list_last_index_$mod_id$__$setting_id$ }

		set_variable = { name = cmm_list_initialized_$mod_id$__$setting_id$ value = 1 }
	}

	_cmm_suppress_list_setting_warnings = { mod_id = $mod_id$ setting_id = $setting_id$ }
}
