# Register a boolean setting for a mod (idempotent by mod_id + setting_id).
# Scope: country
# Args:
# - mod_id (key): Owner mod id.
# - setting_id (key): Setting id unique within mod.
# - tab_id (key): Owner tab id within mod.
# - group_id (key): Owner group id within mod.
# - default_value (1/0): Initial value; applied when the setting does not exist yet.
cmm_register_bool_setting = {
	_cmm_register_bool_setting_internal = {
		mod_id = $mod_id$
		setting_id = $setting_id$
		tab_id = $tab_id$
		group_id = $group_id$
		default_value = $default_value$
		is_global = 0
		vt = variable
	}
}

# Register a global boolean setting for a mod.
# Scope: country
# Args:
# - mod_id (key): Owner mod id.
# - setting_id (key): Setting id unique within mod.
# - tab_id (key): Owner tab id within mod.
# - group_id (key): Owner group id within mod.
# - default_value (1/0): Initial global value when missing.
cmm_register_global_bool_setting = {
	_cmm_register_bool_setting_internal = {
		mod_id = $mod_id$
		setting_id = $setting_id$
		tab_id = $tab_id$
		group_id = $group_id$
		default_value = $default_value$
		is_global = 1
		vt = global_variable
	}
}

# Internal: shared registration for bool settings.
# Scope: country
# Args: mod_id (key), setting_id (key), tab_id (key), group_id (key), default_value (1/0), is_global (1/0), vt (variable/global_variable)
_cmm_register_bool_setting_internal = {
	add_to_variable_map = { name = cmm_type key = flag:$mod_id$__$setting_id$ value = 1 }
	add_to_variable_map = { name = cmm_is_global key = flag:$mod_id$__$setting_id$ value = $is_global$ }
	add_to_variable_map = { name = cmm_default key = flag:$mod_id$__$setting_id$ value = $default_value$ }

	_cmm_register_setting_metadata = {
		mod_id = $mod_id$
		setting_id = $setting_id$
		tab_id = $tab_id$
		group_id = $group_id$
	}

	if = {
		limit = { NOT = { is_key_in_$vt$_map = { name = cmm target = flag:$mod_id$__$setting_id$ } } }
		add_to_$vt$_map = { name = cmm key = flag:$mod_id$__$setting_id$ value = $default_value$ }
	}
}

# Toggle a registered boolean setting value between `0` and `1`.
# Scope: country
# Args:
# - setting (key): Composite setting key `<mod_id>__<setting_id>`.
_cmm_toggle_bool_setting = {
	set_local_variable = { name = cmm_tmp_flag value = flag:$setting$ }
	if = {
		limit = { "variable_map(cmm_is_global|local_var:cmm_tmp_flag)" >= 1 }
		if = {
			limit = { "global_variable_map(cmm|local_var:cmm_tmp_flag)" >= 1 }
			remove_from_global_variable_map = { name = cmm key = flag:$setting$ }
			add_to_global_variable_map = { name = cmm key = flag:$setting$ value = 0 }
		}
		else = {
			remove_from_global_variable_map = { name = cmm key = flag:$setting$ }
			add_to_global_variable_map = { name = cmm key = flag:$setting$ value = 1 }
		}
	}
	else = {
		if = {
			limit = { "variable_map(cmm|local_var:cmm_tmp_flag)" >= 1 }
			remove_from_variable_map = { name = cmm key = flag:$setting$ }
			add_to_variable_map = { name = cmm key = flag:$setting$ value = 0 }
		}
		else = {
			remove_from_variable_map = { name = cmm key = flag:$setting$ }
			add_to_variable_map = { name = cmm key = flag:$setting$ value = 1 }
		}
	}

	# Benign warning suppression
	if = {
		limit = { always = no }
		cmf_suppress = { v = cmm_tmp_flag }
	}
}
