# Run built-in CMM registration and the shared registration hook.
# Scope: country
CMM_RegisterCoreMod = {
	effect = {
		_cmm_register_menu_settings = yes
		trigger_event_silently = {
			on_action = cmf_on_mod_registration
		}
	}
}

# Store the current country as the CMM host country, then run built-in registration and the shared registration hook.
# Scope: country
CMM_SetHostAndRegisterCoreMod = {
	effect = {
		_cmm_mark_host_country = yes
		_cmm_register_menu_settings = yes
		trigger_event_silently = {
			on_action = cmf_on_mod_registration
		}
	}
}

# Reset all settings for all mods to their registered default values.
# Scope: country
CMM_ResetToDefaults = {
	scope = country
	effect = {
		_cmm_reset_to_defaults = yes
	}
}
