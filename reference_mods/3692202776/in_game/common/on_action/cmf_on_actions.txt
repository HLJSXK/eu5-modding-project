on_game_start = {
	on_actions = {
		_cmf_on_init
	}
}

# Set global presence flag for cross-mod detection.
# Scope: global
_cmf_on_init = {
	effect = {
		set_global_variable = { name = cmf_active value = yes }
	}
}

yearly_country_pulse = {
	on_actions = {
		_cmf_on_yearly
	}
}

monthly_country_pulse = {
	on_actions = {
		_cmf_on_monthly
	}
}

# Yearly event dispatcher
# Scope: country (from yearly_country_pulse)
_cmf_on_yearly = {
	trigger = { is_ai = no }
	effect = {
		trigger_event_silently = {
			on_action = cmf_yearly_human_country_pulse
		}
	}
}

# Monthly event dispatcher
# Scope: country (from monthly_country_pulse)
_cmf_on_monthly = {
	trigger = { is_ai = no }
	effect = {
		trigger_event_silently = {
			on_action = cmf_monthly_human_country_pulse
		}
	}
}

# Unified yearly human country pulse
# Slightly more efficient than each mod checking for players individually
# Scope: country
cmf_yearly_human_country_pulse = {
	on_actions = {
		_cmf_on_cmm_registration
		cmf_on_mod_registration
	}
}

# Unified monthly human country pulse
# Slightly more efficient than each mod checking for players individually
# Scope: country
cmf_monthly_human_country_pulse = {
	on_actions = { }
}

# CMM menu settings registration. Runs before cmf_on_mod_registration
# to ensure CMF's own settings appear first in the mod list.
# Scope: country
_cmf_on_cmm_registration = {
	effect = {
		_cmm_register_menu_settings = yes
	}
}

# Integration extension point.
# Scope: country
cmf_on_mod_registration = {
	on_actions = { }
}

# Post-lobby game start hook. Fires once on a new game when the in-game UI
# is created, after the player has selected their country. Vanilla on_game_start
# fires before the country selection screen, so player-scoped effects
# (is_ai = no) don't work there since no country has been selected yet.
# Scope: none
on_game_start_after_lobby = {
	effect = {
		every_country = {
			limit = { is_ai = no }
			trigger_event_silently = {
				on_action = on_game_start_after_lobby_human_country
			}
		}
	}
	on_actions = { }
}

# Human country variant of on_game_start_after_lobby.
# Scope: country
on_game_start_after_lobby_human_country = {
	on_actions = {
		_cmf_on_cmm_registration
		cmf_on_mod_registration
	}
}

# Save load hook. Fires every time a save game is loaded (including from
# the country selection screen). Does NOT fire on new games.
# Scope: none
on_game_load = {
	effect = {
		set_global_variable = { name = cmf_active value = yes }
		every_country = {
			limit = { is_ai = no }
			trigger_event_silently = {
				on_action = on_game_load_human_country
			}
		}
	}
	on_actions = { }
}

# Human country variant of on_game_load.
# Scope: country
on_game_load_human_country = {
	on_actions = {
		_cmf_on_cmm_registration
	}
}

# Post-lobby save load hook. Fires every time a save game is loaded,

on_game_load_after_lobby = {

			}
		}

	on_actions = { }
}

# Human country variant of on_game_load_after_lobby.
# Scope: country
on_game_load_after_lobby_human_country = {
	on_actions = {
		_cmf_on_cmm_registration
		cmf_on_mod_registration
	}
}
