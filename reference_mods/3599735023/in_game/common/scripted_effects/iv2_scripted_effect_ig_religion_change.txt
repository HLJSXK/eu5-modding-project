iv2_scripted_effect_ig_change = {
		
	# Fallback if vars are not initialized for some reason yet
	if = {
			limit = {
				NOT = { 
					has_variable = iv2_var_innovation_points_adm
				}
			}
			iv2_add_vars_effect = yes
	}
	
	if = {
		limit = {
			has_variable = iv2_reli_ideagroup
		}
		# Remove religious Idea groups on religion change
		iv2_scripted_effect_remove_reli_idea_groups_all = yes
		
		# Add back  religious Idea groups on religion change
		iv2_scripted_effect_add_reli_idea_groups_all = yes
	}
	
	
	### Add Ideas Generic
	
	if = {
		limit = { 
			var:iv2_reli_ideagroup = 1
			iv2_scripted_trigger_generic_reli_ig = yes
		}
		add_country_modifier = { modifier = ideagroup_modifier_adm_11_1 years = -1 }
	}
	
	if = {
		limit = { 
			var:iv2_reli_ideagroup = 2
			iv2_scripted_trigger_generic_reli_ig = yes
		}
		add_country_modifier = { modifier = ideagroup_modifier_adm_11_1 years = -1 }
		add_country_modifier = { modifier = ideagroup_modifier_adm_11_2 years = -1 }
	}
	
	if = {
		limit = { 
			var:iv2_reli_ideagroup = 3
			iv2_scripted_trigger_generic_reli_ig = yes
		}
		add_country_modifier = { modifier = ideagroup_modifier_adm_11_1 years = -1 }
		add_country_modifier = { modifier = ideagroup_modifier_adm_11_2 years = -1 }
		add_country_modifier = { modifier = ideagroup_modifier_adm_11_3 years = -1 }
	}
	
	if = {
		limit = { 
			var:iv2_reli_ideagroup = 4
			iv2_scripted_trigger_generic_reli_ig = yes
