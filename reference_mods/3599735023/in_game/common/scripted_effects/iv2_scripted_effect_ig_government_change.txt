iv2_scripted_effect_add_gov_idea_groups_all = {

	# Monarchy
		if = {
			limit = {
				iv2_scripted_trigger_has_government_ig_selected = yes
				government_type = government_type:monarchy
			}
			fire_generic_action = {
				type = iv2_ga_select_ideagroup_adm_7
				actor = root
				recipient = root
			}
		}
		
	# Republic
		if = {
			limit = {
				iv2_scripted_trigger_has_government_ig_selected = yes
				government_type = government_type:republic
			}
			fire_generic_action = {
				type = iv2_ga_select_ideagroup_adm_8
				actor = root
				recipient = root
			}
		}	
		
	# Theocracy
		if = {
			limit = {
				iv2_scripted_trigger_has_government_ig_selected = yes
				government_type = government_type:theocracy
			}
			fire_generic_action = {
				type = iv2_ga_select_ideagroup_adm_10
				actor = root
				recipient = root
			}
		}

	# Tribe
		if = {
			limit = {
				iv2_scripted_trigger_has_government_ig_selected = yes
				government_type = government_type:tribe
			}
			fire_generic_action = {
				type = iv2_ga_select_ideagroup_adm_23
				actor = root
				recipient = root
			}
		}	

	# Horde
		if = {
			limit = {
				iv2_scripted_trigger_has_government_ig_selected = yes
				government_type = government_type:steppe_horde
			}
			fire_generic_action = {
				type = iv2_ga_select_ideagroup_adm_22
				actor = root
				recipient = root
			}
		}	

}
