iv2_gui_stat_name_navy_1 = {
	scope = country 		
	is_valid = {  
		OR = {
			has_variable = iv2_ideagroup_select_navy_1_1 
			
			has_variable = iv2_var_show_all_ig_statistics_yes
			AND = {
				has_variable = iv2_stat_show_navy_1
				has_variable = iv2_var_show_all_ig_statistics_yes
			}
			AND = {

				has_variable = iv2_stat_show_navy_1
				has_variable = iv2_var_show_all_ig_statistics_no

			}
		}		
	}
}

	is_valid = {

			}
			AND = {
				has_variable = iv2_stat_show_navy_1
				has_variable = iv2_var_show_all_ig_statistics_no
			}
		}		
		NOT = {
			OR = {
				AND = {
					has_variable = iv2_stat_hide_navy_1
					has_variable = iv2_var_show_all_ig_statistics_no
				}	
				AND = {
					has_variable = iv2_stat_hide_navy_1
					has_variable = iv2_var_show_all_ig_statistics_yes
				}
			}			
		}
	}
	effect = {
		remove_variable = iv2_stat_show_navy_1
		set_variable = {
			name = iv2_stat_hide_navy_1
			value = 1
		}
	}	
}
