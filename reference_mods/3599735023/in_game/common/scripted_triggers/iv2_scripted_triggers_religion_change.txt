		AND = {
			religion.group = religion_group:folk_asian_group # except tengri
			NOT = { religion = religion:tengri }

		religion.group = religion_group:folk_brazilian_group
		
		religion.group = religion_group:folk_caribbean_group
		
		religion.group = religion_group:folk_melanesian_group
		
		religion.group = religion_group:folk_permic_group
