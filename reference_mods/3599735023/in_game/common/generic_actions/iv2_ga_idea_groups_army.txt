iv2_ga_select_ideagroup_army_1 = {
	type = owncountry	

	ai_tick = monthly
	ai_tick_frequency = 36

	automation_tick = monthly
	automation_tick_frequency = 1
	player_automated_category = governmentreform


	potential = {
		
	}

	allow = {
		scope:actor = {
			custom_tooltip = {
				text = iv2_tt_no_available_idea_army
				var:iv2_var_current_idea_group_army > var:iv2_var_current_picked_idea_group_army
			}
		}
	}

	ai_will_do = {
		add = {
			value = 1000
		}
	}
	
	effect = {
		scope:actor = {
			set_variable = iv2_ideagroup_select_army_1_1 
			change_variable = {
				name = iv2_var_current_picked_idea_group_army
				add = 1
			}
		}
	}
}

	type = owncountry	

	ai_tick = never
	#ai_tick_frequency = 36

	}

		scope:actor = {

			}
		}	
	}
}

	allow = {
		scope:actor = {
			hidden_trigger = {
				NOT = { has_country_modifier = ideagroup_modifier_army_1_1 }

			custom_tooltip = {
			text = iv2_tt_no_points_for_idea_army

			}
		}

		add = {
			value = 1000
		}
	}
	
	effect = {
		scope:actor = {
			hidden_effect = {
				add_country_modifier = { 
					modifier = ideagroup_modifier_army_1_1 
					years = -1 
				}
			}
			custom_tooltip = {
			text = iv2_tt_id_sub_points_army
				change_variable = {
						name = iv2_var_innovation_points_army
						subtract = iv2_idea_cost_army
				}
			}
		}
	}
}
