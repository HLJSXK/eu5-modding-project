iv2_ga_select_ideagroup_navy_1 = {
	type = owncountry	

	ai_tick = monthly
	ai_tick_frequency = 36

	automation_tick = monthly
	automation_tick_frequency = 1
	player_automated_category = governmentreform


	potential = {
		
	}

	allow = {
		scope:actor = {
			custom_tooltip = {
				text = iv2_tt_no_available_idea_navy
				var:iv2_var_current_idea_group_navy > var:iv2_var_current_picked_idea_group_navy
			}
		}
	}

	ai_will_do = {
		add = {
			value = 1000
		}
	}
	
	effect = {
		scope:actor = {
			set_variable = iv2_ideagroup_select_navy_1_1 
			change_variable = {
				name = iv2_var_current_picked_idea_group_navy
				add = 1
			}
		}
	}
}

iv2_ga_deselect_ideagroup_navy_1 = {
	type = owncountry	

	ai_tick = never
	#ai_tick_frequency = 36

	potential = {
		
	}

	allow = {
		
	}
	
	effect = {
		scope:actor = {
			remove_variable = iv2_ideagroup_select_navy_1_1 
			iv2_scripted_effect_ig_revoke_navy_1 = yes
			change_variable = {
				name = iv2_var_current_picked_idea_group_navy
				subtract = 1
			}
		}	
	}
}



iv2_ga_ideagroup_navy_1_1 = {
	type = owncountry	

	ai_tick = monthly
	ai_tick_frequency = 36

	automation_tick = monthly
	automation_tick_frequency = 1
	player_automated_category = governmentreform


	potential = {
		
	}

	allow = {
		scope:actor = {
			hidden_trigger = {
				NOT = { has_country_modifier = ideagroup_modifier_navy_1_1 }
				has_variable = iv2_ideagroup_select_navy_1_1
			}
			custom_tooltip = {
			text = iv2_tt_no_points_for_idea_navy
			var:iv2_var_innovation_points_navy >= var:iv2_var_idea_cost_navy
			}
		}
	}

	ai_will_do = {
		add = {
			value = 1000
		}
	}
	
	effect = {
		scope:actor = {
			hidden_effect = {
				add_country_modifier = { 
					modifier = ideagroup_modifier_navy_1_1 
					years = -1 
				}
			}
			custom_tooltip = {
			text = iv2_tt_id_sub_points_navy
				change_variable = {
						name = iv2_var_innovation_points_navy
						subtract = iv2_idea_cost_navy
				}
			}
		}
	}
}
