iv2_ga_select_ideagroup_dip_1 = {
	type = owncountry	

	ai_tick = monthly
	ai_tick_frequency = 36

	automation_tick = monthly
	automation_tick_frequency = 1
	player_automated_category = governmentreform


	potential = {
		
	}

	allow = {
		scope:actor = {
			custom_tooltip = {
				text = iv2_tt_no_available_idea_dip
				var:iv2_var_current_idea_group_dip > var:iv2_var_current_picked_idea_group_dip
			}
		}
	}

	ai_will_do = {
		add = {
			value = 1000
		}
	}
	
	effect = {
		scope:actor = {
			set_variable = iv2_ideagroup_select_dip_1_1 
			change_variable = {
				name = iv2_var_current_picked_idea_group_dip
				add = 1
			}
		}
	}
}

iv2_ga_deselect_ideagroup_dip_1 = {
	type = owncountry	

	ai_tick = never
	#ai_tick_frequency = 36

	potential = {
		
	}

	allow = {
		
	}
	
	effect = {
		scope:actor = {
			remove_variable = iv2_ideagroup_select_dip_1_1 
			iv2_scripted_effect_ig_revoke_dip_1 = yes
			change_variable = {
				name = iv2_var_current_picked_idea_group_dip
				subtract = 1
			}
		}	
	}
}

	type = owncountry	

	ai_tick = monthly
	ai_tick_frequency = 36

	automation_tick = monthly
	automation_tick_frequency = 1
	player_automated_category = governmentreform


	potential = {
		
	}

	allow = {
		scope:actor = {
			hidden_trigger = {
				NOT = { has_country_modifier = ideagroup_modifier_dip_1_1 }
				has_variable = iv2_ideagroup_select_dip_1_1
			}
			custom_tooltip = {
			text = iv2_tt_no_points_for_idea_dip
			var:iv2_var_innovation_points_dip >= var:iv2_var_idea_cost_dip
			}
		}
	}

	ai_will_do = {
		add = {
			value = 1000
		}
	}
	
	effect = {
		scope:actor = {
			hidden_effect = {
				add_country_modifier = { 
					modifier = ideagroup_modifier_dip_1_1 
					years = -1 
				}
			}
			custom_tooltip = {
			text = iv2_tt_id_sub_points_dip
				change_variable = {
						name = iv2_var_innovation_points_dip
						subtract = iv2_idea_cost_dip
				}
			}
		}
	}
}

iv2_ga_select_ideagroup_dip_23 = {
	type = owncountry	

	ai_tick = never
	ai_tick_frequency = 36

	automation_tick = monthly
	automation_tick_frequency = 1
	player_automated_category = governmentreform


	potential = {
		
	}

	allow = {
		scope:actor = {
			custom_tooltip = {
				text = iv2_tt_no_available_idea_dip
				var:iv2_var_current_idea_group_dip > var:iv2_var_current_picked_idea_group_dip
			}
		}
	}

	ai_will_do = {
		add = {
			value = 1000
		}
	}
	
	effect = {
		scope:actor = {
			set_variable = iv2_ideagroup_select_dip_23_1 
			change_variable = {
				name = iv2_var_current_picked_idea_group_dip
				add = 1
			}
		}
	}
}
