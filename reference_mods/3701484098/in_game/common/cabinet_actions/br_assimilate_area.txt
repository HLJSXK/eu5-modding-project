REPLACE_OR_CREATE:assimilate_area = {
	ability = dip
	allow_multiple = no
	
	potential = {
		modifier:allow_cabinet_assimilate_area = yes
	}
	
	allow = {
		AND = {##
			cultural_unity < 1.0
			is_ai = no
		}
	}

	
	select_trigger = {
		looking_for_a = area
		source = actor
		target_flag = target
		name = "assimilate_area_select_area"
		none_available_msg_key = "assimilate_area_no_areas"
		column = {
			data = name
		}
		column = {
			data = area_primary_culture_percentage
		}
		column = {
			data = population
		}
		default_sort = area_primary_culture_percentage
		visible = {
			OR = {
				AND = { # for humans
					scope:actor = {
						is_ai = no
					}
					OR = {
						scope:actor = {
							culture_percentage_in_country = {
								culture = root
								value >= 0.01
							}
						}
						
						cultural_view = {
							target = scope:actor.culture
							value > neutral 
						}
						reverse_cultural_view = {
							target = scope:actor.culture
							value > neutral 
						}
						
						is_primary_or_accepted_in = scope:actor
					}
				}

				AND = { # for AI
					scope:actor = {
						is_ai = yes
					}
					is_primary_in = scope:actor 
				}
			}
		}
		map_color = {
			lerp = {
				min_color = define:NMapColors|MAP_COLOR_LOW
				max_color = define:NMapColors|MAP_COLOR_HIGH
				factor = "area.culture_percentage(scope:actor.dominant_culture)"
			}
		}
	}

	location_modifier = {
		local_pop_assimilation_speed = 0.01 #br edited from 0.02
	}
	country_modifier = {
		monthly_nationalist_rebel_growth = 0.02 # +2% nationalist rebel growth while active
	}


	map_marker = {
		show_for_owner = yes
	}

	progress = {
		scope:target = {
			value = "culture_percentage_in_area(scope:actor|scope:actor.dominant_culture)"
		}
	}
}
