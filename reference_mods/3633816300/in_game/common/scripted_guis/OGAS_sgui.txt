OGAS_Auto_Road_Initialize_sgui = {
	scope = country
	is_shown = { always = yes }
	is_valid = { always = yes }
	effect = {
		OGAS_Auto_Road_Refresh = yes
	}
}

OGAS_Auto_Road_Construct_sgui = {
	scope = country
	is_shown = { 
		always = yes 
	}
	effect = {
		if = {
			limit = {
				var:OGAS_Auto_Road_Construct = 1
			}
			set_variable = {
				name = OGAS_Auto_Road_Construct
				value = 0
			}
		}
		else_if = {
			limit = {
				var:OGAS_Auto_Road_Construct = 0
			}
			set_variable = {
				name = OGAS_Auto_Road_Construct
				value = 1
			}
			if = {
				limit = {
					NOT = {
						any_owned_location = {
							OR = {
								has_variable = OGAS_Cache_Auto_Road_Complete
								has_variable = OGAS_Cache_Auto_Road_Ongoing
							}
						}
					}
				}
				OGAS_Auto_Road_Refresh = yes
			}
		}
	}
}

OGAS_Auto_Road_Gold_Reserve_Dec_10k_sgui = {
	scope = country
	is_shown = { always = yes }
	is_valid = { always = yes }
	effect = {
		if = {
			limit = { NOT = { has_variable = OGAS_Auto_Road_Min_Gold } }
			set_variable = {
				name = OGAS_Auto_Road_Min_Gold
				value = OGAS_deposit_gold
			}
		}
		change_variable = {
			name = OGAS_Auto_Road_Min_Gold
			add = -10000
		}
		if = {
			limit = { var:OGAS_Auto_Road_Min_Gold < 0 }
			set_variable = {
				name = OGAS_Auto_Road_Min_Gold
				value = 0
			}
		}
		if = {
			limit = { var:OGAS_Auto_Road_Min_Gold > 100000 }
			set_variable = {
				name = OGAS_Auto_Road_Min_Gold
				value = 100000
			}
		}
	}
}
