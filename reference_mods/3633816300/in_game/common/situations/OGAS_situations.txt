OGAS = {
	monthly_spawn_chance = 1
	#hint_tag = hint_OGAS
	
	can_start = {
		#this scope is situation not country
		#is_ai = no
	}
	
	can_end = {
		always = no
	}
	
	visible = {
		is_ai = no
	}

	on_start = {
	}

	on_monthly = {
	}

	tooltip = {
		if = {
			limit = {
				has_owner = yes
				owner ?= {
					has_variable = OGAS_Auto_Road_Mode
					var:OGAS_Auto_Road_Mode = 1
				}
			}
			if = {
				limit = {
					has_variable = OGAS_Cache_Auto_Road_Complete
				}
				custom_tooltip = "OGAS_Complete_desc"
			}
			else_if = {
				limit = {
					has_variable = OGAS_Cache_Auto_Road_Ongoing
				}
				custom_tooltip = "OGAS_Ongoing_desc"
			}
			else = {
				custom_tooltip = "OGAS_Cannot_Reach_desc"
			}
		}
		else_if = {
			limit = {
				has_owner = yes
				owner ?= {
					has_variable = OGAS_Auto_Road_Mode
					var:OGAS_Auto_Road_Mode = 2
				}
			}
			if = {
				limit = {
					has_variable = OGAS_Cache_Auto_Road_Complete
				}
				custom_tooltip = "OGAS_Complete_Market_Access_desc"
			}
			else_if = {
				limit = {
					has_variable = OGAS_Cache_Auto_Road_Ongoing
				}
				custom_tooltip = "OGAS_Ongoing_Market_Access_desc"
			}
			else = {
				custom_tooltip = "OGAS_Cannot_Reach_Market_Access_desc"
			}
		}
	}

	is_data_map = yes
	
	map_color = {
		#this scope is location
		if = {
			limit = {
				has_variable = OGAS_Cache_Auto_Road_Complete
			}
			value = define:NMapColors|MAP_COLOR_HIGH
		}
		else_if = {
			limit = {
				has_variable = OGAS_Cache_Auto_Road_Ongoing
			}
			value = define:NMapColors|MAP_COLOR_LOW
		}
		else = {
			value = define:NMapColors|DEFAULT_COLOR
		}
	}
}
