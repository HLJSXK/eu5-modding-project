# Hook this mod into CMF shared registration on_action.
cmf_on_mod_registration = {
	on_actions = {
		fum_on_register_cmf_mod
	}
}

fum_on_register_cmf_mod = {
	effect = {
		fum_register_cmf_mod = yes
	}
}

# Unified callback hook for setting changes, alert clicks, action bar clicks.
cmf_on_callback = {
	on_actions = {
		fum_on_cmf_callback
	}
}

fum_on_cmf_callback = {
	effect = {
		fum_handle_cmf_callback = yes
	}
}
