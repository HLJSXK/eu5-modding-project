four_yearly_country_pulse = {
	on_actions = { fum_clear_situations }
}

on_game_start_after_lobby = {
	on_actions = {
		fum_game_setup
	}
}

fum_game_setup = {
	effect = {
		remove_from_global_variable_map = {
			name = cmm
			key = flag:fum__enable_save_game_compatability
		}
		add_to_global_variable_map = {
			name = cmm
			key = flag:fum__enable_save_game_compatability
			value = 0
		}
	}
}

fum_clear_situations = {
	trigger = {
		is_ai = no
		is_situation_active = situation:great_pestilence
		or = {
			has_global_variable = fum__remove_low_importance_setting
			has_global_variable = fum__europe_centric_setting
			has_global_variable = fum__asia_centric_setting
		}
	}
	effect = {
		end_situation = situation:great_pestilence
	}
}
