namespace = EE_cabinet_events

# Target working characters.
scripted_trigger EE_has_cabinet_member_to_target = {
	any_cabinet_character = {
		count >= 1
		exists = cabinet_action
	}
}

scripted_effect EE_remove_from_cabinet = {
	ROOT = { remove_from_cabinet = prev }
	add_character_modifier = {
		modifier = refuses_to_serve_in_cabinet
		years = -1 
		mode = add_and_extend
	}
}

scripted_trigger EE_any_cabinet_character_with = {
	any_cabinet_character = {
		count >= 1
		exists = cabinet_action
		$trigger$
	}
}

scripted_effect EE_target_random_cabinet_member = {
	random_cabinet_character = {
		limit = {
			exists = cabinet_action
		}
		save_scope_as = target_character
	}
}

scripted_effect EE_target_random_cabinet_member_with = {
	random_cabinet_character = {
		limit = {
			exists = cabinet_action
			$trigger$
		}
		save_scope_as = target_character
	}
}

scripted_effect EE_cabinet_member_add_fascination_bonus = {
	add_character_modifier = {
		mode = replace
		modifier = cabinet_fascination_bonus
		years = -1
	}
}

scripted_trigger EE_may_target_HoC = {
	any_cabinet_character = {
		modifier:is_head_of_cabinet = yes
	}
}

scripted_effect EE_target_HoC = {
	random_cabinet_character = {
		limit = {
			modifier:is_head_of_cabinet = yes
		}
		save_scope_as = target_character
	}
}

scripted_trigger EE_spouse_requirements = {
	has_trait = sinner
}

scripted_trigger may_cheat_with_scope_spouse = {
	NOR = {
		is_same_gender = scope:spouse
		AND = {
			exists = dynasty
			scope:spouse.dynasty ?= dynasty
		}
	}
}
