namespace = decentralization_mission

decentralization_mission.1 = {	#Nobles Seize Opportunity to claim land from serfs in [location]
	type = country_event

	title = decentralization_mission.1.title
	desc = decentralization_mission.1.desc

	immediate = {
		random_owned_location = {
			limit = {
				this != root.capital
			}
			save_scope_as = target_location
		}

		ruler_or_regent ?= {
			save_scope_as = target_character
		}
	}

	option = {	#Allow them to claim it
		name = decentralization_mission.1.a

		scope:target_location = {
			add_location_modifier = {
				mode = add_and_extend
				modifier = decentralization_mission_nobility_owned_land
			}
			custom_tooltip = {
				text = mission_peasants_lose_satisfaction_mild
				every_pop = {
					limit = {
						owner ?= root
						pop_type = pop_type:peasants
					}
					add_pop_satisfaction = pop_satisfaction_mild_penalty
				}
			}
		}
	}

	option = {	#Restrict them
		name = decentralization_mission.1.b

		add_estate_satisfaction = {
			type = estate_type:nobles_estate
			value = estate_satisfaction_mild_penalty
		}
	}
}
