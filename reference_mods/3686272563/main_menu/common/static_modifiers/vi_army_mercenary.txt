# Applied dynamically via every_unit in monthly_country_pulse.
# size is set to war_exhaustion * 0.025 via change_unit_modifier_size,
# cancelling the war_exhaustion_impact land_morale_modifier penalty exactly.
# Only applied to pure mercenary units (no levies, no regulars).
vi_unit_we_morale_offset = {
	game_data = {
		category = unit
	}
	land_morale_modifier = 0.025
}

# VI Army Mod - Mercenary XP Initialization Flag
# Applied once to Pool Companies and Captain Companies (your own Gather + others')
# on first hire. Flags that starting XP has already been set so the monthly
# pulse never re-applies it. Long duration (1000 years) = effectively permanent.
# category = mercenary so it persists on the company object across contracts.
vi_merc_xp_initialized = {
	game_data = {
		category = mercenary
	}
	# Dummy key required: engine rejects empty modifier bodies as invalid database objects.
	# Zero value = no gameplay effect. This modifier is a flag only.
	monthly_experience_gain = 0.01
}
