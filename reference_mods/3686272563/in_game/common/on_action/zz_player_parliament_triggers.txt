# Player Parliament Triggers
# - on_ruler_death: root = country
# - on_war_declared: scope:recipient = country who gets declared on
# - on_civil_war_start: root = original country targeted by the civil war (vanilla hardcoded comment)

on_ruler_death = {
    on_actions = {
        ppt_on_ruler_death_unlock_call_parliament
    }
}

ppt_on_ruler_death_unlock_call_parliament = {
    effect = {
        root = {
            if = {
                limit = {
                    is_ai = no
                    has_parliament = yes
                    is_active_parliament = no
                    NOT = { modifier:parliament_abolished = yes }
                }
                set_variable = { name = ppt_call_parliament_unlocked value = 1 }
            }
        }
    }
}

on_war_declared = {
    on_actions = {
        ppt_on_war_declared_unlock_call_parliament
    }
}

ppt_on_war_declared_unlock_call_parliament = {
    effect = {
        if = {
            limit = { exists = scope:recipient }
            scope:recipient = {
                if = {
                    limit = {
                        is_ai = no
                        has_parliament = yes
                        is_active_parliament = no
                        NOT = { modifier:parliament_abolished = yes }
                    }
                    set_variable = { name = ppt_call_parliament_unlocked value = 1 }
                }
            }
        }
    }
}

# Civil wars do NOT necessarily fire on_war_declared; they use on_civil_war_start in vanilla.
on_civil_war_start = {
    on_actions = {
        ppt_on_civil_war_start_unlock_call_parliament
    }
}

ppt_on_civil_war_start_unlock_call_parliament = {
    effect = {
        root = {
            if = {
                limit = {
                    is_ai = no
                    has_parliament = yes
                    is_active_parliament = no
                    NOT = { modifier:parliament_abolished = yes }
                }
                set_variable = { name = ppt_call_parliament_unlocked value = 1 }
            }
        }
    }
}
