    effect = {
        # Fire Stage 5 event based on gender
        if = {
            limit = {
                scope:target_1 = { is_female = no }
            }

            trigger_event_non_silently = wedding_costs.5
        }
        else = {
            # Female - dowry costs event
            trigger_event_non_silently = wedding_costs.6
        }
    }
