limitation_of_crown_authority = {
    type = parliament
    sound = PARLIAMENT_ASK_FOR_LAW_CHANGES

    ai_tick = daily
    ai_tick_frequency = 100
    automation_tick = daily
    automation_tick_frequency = 7
    player_automated_category = parliament

    allow = {
        scope:actor = { is_active_parliament = yes }

        # Blocked while the 3-year modifier is active
        custom_tooltip = {
            text = limitation_of_crown_authority_req_not_active_tt
            scope:actor = { NOT = { has_country_modifier = limitation_of_crown_authority_modifier } }
        }

        # Fixed support requirement: 25%
        custom_tooltip = {
            text = limitation_of_crown_authority_req_support_tt
            scope:actor = {
                parliament_issue_support >= 0.25
            }
        }
    }

    effect = {
        scope:actor = {
            # Spend the same amount of support: -25%
            change_parliament_issue_support = {
                value = 0.25
                multiply = -1
            }

            add_country_modifier = {
                modifier = limitation_of_crown_authority_modifier
                years = 3
                mode = add
            }

            # Immediate estate satisfaction increase for all estates
            add_all_estate_satisfaction = { value = 0.10 }
        }
    }

    ai_will_do = { add = -1000 }
}
