recoinage = {
    type = parliament
    sound = PARLIAMENT_ASK_FOR_LAW_CHANGES

    ai_tick = daily
    ai_tick_frequency = 100
    automation_tick = daily
    automation_tick_frequency = 7
    player_automated_category = parliament

    allow = {
        hidden_trigger = {
            scope:actor = { is_ai = no }
        }

        scope:actor = { is_active_parliament = yes }

        custom_tooltip = {
            text = recoinage_req_once_tt
            scope:actor = { NOT = { has_variable = recoinage_used_this_parliament } }
        }

        custom_tooltip = {
            text = recoinage_req_gold_tt
            scope:actor = {
                gold >= {
                    value = country_tax_base
                    multiply = 12
                }
            }
        }

        custom_tooltip = {
            text = recoinage_req_no_bankruptcy_tt
            scope:actor = { is_during_bankruptcy = no }
        }

        custom_tooltip = {
            text = recoinage_req_precious_metals_tt
            scope:actor = {
                any_market_present_in_country = {
                    OR = {
                        is_produced_in_market = goods:goods_gold
                        is_traded_in_market = goods:goods_gold
                        is_produced_in_market = goods:silver
                        is_traded_in_market = goods:silver
                    }
                }
            }
        }
    }

    effect = {
        scope:actor = {
            # Per-session lock
            set_variable = { name = recoinage_used_this_parliament value = 1 }

            # Cost: tax_base * 12
            save_temporary_scope_value_as = {
                name = vi_recoinage_cost
                value = {
                    value = country_tax_base
                    multiply = 12
                }
            }
            add_gold = { value = scope:vi_recoinage_cost multiply = -1 }

            # Prestige +3 absolute
            add_prestige = 3

            # Single combined modifier: tax efficiency + inflation reduction + minting threshold
            add_country_modifier = {
                modifier = recoinage_combined_effects
                years = 10
                mode = add_and_extend
            }
        }
    }

    ai_will_do = { add = -1000 }
}
