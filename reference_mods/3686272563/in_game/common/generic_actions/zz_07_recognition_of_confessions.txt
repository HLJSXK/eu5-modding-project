recognition_of_confessions = {
    type = parliament
    sound = PARLIAMENT_ASK_FOR_LAW_CHANGES

    ai_tick = daily
    ai_tick_frequency = 100
    automation_tick = daily
    automation_tick_frequency = 7
    player_automated_category = parliament

    allow = {
        hidden_trigger = {
            scope:actor = { is_ai = no } # Player-only
        }

        scope:actor = { is_active_parliament = yes }

        # Blocked while the 10-year modifier is active
        custom_tooltip = {
            text = recognition_of_confessions_req_not_active_tt
            scope:actor = { NOT = { has_country_modifier = recognition_of_confessions_modifier } }
        }

        # Required support scales with Religious Unity: RU * 0.75
        custom_tooltip = {
            text = recognition_of_confessions_req_support_tt
            scope:actor = {
                parliament_issue_support >= {
                    value = religious_unity
                    multiply = 0.75
                }
            }
        }
    }

    effect = {
        scope:actor = {
            # Spend the same amount of support: -(RU * 0.75)
            change_parliament_issue_support = {
                value = religious_unity
                multiply = -0.75
            }

            add_country_modifier = {
                modifier = recognition_of_confessions_modifier
                years = 10
                mode = add
            }

            # Immediate estate satisfaction changes
            add_estate_satisfaction = { type = estate_type:clergy_estate  value = -0.10 }
            add_estate_satisfaction = { type = estate_type:dhimmi_estate  value =  0.05 }
        }
    }

    ai_will_do = { add = -1000 }
}
