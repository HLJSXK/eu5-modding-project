location_gate2_allows_town = {
	NOT = { topography = mountains }
	NOT = { climate = arctic }
	OR = {
		topography = hills
		topography = plateau
		AND = {
			topography = wetlands
			NOT = { climate = tropical }
		}
		topography = flatland
	}
}

location_gate2_allows_town_arctic = {
	NOT = { topography = mountains }
	climate = arctic
	modifier:natural_harbor_suitability >= 0.25
	OR = {
		topography = hills
		topography = plateau
		topography = wetlands
		topography = flatland
	}
}

location_gate2_allows_city = {
	NOT = { climate = arctic }
	NOT = { topography = mountains }
	OR = {
		AND = { topography = hills climate = mediterranean }
		AND = { topography = hills climate = continental }
		AND = { topography = hills climate = oceanic }
		AND = { topography = hills climate = subtropical }
		AND = { topography = plateau climate = mediterranean }
		AND = { topography = plateau climate = subtropical }
		topography = flatland
		AND = { topography = wetlands climate = mediterranean }
		AND = { topography = wetlands climate = continental }
		AND = { topography = wetlands climate = oceanic }
	}
}

location_gate3_allows_town = {
	NOT = { vegetation = jungle }
	NOT = { vegetation = desert }
	OR = {
		vegetation = sparse
		vegetation = grasslands
		vegetation = farmland
		vegetation = woods
		vegetation = forest
	}
}

location_gate3_allows_town_desert = {
	vegetation = desert
	OR = {
		has_river = yes
		modifier:natural_harbor_suitability >= 0.25
	}
}

location_gate3_allows_city = {
	OR = {
		vegetation = grasslands
		vegetation = farmland
		AND = {
			vegetation = woods
			NOT = { topography = mountains }
			NOT = { climate = arctic }
			OR = {
				AND = { topography = hills climate = mediterranean }
				AND = { topography = hills climate = continental }
				AND = { topography = hills climate = oceanic }
				AND = { topography = hills climate = subtropical }
				AND = { topography = plateau climate = mediterranean }
				AND = { topography = plateau climate = subtropical }
				topography = flatland
			}
			OR = {
				has_river = yes
				modifier:natural_harbor_suitability >= 0.50
			}
		}
	}
}

location_allows_town_final = {
	OR = {
		AND = {
			location_gate2_allows_town = yes
			location_gate3_allows_town = yes
		}
		AND = {
			location_gate2_allows_town_arctic = yes
			location_gate3_allows_town = yes
		}
		AND = {
			location_gate2_allows_town = yes
			location_gate3_allows_town_desert = yes
		}
		location_key = madrid
	}
}

location_allows_city_final = {
	OR = {
		AND = {
			location_gate2_allows_city = yes
			location_gate3_allows_city = yes
		}
		location_key = madrid
	}
}
