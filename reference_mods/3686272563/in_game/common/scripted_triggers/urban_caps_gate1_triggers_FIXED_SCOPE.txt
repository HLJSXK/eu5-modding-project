# Gate 1 Urban Caps - Fixed scope issues
# Use any_location_in_province with count to check province size

# ============================================
# CAN UPGRADE TO TOWN - Gate 1 Rules
# ============================================

can_upgrade_to_town_under_province_cap = {
    exists = province
    province = {
        OR = {
            # 1 location: Allow if no towns AND no cities
            AND = {
                any_location_in_province = {
                    always = yes
                    count = 1
                }
                NOT = {
                    any_location_in_province = {
                        location_rank = location_rank:town
                    }
                }
                NOT = {
                    any_location_in_province = {
                        location_rank = location_rank:city
                    }
                }
            }
            
            # 2 locations: Allow if no towns AND no cities
            AND = {
                any_location_in_province = {
                    always = yes
                    count = 2
                }
                NOT = {
                    any_location_in_province = {
                        location_rank = location_rank:town
                    }
                }
                NOT = {
                    any_location_in_province = {
                        location_rank = location_rank:city
                    }
                }
            }
            
            # 3 locations: Allow if no towns AND no cities
            AND = {
                any_location_in_province = {
                    always = yes
                    count = 3
                }
                NOT = {
                    any_location_in_province = {
                        location_rank = location_rank:town
                    }
                }
                NOT = {
                    any_location_in_province = {
                        location_rank = location_rank:city
                    }
                }
            }
            
            # 4 locations: Block if town >= 2 OR city >= 2 OR (town >= 1 AND city >= 1)
            AND = {
                any_location_in_province = {
                    always = yes
                    count = 4
                }
                # Less than 2 towns
                NOT = {
                    any_location_in_province = {
                        location_rank = location_rank:town
                        count >= 2
                    }
                }
                # Less than 2 cities
                NOT = {
                    any_location_in_province = {
                        location_rank = location_rank:city
                        count >= 2
                    }
                }
                # NOT (has town AND has city)
                NOT = {
                    AND = {
                        any_location_in_province = {
                            location_rank = location_rank:town
                        }
                        any_location_in_province = {
                            location_rank = location_rank:city
                        }
                    }
                }
            }
            
            # 5 locations: Block if town >= 2 OR city >= 2 OR (town >= 1 AND city >= 1)
            AND = {
                any_location_in_province = {
                    always = yes
                    count = 5
                }
                # Less than 2 towns
                NOT = {
                    any_location_in_province = {
                        location_rank = location_rank:town
                        count >= 2
                    }
                }
                # Less than 2 cities
                NOT = {
                    any_location_in_province = {
                        location_rank = location_rank:city
                        count >= 2
                    }
                }
                # NOT (has town AND has city)
                NOT = {
                    AND = {
                        any_location_in_province = {
                            location_rank = location_rank:town
                        }
                        any_location_in_province = {
                            location_rank = location_rank:city
                        }
                    }
                }
            }
            
            # 6+ locations: Block if town >= 3 OR city >= 3 OR (town >= 1 AND city >= 2) OR (town >= 2 AND city >= 1)
            AND = {
                any_location_in_province = {
                    always = yes
                    count >= 6
                }
                # Less than 3 towns
                NOT = {
                    any_location_in_province = {
                        location_rank = location_rank:town
                        count >= 3
                    }
                }
                # Less than 3 cities
                NOT = {
                    any_location_in_province = {
                        location_rank = location_rank:city
                        count >= 3
                    }
                }
                # NOT (1+ town AND 2+ cities)
                NOT = {
                    AND = {
                        any_location_in_province = {
                            location_rank = location_rank:town
                        }
                        any_location_in_province = {
                            location_rank = location_rank:city
                            count >= 2
                        }
                    }
                }
                # NOT (2+ towns AND 1+ city)
                NOT = {
                    AND = {
                        any_location_in_province = {
                            location_rank = location_rank:town
                            count >= 2
                        }
                        any_location_in_province = {
                            location_rank = location_rank:city
                        }
                    }
                }
            }
        }
    }
}

# ============================================
# CAN UPGRADE TO CITY - Gate 1 Rules
# ============================================

can_upgrade_to_city_under_province_cap = {
    exists = province
    province = {
        # Must have at least 1 town to upgrade
        any_location_in_province = {
            location_rank = location_rank:town
        }
        # Must not already have a city
        NOT = {
            any_location_in_province = {
                location_rank = location_rank:city
            }
        }
    }
}
