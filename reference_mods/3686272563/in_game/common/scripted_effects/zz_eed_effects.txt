# Estate Expense Distribution - Court of Chancery Effects
# 60% of CoC expense -> add_temporary_demand at capital market
# 40% -> commissions: Clergy 30%, Labourers 70%
#
# Conditional goods (capital market presence required):
#   tea, coffee, chili, cocoa        - produced OR traded in capital market
#   wild_game                        - produced in capital market only
#   millet, legumes, olives, rice    - produced OR traded in capital market
#   potato, maize                    - produced OR traded in capital market

eed_distribute_court_expense_effect = {
	# Total CoC expense this tick
	save_temporary_scope_value_as = {
		name = coc_total
		value = {
			value = country_economical_base
			multiply = modifier:court_spending_cost
			multiply = court_maintenance
			min = 0
		}
	}
	# 60% goods demand budget, 40% commissions
	save_temporary_scope_value_as = {
		name = coc_demand_budget
		value = { value = scope:coc_total multiply = 0.6 min = 0 }
	}
	save_temporary_scope_value_as = {
		name = coc_commissions
		value = { value = scope:coc_total multiply = 0.4 min = 0 }
	}
	# Distribute commissions: Clergy 30%, Labourers 70%
	add_gold_to_estate = {
		estate_type = estate_type:clergy_estate
		value = { value = scope:coc_commissions multiply = -0.3 }
	}
	add_gold_to_estate = {
		estate_type = estate_type:peasants_estate
		value = { value = scope:coc_commissions multiply = -0.7 }
	}
	# Inject court goods demand into capital market
	eed_coc_place_market_demand_effect = yes
}

	eed_dip_place_market_demand_effect = yes
}

# ── Cultural Maintenance Distribution ────────────────────────────────────────
# Only fires if modifier:has_cultural_maintenance = yes, slider > 0, and has at least 1 artist
# Formula: cul_total = SUM(artist.yearly_salary) × cultural_maintenance (slider 0-1)
# 100% -> Burghers estate gold (artist patrons, cultural entrepreneurs)

eed_distribute_cultural_expense_effect = {
	# Only fires if cultural maintenance is unlocked and slider > 0
	if = {
		limit = {
			modifier:has_cultural_maintenance = yes
			cultural_maintenance > 0
			any_artist = { count >= 1 }
		}
		# Sum all artist salaries into accumulator variable
		# yearly_salary is misleadingly named - returns the MONTHLY cost per artist
		set_variable = { name = eed_culture_salary_sum value = 0 }
		every_artist = {
			root = {
				change_variable = {
					name = eed_culture_salary_sum
					add = prev.yearly_salary
				}
			}
		}
		# Total cultural expense = sum of monthly salaries × slider position
		save_temporary_scope_value_as = {
			name = cul_total
			value = {
				value = var:eed_culture_salary_sum
				multiply = cultural_maintenance
				min = 0
			}
		}
		# 100% to Burghers (artist patrons, cultural entrepreneurs)
		add_gold_to_estate = {
			estate_type = estate_type:burghers_estate
			value = { value = scope:cul_total multiply = -1.0 }
		}
	}
}

# ── Fort Maintenance Distribution ────────────────────────────────────────────
# Iterate owned locations with forts, identify tier by has_building, accumulate
# soldier wage cost into eed_fort_soldier_wages variable on the country.
# Soldier counts (confirmed in-game): Stockade 100, Castle 500, Bastion 1000,
# Star Fort 2000, Fortress 4000. Wage rate: 0.002 gold/soldier.
# Raw wages: Stockade 0.2, Castle 1.0, Bastion 2.0, Star Fort 4.0, Fortress 8.0
# Multiply by fort_maintenance slider -> actual soldier wage expense.
# 50% -> Nobles (officers), 50% -> Peasants (common soldiers)

eed_distribute_fort_expense_effect = {
	# Reset accumulator
	set_variable = { name = eed_fort_soldier_wages value = 0 }

	# Iterate all owned locations with a fort - identify tier and add soldier wage cost
	every_owned_location = {
		limit = { has_fort = yes }
		# Fortress:   4000 soldiers × 0.002 = 8.0 gold
		if = {
			limit = { has_building = building_type:fortress }
			root = { change_variable = { name = eed_fort_soldier_wages add = 8.0 } }
		}
		# Star Fort:  2000 soldiers × 0.002 = 4.0 gold
		else_if = {
			limit = { has_building = building_type:star_fort }
			root = { change_variable = { name = eed_fort_soldier_wages add = 4.0 } }
		}
		# Bastion:    1000 soldiers × 0.002 = 2.0 gold
		else_if = {
			limit = { has_building = building_type:bastion }
			root = { change_variable = { name = eed_fort_soldier_wages add = 2.0 } }
		}
		# Castle:     500 soldiers × 0.002 = 1.0 gold
		else_if = {
			limit = { has_building = building_type:castle }
			root = { change_variable = { name = eed_fort_soldier_wages add = 1.0 } }
		}
		# Stockade:   100 soldiers × 0.002 = 0.2 gold
		else_if = {
			limit = { has_building = building_type:stockade }
			root = { change_variable = { name = eed_fort_soldier_wages add = 0.2 } }
		}
	}

	# Apply fort_maintenance slider to get actual wage expense
	save_temporary_scope_value_as = {
		name = fort_wage_total
		value = {
			value = var:eed_fort_soldier_wages
			multiply = fort_maintenance
			min = 0
		}
	}

	# 50% to Nobles (officers)
	add_gold_to_estate = {
		estate_type = estate_type:nobles_estate
		value = { value = scope:fort_wage_total multiply = -0.5 }
	}
	# 50% to Peasants (common soldiers)
	add_gold_to_estate = {
		estate_type = estate_type:peasants_estate
		value = { value = scope:fort_wage_total multiply = -0.5 }
	}
}

# ── Stability Investment Distribution ────────────────────────────────────────
# Approximates the gold spent on stability investment each month using the
# stability delta method, then distributes it:
#   60% -> capital market demand (stability basket)
#   40% -> Clergy estate gold
#
# Method:
#   Each month we store stability at end of pulse (eed_last_stability).
#   Next month: raw_delta = stability_now - eed_last_stability
#   Stability_Change_B = raw_delta - modifier:stability_investment
#                        + (eed_last_stability x modifier:stability_decay)
#   This isolates the slider investing effect. Clamped to [0, 0.5].
#   slider_approx = stab_change_b / 0.5  (fraction of max investment)
#
#   Actual gold cost formula (derived from tooltip):
#     base_cost_100pct = country_economical_base x 0.10 x (1 + modifier:stability_cost)
#     stab_cost = slider_approx x base_cost_100pct
#               = stab_change_b x country_economical_base x 0.20 x (1 + modifier:stability_cost)
#   Verified: 0.50 x 304.10 x 0.20 x (1 - 0.2407) = 23.09 gold
#
# Player-only. Skips on first run (no stored variable yet).

eed_distribute_stability_expense_effect = {
	if = {
		limit = {
			is_ai = no
			has_variable = eed_last_stability
		}
		# Compute Stability_Change_B (slider fraction x 0.5, clamped 0-0.5)
		save_temporary_scope_value_as = {
			name = stab_change_b
			value = {
				value = stability
				subtract = var:eed_last_stability
				subtract = modifier:stability_investment
				add = {
					value = var:eed_last_stability
					multiply = modifier:stability_decay
				}
				min = 0
				max = 0.5
			}
		}
		# Only proceed if there is meaningful investment (change_b > 0)
		if = {
			limit = { scope:stab_change_b > 0 }
			# slider_approx = stab_change_b / 0.5  (0-1 fraction of max investment)
			# base_cost_100pct = country_economical_base x 0.10 x (1 + modifier:stability_cost)
			# stab_cost = slider_approx x base_cost_100pct
			save_temporary_scope_value_as = {
				name = slider_approx
				value = {
					value = scope:stab_change_b
					divide = 0.5
					min = 0
					max = 1
				}
			}
			save_temporary_scope_value_as = {
				name = stab_cost
				value = {
					value = scope:slider_approx
					multiply = country_economical_base
					multiply = 0.10
					multiply = {
						value = 1
						add = modifier:stability_cost
					}
					min = 0
				}
			}
			# 60% -> market demand budget
			save_temporary_scope_value_as = {
				name = stab_demand_budget
				value = { value = scope:stab_cost multiply = 0.6 min = 0 }
			}
			# 40% -> Clergy estate
			add_gold_to_estate = {
				estate_type = estate_type:clergy_estate
				value = { value = scope:stab_cost multiply = -0.4 }
			}
			# Inject goods demand into capital market
			eed_stab_place_market_demand_effect = yes
		}
	}
	# Store this month's stability for next month's delta calculation
	set_variable = { name = eed_last_stability value = stability }
}

# army_maintenance = slider position (0.5 to 1.0)
# modifier:army_maintenance_cost = sum of all army_maintenance_cost modifiers (percent, base 0)

# Split: 50% Nobles (officers, cavalry owners), 50% Peasants (common soldiers)

# ── Navy Maintenance Distribution ────────────────────────────────────────────
# Same approach as army. Navy cost per ship = price_gold * crew_size.
# navy_size = SUM(crew_size), so cost per navy_size unit = price_gold directly (no K factor).
# navy_maintenance same scale as army_maintenance: 1.0 at minimum (50%), 2.0 at maximum.
#
# Categories: heavy ships = 0.50 gold/crew-unit, all others = 0.25, panokseon = 0.45.
# Split: 50% Nobles (admirals, captains), 50% Peasants (sailors, oarsmen)

			is_ai = no

eed_stab_place_market_demand_effect = {
	capital.market = {
		# Remove previous entries to avoid stacking
		remove_temporary_demand = demand:eed_stab_base
		remove_temporary_demand = demand:eed_stab_base_muslim
		remove_temporary_demand = demand:eed_stab_rice
		remove_temporary_demand = demand:eed_stab_potato
		remove_temporary_demand = demand:eed_stab_maize

		# Muslim: no beer/wine, coffee replaces wine
		if = {
			limit = {
				prev.religion = { group = religion_group:muslim }
			}
			add_temporary_demand = {
				type = demand:eed_stab_base_muslim
				scale = scope:stab_demand_budget
				months = 2
			}
		}
		else = {
			add_temporary_demand = {
				type = demand:eed_stab_base
				scale = scope:stab_demand_budget
				months = 2
			}
		}

		# Rice: only if produced in capital market
		if = {
			limit = { is_produced_in_market = goods:rice }
			add_temporary_demand = {
				type = demand:eed_stab_rice
				scale = scope:stab_demand_budget
				months = 2
			}
		}

		# Potato: produced or traded in capital market
		if = {
			limit = {
				OR = {
					is_produced_in_market = goods:potato
					is_traded_in_market = goods:potato
				}
			}
			add_temporary_demand = {
				type = demand:eed_stab_potato
				scale = scope:stab_demand_budget
				months = 2
			}
		}

		# Maize: produced or traded in capital market
		if = {
			limit = {
				OR = {
					is_produced_in_market = goods:maize
					is_traded_in_market = goods:maize
				}
			}
			add_temporary_demand = {
				type = demand:eed_stab_maize
				scale = scope:stab_demand_budget
				months = 2
			}
		}
	}
}
