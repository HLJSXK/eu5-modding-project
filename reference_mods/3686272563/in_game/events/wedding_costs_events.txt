namespace = wedding_costs

# Stage 5 - Royal Marriage Payment (Male character marries)
wedding_costs.5 = {
    type = country_event
    title = wedding_costs.5.t
    desc = wedding_costs.5.desc
    
    option = {
        name = wedding_costs.5.a
        
        # Only apply to player countries
        if = {
            limit = { is_ai = no }
            
            # Deduct wedding cost
            change_gold_effect = { scale = -6 }
            
            # Add prestige based on rank of scope:recipient (the other country)
            if = {
                limit = {
                    scope:recipient = {
                        country_rank = country_rank:rank_empire
                    }
                }
                add_prestige = { value = 2 desc = "ROYAL_WEDDING_SON_PRESTIGE" }
            }
            else_if = {
                limit = {
                    scope:recipient = {
                        country_rank = country_rank:rank_kingdom
                    }
                }
                add_prestige = { value = 1.5 desc = "ROYAL_WEDDING_SON_PRESTIGE" }
            }
            else_if = {
                limit = {
                    scope:recipient = {
                        country_rank = country_rank:rank_duchy
                    }
                }
                add_prestige = { value = 1 desc = "ROYAL_WEDDING_SON_PRESTIGE" }
            }
            else = {
                # County rank (rank_county) or fallback
                add_prestige = { value = 0.5 desc = "ROYAL_WEDDING_SON_PRESTIGE" }
            }
        }
    }
}
