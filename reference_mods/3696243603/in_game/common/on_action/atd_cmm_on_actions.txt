# Hook this mod into CMF shared registration on_action.
cmf_on_mod_registration = {
	on_actions = {
		atd_on_register_cmf_mod
	}
}

atd_on_register_cmf_mod = {
	effect = {
		atd_register_cmf_mod = yes
	}
}

# Unified callback hook for setting changes, alert clicks, action bar clicks.
cmf_on_callback = {
	on_actions = {
		atd_on_cmf_callback
	}
}

atd_on_cmf_callback = {
	effect = {
		atd_handle_cmf_callback = yes
	}
}
