# Callback handler for cmf_on_callback.
# var:cmf_callback is the flag of the setting, alert, or action bar element that was interacted with.
# Scope: country
atd_handle_cmf_callback = {
	switch = {
		trigger = var:cmf_callback
		# General (general) Tab
		## Reserve (reserve) Group
		flag:atd__min_diplomats = {
			cmm_sync_setting_alias = {
				setting = atd__min_diplomats
				alias = atd_min_diplomats
			}
		}
		flag:atd__min_diplomat_gain = {
			cmm_sync_setting_alias = {
				setting = atd__min_diplomat_gain
				alias = atd_min_diplomat_gain
			}
		}
	}
}
