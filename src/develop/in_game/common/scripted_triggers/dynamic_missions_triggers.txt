# Dynamic Missions: Scripted Triggers (Unified Mission Manager)

# =============================================================================
# MISSION START TRIGGERS
# =============================================================================

can_start_develop_city_mission = {
	any_owned_location = {
		OR = {
			location_rank = location_rank:town
			location_rank = location_rank:city
		}
		prosperity < 100
	}
	gold >= 300
}

can_start_large_research_project = {
	OR = {
		any_subject = { exists = yes }
		any_neighbor_country = { exists = yes }
	}
}

can_start_establish_new_city_mission = {
	any_owned_location = {
		location_rank = location_rank:town
	}
	NOT = { has_variable = has_completed_establish_city_mission }
	gold >= 300
}

can_start_claim_province_mission = {
	any_neighbor_country = {
		exists = yes
		NOT = { this = root }
		NOT = { is_subject_of = root }
	}
	gold >= 200
	prestige >= 10
}

# =============================================================================
# GENERAL ELIGIBILITY
# =============================================================================

can_receive_dynamic_mission = {
	is_ai = no
	NOT = { has_variable = dm_active_mission }
	OR = {
		can_start_develop_city_mission = yes
		can_start_large_research_project = yes
		can_start_establish_new_city_mission = yes
		can_start_claim_province_mission = yes
	}
}
