# Dynamic Missions — Scripted GUIs
# Invoked by situation panel buttons.

# Toggle research funding modifier on/off.
dm_toggle_research_funding_sgui = {
	scope = country
	is_shown = { has_variable = dm_mission_large_research }
	effect = {
		if = {
			limit = { has_country_modifier = dm_increased_funding_modifier }
			remove_country_modifier = dm_increased_funding_modifier
			remove_variable = dm_research_funding_on
		}
		else = {
			add_country_modifier = {
				modifier = dm_increased_funding_modifier
				years = -1
				mode = add_and_extend
			}
			set_variable = dm_research_funding_on
		}
	}
}

# Toggle urbanization investment modifier on/off.
dm_toggle_urbanization_investment_sgui = {
	scope = country
	is_shown = { has_variable = dm_mission_promote_urbanization }
	effect = {
		if = {
			limit = { has_country_modifier = dm_urbanization_investment_active }
			remove_country_modifier = dm_urbanization_investment_active
			remove_variable = dm_urbanization_investment_on
			var:dm_target_location = {
				remove_location_modifier = dm_urbanization_investment_bonus
			}
		}
		else = {
			add_country_modifier = {
				modifier = dm_urbanization_investment_active
				years = -1
				mode = add_and_extend
			}
			set_variable = dm_urbanization_investment_on
			var:dm_target_location = {
				add_location_modifier = {
					modifier = dm_urbanization_investment_bonus
					years = -1
					mode = add_and_extend
				}
			}
		}
	}
}

# Execute border reconnaissance (one-time, 1-year cooldown).
dm_claim_border_recon_sgui = {
	scope = country
	is_shown = { has_variable = dm_mission_claim_province }
	is_valid = {
		NOT = { has_country_modifier = dm_border_reconnaissance_cooldown }
		manpower >= 200
	}
	effect = {
		add_manpower = -200
		add_army_tradition = 5
		add_country_modifier = {
			modifier = dm_border_reconnaissance_cooldown
			years = 1
			mode = add_and_extend
		}
		custom_tooltip = dm_claim_border_reconnaissance_tooltip
	}
}
