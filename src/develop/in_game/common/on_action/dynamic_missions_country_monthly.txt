# Dynamic Missions — Monthly Country Pulse
#
# Fires dm_monthly_tick on the player country each month while a mission is active.
# All progress tracking (dm_months_elapsed, dm_bar_value) is driven from here.

monthly_country_pulse = {
	trigger = {
		has_variable = dm_active_mission
	}
	effect = {
		dm_monthly_tick = yes
	}
}
