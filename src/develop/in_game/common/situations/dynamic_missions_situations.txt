# Dynamic Missions - Situations
# Global situations that players can join as participants

# Develop City Mission
develop_city_situation = {
	# Always active, never ends
	can_start = {
		always = yes
	}
	
	can_end = {
		always = no
	}
	
	visible = {
		always = yes
	}
	
	monthly_spawn_chance = {
		add = 100  # Spawn immediately at game start
	}
	
	on_start = {
		# Initialize global situation
		set_variable = { name = situation_active value = yes }
	}
	
	on_monthly = {
		# Check each participant's progress
		every_participant = {
			# Check if target location reached prosperity goal
			if = {
				limit = {
					has_variable = dm_target_location
					var:dm_target_location ?= {
						prosperity >= 100
					}
				}
				
				# Mission success
				set_variable = { name = dm_mission_success value = yes }
				trigger_event_non_silently = dynamic_missions.3
			}
			
			# Check time limit (60 months = 5 years)
			if = {
				limit = {
					has_variable = dm_months_elapsed
					var:dm_months_elapsed >= 60
					NOT = { has_variable = dm_mission_success }
				}
				
				# Mission failed
				set_variable = { name = dm_mission_failed value = yes }
				trigger_event_non_silently = dynamic_missions.4
			}
			
			# Increment month counter
			if = {
				limit = { has_variable = dm_months_elapsed }
				change_variable = { name = dm_months_elapsed add = 1 }
			}
		}
	}
}
