# Dynamic Missions - Single Unified Situation
#
# One always-active situation acts as mission manager UI.
# Mission runtime state is stored on COUNTRY variables.
#
# dm_gui_mission_type routing:
#   0 = idle
#   1 = develop_city
#   2 = large_research_project
#   3 = establish_new_city
#   4 = claim_province

dynamic_missions_situation = {
	monthly_spawn_chance = 1
	hint_tag = hint_dynamic_missions

	can_start = {
		always = yes
	}

	can_end = {
		always = no
	}

	visible = {
		is_ai = no
	}

	on_start = {
		every_country = {
			limit = { is_ai = no }
			set_variable = { name = dm_gui_mission_type value = 0 }
			# First prompt shortly after game start so players can verify flow early.
			trigger_event_non_silently = { id = dynamic_missions.1 days = 30 }
		}
	}

	on_monthly = {
		every_country = {
			limit = { is_ai = no }

			# No active mission -> idle panel
			if = {
				limit = { NOT = { has_variable = dm_active_mission } }
				set_variable = { name = dm_gui_mission_type value = 0 }
			}

			# Mission 1: Develop City
			else_if = {
				limit = { has_variable = dm_mission_develop_city }
				set_variable = { name = dm_gui_mission_type value = 1 }

				if = {
					limit = {
						has_variable = dm_target_location
						var:dm_target_location ?= { prosperity >= 100 }
					}
					trigger_event_non_silently = dynamic_missions.3
				}
				else_if = {
					limit = {
						has_variable = dm_months_elapsed
						var:dm_months_elapsed >= 60
					}
					trigger_event_non_silently = dynamic_missions.4
				}
				else = {
					change_variable = { name = dm_months_elapsed add = 1 }
				}
			}

			# Mission 2: Large Research Project
			else_if = {
				limit = { has_variable = dm_mission_large_research }
				set_variable = { name = dm_gui_mission_type value = 2 }

				# Base progress tick
				if = {
					limit = { NOT = { has_variable = dm_research_progress } }
					set_variable = { name = dm_research_progress value = 0 }
				}
				change_variable = { name = dm_research_progress add = 1 }

				# Optional funding acceleration
				if = {
					limit = { has_country_modifier = dm_increased_funding_modifier }
					change_variable = { name = dm_research_progress add = 2 }
				}

				if = {
					limit = { var:dm_research_progress >= 100 }
					trigger_event_non_silently = dynamic_missions.23
				}
				else_if = {
					limit = {
						has_variable = dm_months_elapsed
						var:dm_months_elapsed >= 60
					}
					trigger_event_non_silently = dynamic_missions.24
				}
				else = {
					change_variable = { name = dm_months_elapsed add = 1 }
				}
			}

			# Mission 3: Establish New City
			else_if = {
				limit = { has_variable = dm_mission_establish_city }
				set_variable = { name = dm_gui_mission_type value = 3 }

				if = {
					limit = {
						has_variable = dm_target_location
						var:dm_target_location ?= { location_rank = location_rank:city }
					}
					trigger_event_non_silently = dynamic_missions.33
				}
				else_if = {
					limit = {
						has_variable = dm_months_elapsed
						var:dm_months_elapsed >= 120
					}
					trigger_event_non_silently = dynamic_missions.34
				}
				else = {
					change_variable = { name = dm_months_elapsed add = 1 }
				}
			}

			# Mission 4: Claim Province (simplified: one target location)
			else_if = {
				limit = { has_variable = dm_mission_claim_province }
				set_variable = { name = dm_gui_mission_type value = 4 }

				if = {
					limit = {
						has_variable = dm_target_location
						var:dm_target_location ?= { owner = root }
					}
					trigger_event_non_silently = dynamic_missions.50
				}
				else_if = {
					limit = {
						has_variable = dm_months_elapsed
						var:dm_months_elapsed >= 60
					}
					trigger_event_non_silently = dynamic_missions.51
				}
				else = {
					change_variable = { name = dm_months_elapsed add = 1 }
				}
			}

			# Recover from inconsistent state
			else = {
				dm_clear_all_mission_state = yes
			}
		}
	}
}
