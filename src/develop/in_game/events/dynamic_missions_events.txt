# Dynamic Missions: Select Target Town for New City
dynamic_missions.30 = {
	type = country_event
	title = dynamic_missions.30.title
	desc = dynamic_missions.30.desc
	fire_only_once = no
	
	illustration_tags = {
		10 = exterior
		10 = town
	}
	
	immediate = {
		# Build list of eligible towns
		every_owned_location = {
			limit = {
				location_rank = location_rank:town
				# is_occupied = no
				# is_under_siege = no
			}
			add_to_list = eligible_towns
		}
	}
	
	# Dynamic options for each eligible town
	# Note: In actual implementation, you may need to use a fixed number of options
	# or implement a selection mechanism via variables
	
	# Option 1: Select Capital (if it's a town)
	option = {
		name = dynamic_missions.30.option_capital
		trigger = {
			capital = {
				location_rank = location_rank:town
			}
		}
		
			capital = {
				save_scope_as = target_town
				set_variable = { name = selected_for_new_city value = yes }
			}
			
			# Join the global situation as participant
			situation:establish_new_city_situation = {
				add_participant = root
			}
			# Initialize mission variables
			set_variable = { name = dm_months_elapsed value = 0 }
		
		custom_tooltip = dynamic_missions.30.option_capital.tooltip
	}
	
	# Option 2: Select First Eligible Town
	option = {
		name = dynamic_missions.30.option_first_town
		trigger = {
			any_in_list = {
				list = eligible_towns
				count >= 1
			}
		}
		
			random_in_list = {
				list = eligible_towns
				save_scope_as = target_town
				set_variable = { name = selected_for_new_city value = yes }
			}
			
			# Join the global situation as participant
			situation:establish_new_city_situation = {
				add_participant = root
			}
			# Initialize mission variables
			set_variable = { name = dm_months_elapsed value = 0 }
		
		custom_tooltip = dynamic_missions.30.option_first_town.tooltip
	}
	
	# Option 3: Cancel
	option = {
		name = dynamic_missions.30.option_cancel
		remove_variable = active_dynamic_mission
		custom_tooltip = dynamic_missions.30.option_cancel.tooltip
	}
}

dynamic_missions.100 = {
	type = country_event
	title = dynamic_missions.100.title
	desc = dynamic_missions.100.desc
	fire_only_once = no
	hidden = yes  # This is a silent event for AI decision-making
	
	trigger = {
		is_ai = yes
		is_subject = yes
		overlord = {
			# has_active_situation = develop_city_situation
		}
	}
	
	# AI evaluates whether to use available actions
	immediate = {
		# Check if disloyal and can steal progress
		if = {
			limit = {
				subject_loyalty < 50
				NOT = { has_country_modifier = dm_steal_progress_cooldown }
				opinion = {
					target = overlord
					value < 0
				}
			}
			# 30% chance to steal if conditions are met
			random = {
				chance = 30
				# Execute the steal progress action
				# Note: This would normally trigger the generic action
				# For now, we'll just set a flag for future implementation
			}
		}
		
		# Check if loyal and can contribute
		if = {
			limit = {
				subject_loyalty >= 50
				NOT = { has_country_modifier = dm_contribute_development_cooldown }
				opinion = {
					target = overlord
					value >= 50
				}
			}
			# 20% chance to contribute if conditions are met
			random = {
				chance = 20
				# Execute the contribute action
				# Note: This would normally trigger the generic action
			}
		}

		name = "OK"

# Dynamic Missions: Select Province to Claim
dynamic_missions.41 = {
	type = country_event
	title = dynamic_missions.41.title
	desc = dynamic_missions.41.desc
	fire_only_once = no
	
	illustration_tags = {
		10 = map
		10 = military
	}
	
	immediate = {
		# Build list of eligible neighboring provinces
		# Note: every_neighboring_location is commented out, using simplified version
		every_owned_location = {
			limit = {
				NOT = { is_in_list = eligible_provinces }
			}
			add_to_list = eligible_provinces
		}
	}
	
	# Dynamic options for each eligible province
	# Note: In actual implementation, would need multiple option blocks
	# For now, showing the pattern for the first province
	option = {
		name = dynamic_missions.41.option_province
		trigger = {
			any_in_list = {
				list = eligible_provinces
				count >= 1
			}
		}
		
		# Get first eligible province
		random_in_list = {
			list = eligible_provinces
			limit = { always = yes }
			save_scope_as = selected_province
			owner = { save_scope_as = target_owner }
		}
		
		# Save target information to country variables
		scope:selected_province = {
			save_scope_as = target_province
			set_variable = { name = claimed_by_player value = yes }
			owner = { save_scope_as = target_owner }
		}
		set_variable = { name = claim_target_province value = scope:target_province }
		set_variable = { name = claim_target_country value = scope:target_owner }
		
		# Join the global situation as participant
		situation:claim_province_situation = {
			add_participant = root
		}
		# Initialize mission variables
		set_variable = { name = dm_months_elapsed value = 0 }
		
		custom_tooltip = dynamic_missions.41.option_province.tooltip
	}
	
	# Option: Cancel
	option = {
		name = dynamic_missions.41.option_cancel
		remove_variable = active_dynamic_mission
		custom_tooltip = dynamic_missions.41.option_cancel.tooltip
	}
}
