# Dynamic Missions: Player Actions for City Development

# Action 1: Convene Builders
# Reduces global construction speed but massively boosts target city construction
dm_convene_builders = {
	type = situation
	
	ai_tick = daily
	ai_tick_frequency = 40
	automation_tick = daily
	automation_tick_frequency = 7
	
	# Only visible when the develop_city situation is active
	potential = {
		scope:actor ?= {
			NOT = { has_country_modifier = dm_convene_builders_cooldown }
		}
	}
	
	select_trigger = {
		looking_for_a = situation
		target_flag = recipient
		name = "choose_situation"
		column = {data = name}
		visible = {
			situation:develop_city_situation = this
			situation_is_active = yes
		}
	}
	
	# Effects when action is taken
	effect = {
		scope:actor ?= {
			# Apply global construction speed penalty
			add_country_modifier = {
				modifier = dm_global_construction_penalty
				years = 1
			}
			
			# Apply cooldown
			add_country_modifier = {
				modifier = dm_convene_builders_cooldown
				years = 1
			}
		}
		
		# Apply local construction speed bonus to target city
		var:target_location = {
			add_location_modifier = {
				modifier = dm_target_construction_bonus
				years = 1
			}
		}
	}
	
	# AI will not use this action (player-only feature)
	ai_will_do = {
		add = 0
	}
}

# Action 3: Demand Contributions
# Reduce control in other cities to boost target city control cap
dm_demand_contributions = {
	type = situation
	
	ai_tick = daily
	ai_tick_frequency = 40
	automation_tick = daily
	automation_tick_frequency = 7
	
	# Only visible when the develop_city situation is active
	potential = {
		scope:actor ?= {
			NOT = { has_country_modifier = dm_demand_contributions_cooldown }
		}
	}
	
	select_trigger = {
		looking_for_a = situation
		target_flag = recipient
		name = "choose_situation"
		column = {data = name}
		visible = {
			situation:develop_city_situation = this
			situation_is_active = yes
		}
	}
	
	# Effects when action is taken
	effect = {
		scope:actor ?= {
			# Reduce control in all other cities
			every_owned_location = {
				limit = {
					OR = {
						location_rank = town
						location_rank = city
					}
					NOT = { this = root.var:target_location }
				}
				
				# Apply control penalty
				add_location_modifier = {
					modifier = dm_control_penalty
					years = 1
				}
			}
			
			# Apply cooldown
			add_country_modifier = {
				modifier = dm_demand_contributions_cooldown
				years = 1
			}
		}
		
		# Boost max control cap in target city
		var:target_location = {
			add_location_modifier = {
				modifier = dm_control_cap_bonus
				years = 1
			}
		}
	}
	
	# AI will not use this action (player-only feature)
	ai_will_do = {
		add = 0
	}
}

# Subject Action 1: Steal Progress (for disloyal subjects)
# Reduces target city prosperity, increases subject's capital prosperity
dm_steal_progress = {
	type = situation
	
	ai_tick = daily
	ai_tick_frequency = 40
	automation_tick = daily
	automation_tick_frequency = 7
	
	# Only visible to subjects when the develop_city situation is active
	potential = {
		scope:actor ?= {
			is_subject = yes
			NOT = { has_country_modifier = dm_steal_progress_cooldown }
			is_disloyal = yes
		}
	}
	
	select_trigger = {
		looking_for_a = situation
		target_flag = recipient
		name = "choose_situation"
		column = {data = name}
		visible = {
			situation:develop_city_situation = this
			situation_is_active = yes
			owner = {
				any_subject = {
					this = scope:actor
				}
			}
		}
	}
	
	# Effects when action is taken
	effect = {
		# Reduce target city prosperity
		var:target_location = {
			add_location_modifier = {
				modifier = dm_prosperity_drain
				years = 1
			}
		}
		
		scope:actor ?= {
			# Increase subject's capital prosperity
			capital = {
				add_location_modifier = {
					modifier = dm_prosperity_gain
					years = 1
				}
			}
			
			# Apply cooldown
			add_country_modifier = {
				modifier = dm_steal_progress_cooldown
				years = 1
			}
		}
		
		# Notify overlord
		owner = {
			trigger_event = { id = dynamic_missions.10 }
		}
	}
	
	# AI decision-making: Use if disloyal
	ai_will_do = {
		add = 0
		
		add = {
			desc = "disloyal subject"
			value = 100
			scope:actor ?= {
				is_disloyal = yes
				opinion = {
					target = owner
					value < 0
				}
			}
		}
	}
}

# Subject Action 2: Contribute to Development (for loyal subjects)
# Increases target city prosperity, decreases subject's capital prosperity
dm_contribute_development = {
	type = situation
	
	ai_tick = daily
	ai_tick_frequency = 40
	automation_tick = daily
	automation_tick_frequency = 7
	
	# Only visible to subjects when the develop_city situation is active
	potential = {
		scope:actor ?= {
			is_subject = yes
			NOT = { has_country_modifier = dm_contribute_development_cooldown }
			is_disloyal = no
		}
	}
	
	select_trigger = {
		looking_for_a = situation
		target_flag = recipient
		name = "choose_situation"
		column = {data = name}
		visible = {
			situation:develop_city_situation = this
			situation_is_active = yes
			owner = {
				any_subject = {
					this = scope:actor
				}
			}
		}
	}
	
	# Effects when action is taken
	effect = {
		# Increase target city prosperity
		var:target_location = {
			add_location_modifier = {
				modifier = dm_prosperity_contribution
				years = 1
			}
		}
		
		scope:actor ?= {
			# Decrease subject's capital prosperity
			capital = {
				add_location_modifier = {
					modifier = dm_prosperity_sacrifice
					years = 1
				}
			}
			
			# Apply cooldown
			add_country_modifier = {
				modifier = dm_contribute_development_cooldown
				years = 1
			}
		}
		
		# Notify overlord
		owner = {
			trigger_event = { id = dynamic_missions.11 }
		}
	}
	
	# AI decision-making: Use if loyal and has good opinion
	ai_will_do = {
		add = 0
		
		add = {
			desc = "loyal subject"
			value = 100
			scope:actor ?= {
				is_disloyal = no
				opinion = {
					target = owner
					value >= 50
				}
			}
		}
	}
}
