# Dynamic Missions: Player Actions for City Development

# Action 1: Convene Builders
# Reduces global construction speed but massively boosts target city construction
dm_convene_builders = {
	# Only visible when the develop_city situation is active
	potential = {
		has_variable = active_dynamic_mission
		has_active_situation = develop_city_situation
	}
	
	# Requirements to use this action
	allow = {
		# Must not be on cooldown
		NOT = { has_country_modifier = dm_convene_builders_cooldown }
		
		# Must have the situation active
		has_active_situation = develop_city_situation
		
		custom_tooltip = {
			text = dm_convene_builders.allow.tooltip
			always = yes
		}
	}
	
	# Effects when action is taken
	effect = {
		# Apply global construction speed penalty
		add_country_modifier = {
			name = dm_global_construction_penalty
			duration = 365  # 1 year
		}
		
		# Apply local construction speed bonus to target city
		situation:develop_city_situation = {
			var:target_location = {
				add_location_modifier = {
					name = dm_target_construction_bonus
					duration = 365  # 1 year
				}
			}
		}
		
		# Apply cooldown
		add_country_modifier = {
			name = dm_convene_builders_cooldown
			duration = 365  # 1 year cooldown
			hidden = yes
		}
		
		custom_tooltip = dm_convene_builders.effect.tooltip
	}
	
	# AI will not use this action (player-only feature)
	ai_will_do = {
		base = 0
	}
}

			text = dm_demand_contributions.allow.tooltip
			always = yes
		}

	# Effects when action is taken
	effect = {
		# Reduce control in all other cities
		every_owned_location = {
			limit = {
				OR = {
					location_rank = town
					location_rank = city
				}
				NOT = { this = situation:develop_city_situation.var:target_location }
			}
			
			# Apply control penalty
			add_location_modifier = {
				name = dm_control_penalty
				duration = 365  # 1 year
			}
		}
		
		# Boost max control cap in target city

			var:target_location = {
				add_location_modifier = {
					name = dm_control_cap_bonus

	}

# Subject Action 1: Steal Progress (for disloyal subjects)
# Reduces target city prosperity, increases subject's capital prosperity
dm_steal_progress = {
	# Only visible to subjects when the develop_city situation is active
	potential = {
		is_subject = yes
		overlord = {
			has_variable = active_dynamic_mission
			has_active_situation = develop_city_situation
		}
	}
	
	# Requirements to use this action
	allow = {
		# Must not be on cooldown
		NOT = { has_country_modifier = dm_steal_progress_cooldown }
		
		# Must be disloyal
		is_disloyal = yes
		
		custom_tooltip = {
			text = dm_steal_progress.allow.tooltip
			always = yes
		}
	}
	
	# Effects when action is taken
	effect = {
		# Reduce target city prosperity
		overlord = {
			situation:develop_city_situation = {
				var:target_location = {
					add_location_modifier = {
						name = dm_prosperity_drain
						duration = 365  # 1 year
					}
				}
			}
		}
		
		# Increase subject's capital prosperity
		capital = {
			add_location_modifier = {
				name = dm_prosperity_gain
				duration = 365  # 1 year
			}
		}
		
		# Apply cooldown
		add_country_modifier = {
			name = dm_steal_progress_cooldown
			duration = 365  # 1 year cooldown
			hidden = yes
		}
		
		# Notify overlord
		overlord = {
			trigger_event = { id = dynamic_missions.10 }  # Notification event
		}
		
		custom_tooltip = dm_steal_progress.effect.tooltip
	}
	
	# AI decision-making: Use if disloyal
	ai_will_do = {
		base = 0
		
		modifier = {
			add = 100
			is_disloyal = yes
			opinion = {
				who = overlord
				value < 0
			}
		}
	}
}

# Dynamic Missions: Player Actions for Large Research Project

# Action: Conscript University
dm_conscript_university = {
    potential = { has_active_situation = large_research_project_situation }
    allow = {
        NOT = { has_country_modifier = dm_conscript_university_cooldown }
        any_owned_location = { 
            has_building = university 
            # Optional: Add a check to see if the location is already conscripted
        }
    }
    effect = {
        # Trigger an event to let the player choose which university to conscript
        trigger_event = { id = dynamic_missions.21 }
        # Apply a 1-year cooldown
        add_country_modifier = { name = dm_conscript_university_cooldown duration = 365 hidden = yes }
    }
    ai_will_do = { base = 0 }
}

# Action: Appoint Lead Scientist
dm_appoint_lead_scientist = {
    potential = { has_active_situation = large_research_project_situation }

    }
    ai_will_do = { base = 0 }
}

# Appoint Regional Governor (One-time action)
dm_appoint_regional_governor = {
	# Only visible during the Establish New City mission
	potential = {
		has_variable = active_dynamic_mission
		has_active_situation = establish_new_city_situation
	}
	
	# Can only be used once per mission
	allow = {
		NOT = { has_country_modifier = dm_appoint_governor_used }
		any_character = {
			is_in_court = yes
		}
		
		custom_tooltip = {
			text = dm_appoint_governor.allow.tooltip
			always = yes
		}
	}
	
	# Trigger event to select character
	effect = {
		trigger_event = { id = dynamic_missions.31 }
		add_country_modifier = {
			name = dm_appoint_governor_used
			duration = -1
			hidden = yes
		}
		
		custom_tooltip = dm_appoint_governor.effect.tooltip
	}
	
	# AI will not use this
	ai_will_do = {
		base = 0
	}
}
