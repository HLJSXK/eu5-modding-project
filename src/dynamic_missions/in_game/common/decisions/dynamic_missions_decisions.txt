# Dynamic Missions: Decisions
# These decisions provide toggleable actions for various dynamic missions

# ESTABLISH NEW CITY MISSION DECISIONS
# =============================================================================

# Encourage Migration (Toggleable)
dm_encourage_migration_decision = {
	# Only visible during the Establish New City mission
	potential = {
		has_active_situation = establish_new_city_situation
	}
	
	# Can always be toggled
	allow = {
		always = yes
	}
	
	# Toggle the modifier on/off
	effect = {
		if = {
			limit = { has_country_modifier = dm_encourage_migration_modifier }
			# Turn off
			remove_country_modifier = dm_encourage_migration_modifier
			situation:establish_new_city_situation = {
				var:target_location = {
					remove_location_modifier = dm_migration_boost
				}
			}
		}
		else = {
			# Turn on
			add_country_modifier = {
				name = dm_encourage_migration_modifier
				duration = -1  # Permanent until removed
			}
			situation:establish_new_city_situation = {
				var:target_location = {
					add_location_modifier = {
						name = dm_migration_boost
						duration = -1
					}
				}
			}
		}
	}
	
	# AI will not use this
	ai_will_do = {
		base = 0
	}
}

# Seek Rival Support (One-Time Action)
dm_claim_seek_rival_support_decision = {
	# Only visible during the Claim Province mission
	potential = {
		has_active_situation = claim_province_situation
		NOT = { has_variable = used_seek_rival_support }
	}
	
	# Requires prestige
	allow = {
		prestige >= 10
	}
	
	# Execute the action
	effect = {
		add_prestige = -10
		set_variable = { name = used_seek_rival_support value = yes }
		
		# Trigger events for each rival of the target country
		situation:claim_province_situation = {
			var:target_country = {
				every_rival = {
					limit = { exists = yes }
					trigger_event = { id = dynamic_missions.42 }
				}
			}
		}
		
		custom_tooltip = dm_claim_seek_rival_support_tooltip
	}
	
	# AI will not use this
	ai_will_do = {
		base = 0
	}
}

# Fabricate Legitimacy (One-Time Action)
dm_claim_fabricate_legitimacy_decision = {
	# Only visible during the Claim Province mission
	potential = {
		has_active_situation = claim_province_situation
		NOT = { has_variable = used_fabricate_legitimacy }
	}
	
	# Requires 20 spy network in target country
	allow = {
		custom_tooltip = {
			text = dm_claim_requires_spy_network_20
			situation:claim_province_situation = {
				var:target_country = {
					root = { 
						has_spy_network_in = { 
							target = prev 
							value >= 20 
						} 
					}
				}
			}
		}
	}
	
	# Execute the action
	effect = {
		# Consume spy network
		situation:claim_province_situation = {
			var:target_country = {
				root = { 
					reduce_spy_network = { 
						target = prev 
						value = 20 
					} 
				}
			}
		}
		
		# Add diplomatic reputation modifier
		add_country_modifier = {
			name = dm_fabricated_legitimacy
			duration = -1  # Until situation ends
		}
		
		set_variable = { name = used_fabricate_legitimacy value = yes }
		custom_tooltip = dm_claim_fabricate_legitimacy_tooltip
	}
	
	# AI will not use this
	ai_will_do = {
		base = 0
	}
}

# Conduct Border Reconnaissance (Annual Action)
dm_claim_border_reconnaissance_decision = {
	# Only visible during the Claim Province mission
	potential = {
		has_active_situation = claim_province_situation
	}
	
	# Requires manpower and no cooldown
	allow = {
		NOT = { has_country_modifier = dm_border_reconnaissance_cooldown }
		manpower >= 200
	}
	
	# Execute the action
	effect = {
		add_manpower = -200
		add_army_tradition = 5
		
		# Add 1-year cooldown
		add_country_modifier = {
			name = dm_border_reconnaissance_cooldown
			duration = 365
			hidden = yes
		}
		
		custom_tooltip = dm_claim_border_reconnaissance_tooltip
	}
	
	# AI will not use this
	ai_will_do = {
		base = 0
	}
}
