# Dynamic Missions: Develop City Situation
# This situation tracks the player's progress in developing a chosen city
develop_city_situation = {
	# This situation is manually activated, not by monthly chance
	monthly_spawn_chance = 0
	
	# Visibility: Only the player and their subjects can see this situation
	visible = {
		OR = {
			this = root.owner  # The player
			is_subject_of = root.owner  # Subjects of the player
		}
	}
	
	# Start conditions: Manually activated via event
	can_start = {
		always = no  # Never starts automatically
	}
	
	# End conditions: Either goal is met or time limit reached
	can_end = {
		OR = {
			# Success condition: Target city has 100% prosperity and 10+ new buildings
			AND = {
				var:target_location = {
					prosperity >= 100
				}
				# Check if building count increased by at least 10
				trigger_if = {
					limit = { has_variable = initial_building_count }
					var:target_location = {
						num_buildings >= {
							add = var:initial_building_count
							add = 10
						}
					}
				}
				# Set success flag
				set_variable = {
					name = mission_success
					value = yes
				}
			}
			# Failure condition: 5 years (60 months) have passed
			AND = {
				has_variable = situation_start_date
				months_since_variable = {
					variable = situation_start_date
					months >= 60
				}
				# Set failure flag
				set_variable = {
					name = mission_failed
					value = yes
				}
			}
		}
	}
	
	# On start: Initialize variables and add participants
	on_start = {
		# Record the start date
		set_variable = {
			name = situation_start_date
			value = current_date
		}
		
		# Initialize progress tracking
		set_variable = {
			name = completion_progress
			value = 0
		}
		
		# Add all subjects as participants
		owner = {
			every_subject = {
				add_to_list = situation_participants
			}
		}
	}
	
	# On monthly: Update progress and check conditions
	on_monthly = {
		# Update completion progress based on prosperity and buildings
		if = {
			limit = { has_variable = target_location }
			
			# Calculate progress: (prosperity + building_progress) / 2
			var:target_location = {
				set_variable = {
					name = prosperity_progress
					value = prosperity
				}
				
				# Calculate building progress
				set_variable = {
					name = building_progress
					value = {
						value = num_buildings
						subtract = var:initial_building_count
						multiply = 10  # Each building is 10% progress
						max = 100
					}
				}
			}
			
			# Average the two progress values
			set_variable = {
				name = completion_progress
				value = {
					value = var:target_location.prosperity_progress
					add = var:target_location.building_progress
					divide = 2
				}
			}
		}
		
		# Notify participants monthly (for AI decision-making)
		every_in_list = {
			list = situation_participants
			trigger_event_silently = dynamic_missions.100  # AI evaluation event
		}
	}
	
	# On ending: Trigger appropriate completion event
	on_ending = {
		owner = {
			if = {
				limit = {
					situation:develop_city_situation = {
						has_variable = mission_success
					}
				}
				# Success event
				trigger_event = { id = dynamic_missions.3 }
			}
			else_if = {
				limit = {
					situation:develop_city_situation = {
						has_variable = mission_failed
					}
				}
				# Failure event
				trigger_event = { id = dynamic_missions.4 }
			}
		}
	}
	
	# On ended: Clean up variables
	on_ended = {
		# Remove all situation variables to reduce save file size
		remove_variable = target_location
		remove_variable = situation_start_date
		remove_variable = completion_progress
		remove_variable = mission_success
		remove_variable = mission_failed
		remove_variable = initial_building_count
		
		# Clear participant list
		clear_list = situation_participants
	}
	
	# Map colors for the situation mapmode
	map_color = {
		if = {
			limit = {
				# Highlight the target city
				this = situation:develop_city_situation.var:target_location
			}
			value = rgb { 50 200 50 }  # Green for target
		}
		else_if = {
			limit = {
				# Show player's other cities in yellow
				owner = situation:develop_city_situation.owner
				OR = {
					location_rank = town
					location_rank = city
				}
			}
			value = rgb { 200 200 50 }  # Yellow for other cities
		}
	}
	
	# Tooltip for map
	tooltip = {
		if = {
			limit = {
				this = situation:develop_city_situation.var:target_location
			}
			custom_tooltip = dynamic_missions.situation.target_city_tooltip
		}
	}
	
	# Legend entries for the mapmode
	legend_key = {
		color = rgb { 50 200 50 }
		desc = dynamic_missions.situation.legend_target
	}
	
	legend_key = {
		color = rgb { 200 200 50 }
		desc = dynamic_missions.situation.legend_other_cities
	}
}


# Dynamic Missions: Large Research Project Situation
large_research_project_situation = {
    monthly_spawn_chance = 0

    visible = {
        OR = {
            this = root.owner
            is_subject_of = root.owner
            is_neighbor_of = root.owner
        }
    }

    can_start = { always = no } # Manually triggered by event

    can_end = {
        OR = {
            # Success Condition
            AND = {
                has_variable = research_progress
                var:research_progress >= 100
                set_variable = { name = mission_success value = yes }
            },
            # Failure Condition
            AND = {
                has_variable = situation_start_date
                months_since_variable = { variable = situation_start_date months >= 60 }
                set_variable = { name = mission_failed value = yes }
            }
        }
    }

    on_start = {
        set_variable = { name = situation_start_date value = current_date }
        set_variable = { name = research_progress value = 0 }
        owner = {
            every_subject = { add_to_list = situation_participants }
            every_neighbor_country = { add_to_list = situation_participants }
        }
    }

    on_monthly = {
        # Base progress (can be 0)
        # Progress from 'Increase Research Funding'
        if = {
            limit = { has_country_modifier = dm_increased_funding_modifier }
            change_variable = { name = research_progress value = 1 }
        }
        # Progress from 'Conscript University'
        if = {
            limit = { has_variable = conscripted_universities_count }
            change_variable = { name = research_progress value = var:conscripted_universities_count }
        }
    }

    on_ending = {
        owner = {
            if = {
                limit = { situation:large_research_project_situation = { has_variable = mission_success } }
                trigger_event = { id = dynamic_missions.23 } # Success Event
            }
            else_if = {
                limit = { situation:large_research_project_situation = { has_variable = mission_failed } }
                trigger_event = { id = dynamic_missions.24 } # Failure Event
            }
        }
    }

    on_ended = {
        # Remove character modifier from appointed scientist if exists
        if = {
            limit = { has_variable = lead_scientist_char }
            var:lead_scientist_char = {
                remove_character_modifier = dm_lead_scientist_debuff
            }
            remove_variable = lead_scientist_char
        }
        
        # Remove location modifiers from conscripted universities
        owner = {
            every_owned_location = {
                limit = { has_location_modifier = dm_conscripted_university_debuff }
                remove_location_modifier = dm_conscripted_university_debuff
            }
        }
        
        # Clean up variables
        remove_variable = situation_start_date
        remove_variable = research_progress
        remove_variable = mission_success
        remove_variable = mission_failed
        remove_variable = conscripted_universities_count
        remove_variable = lead_scientist_appointed
        clear_list = situation_participants
    }
}



# Dynamic Missions: Establish New City Situation
# This situation tracks the player's progress in transforming a town into a city
establish_new_city_situation = {
	# This situation is manually activated, not by monthly chance
	monthly_spawn_chance = 0
	
	# Visibility: Only the player can see this situation
	visible = {
		this = root.owner
	}
	
	# Start conditions: Manually activated via event
	can_start = {
		always = no  # Never starts automatically
	}
	
	# End conditions: Either goal is met or time limit reached
	can_end = {
		OR = {
			# Success condition: Target location becomes a city
			AND = {
				has_variable = target_location
				var:target_location = {
					location_rank = city
				}
				# Set success flag
				set_variable = {
					name = mission_success
					value = yes
				}
			}
			# Failure condition: 10 years (120 months) have passed
			AND = {
				has_variable = situation_start_date
				months_since_variable = {
					variable = situation_start_date
					months >= 120
				}
				# Set failure flag
				set_variable = {
					name = mission_failed
					value = yes
				}
			}
		}
	}
	
	# On start: Initialize variables
	on_start = {
		# Record the start date
		set_variable = {
			name = situation_start_date
			value = current_date
		}
		
		# Target location is set by the triggering event
	}
	
	# On monthly: Update progress and handle value shifts
	on_monthly = {
		owner = {
			# Handle value shifts from toggleable actions
			if = {
				limit = { has_country_modifier = dm_encourage_migration_modifier }
				change_value = {
					value = freemen
					amount = 0.10
				}
			}
			
			if = {
				limit = { has_country_modifier = dm_encourage_proselytization_modifier }
				change_value = {
					value = spiritual
					amount = 0.10
				}
			}
			
			if = {
				limit = { has_country_modifier = dm_encourage_folk_culture_modifier }
				change_value = {
					value = humanist
					amount = 0.10
				}
			}
			
			# Handle estate satisfaction penalty from requisition
			if = {
				limit = { has_country_modifier = dm_requisition_materials_modifier }
				every_estate = {
					change_satisfaction = -0.05
				}
			}
		}
		
		# Apply governor's prosperity bonus if appointed
		if = {
			limit = { 
				has_variable = target_location
				var:target_location = { has_variable = governor_prosperity_bonus }
			}
			var:target_location = {
				change_prosperity = var:governor_prosperity_bonus
			}
		}
	}
	
	# On ending: Trigger appropriate completion event
	on_ending = {
		owner = {
			if = {
				limit = {
					situation:establish_new_city_situation = {
						has_variable = mission_success
					}
				}
				# Success event
				trigger_event = { id = dynamic_missions.33 }
			}
			else_if = {
				limit = {
					situation:establish_new_city_situation = {
						has_variable = mission_failed
					}
				}
				# Failure event
				trigger_event = { id = dynamic_missions.34 }
			}
		}
	}
	
		# On ended: Clean up variables and modifiers
		on_ended = {
			owner = {
				# Remove all toggleable modifiers
				remove_country_modifier = dm_encourage_migration_modifier
				remove_country_modifier = dm_encourage_proselytization_modifier
				remove_country_modifier = dm_encourage_folk_culture_modifier
				remove_country_modifier = dm_requisition_materials_modifier
				remove_country_modifier = dm_appoint_governor_used
				
				# Remove location modifiers and variables from target
				if = {
					limit = { situation:establish_new_city_situation = { has_variable = target_location } }
					situation:establish_new_city_situation = {
						var:target_location = {
							remove_location_modifier = dm_migration_boost
							remove_location_modifier = dm_conversion_boost
							remove_location_modifier = dm_assimilation_boost
							remove_location_modifier = dm_governor_control_bonus
							# Remove governor prosperity bonus variable
							remove_variable = governor_prosperity_bonus
						}
					}
				}
				
				# Remove character modifier from governor if appointed
				if = {
					limit = { 
						situation:establish_new_city_situation = { 
							has_variable = governor_character 
						} 
					}
					situation:establish_new_city_situation = {
						var:governor_character = {
							remove_character_modifier = dm_regional_governor_debuff
						}
					}
				}
			}
			
			# Remove all situation variables
			remove_variable = target_location
			remove_variable = situation_start_date
			remove_variable = mission_success
			remove_variable = mission_failed
			remove_variable = governor_character
		}
	
	# Map colors for the situation mapmode
	map_color = {
		if = {
			limit = {
				# Highlight the target town
				this = situation:establish_new_city_situation.var:target_location
			}
			value = rgb { 100 150 255 }  # Blue for target town
		}
		else_if = {
			limit = {
				# Show player's other towns in light blue
				owner = situation:establish_new_city_situation.owner
				location_rank = town
			}
			value = rgb { 150 200 255 }  # Light blue for other towns
		}
	}
	
	# Tooltip for map
	tooltip = {
		if = {
			limit = {
				this = situation:establish_new_city_situation.var:target_location
			}
			custom_tooltip = dynamic_missions.establish_city.target_town_tooltip
		}
	}
	
	# Legend entries for the mapmode
	legend_key = {
		color = rgb { 100 150 255 }
		desc = dynamic_missions.establish_city.legend_target
	}
	
	legend_key = {
		color = rgb { 150 200 255 }
		desc = dynamic_missions.establish_city.legend_other_towns
	}
}
