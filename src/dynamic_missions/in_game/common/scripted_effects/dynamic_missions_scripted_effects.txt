# Calculate Player Coalition Strength
# Sums up military strength of player and all supporters
dm_calculate_player_coalition_strength = {
	# Start with player's own military strength
	set_variable = { 
		name = player_coalition_strength 
		value = root.military_strength 
	}
	
	# Add strength from each supporter
	if = {
		limit = { has_variable = player_supporters_list }
		every_in_list = {
			list = player_supporters_list
			limit = { exists = yes }
			root = {
				change_variable = {
					name = player_coalition_strength
					add = prev.military_strength
				}
			}
		}
	}
}

# Calculate Target Coalition Strength
# Sums up military strength of target, their supporters, allies, and guarantors
dm_calculate_target_coalition_strength = {
	situation:claim_province_situation = {
		var:target_country = {
			# Start with target's own military strength
			set_variable = { 
				name = target_coalition_strength 
				value = military_strength 
			}
			
			# Add strength from rivals who accepted support
			if = {
				limit = { 
					situation:claim_province_situation = { 
						has_variable = target_supporters_list 
					}
				}
				situation:claim_province_situation = {
					every_in_list = {
						list = target_supporters_list
						limit = { exists = yes }
						situation:claim_province_situation.var:target_country = {
							change_variable = {
								name = target_coalition_strength
								add = prev.military_strength
							}
						}
					}
				}
			}
			
			# Add strength from all allies
			every_ally = {
				limit = { exists = yes }
				prev = {
					change_variable = {
						name = target_coalition_strength
						add = prev.military_strength
					}
				}
			}
			
			# Add strength from defensive pact members
			every_country_in_defensive_pact = {
				limit = { exists = yes }
				prev = {
					change_variable = {
						name = target_coalition_strength
						add = prev.military_strength
					}
				}
			}
			
			# Add strength from countries guaranteeing independence
			every_country_guaranteeing_independence = {
				limit = { exists = yes }
				prev = {
					change_variable = {
						name = target_coalition_strength
						add = prev.military_strength
					}
				}
			}
		}
	}
}

# Check if AI Should Surrender
# Determines if target should surrender based on 5x strength ratio
dm_check_ai_surrender_condition = {
	# Calculate both coalition strengths
	dm_calculate_player_coalition_strength = yes
	dm_calculate_target_coalition_strength = yes
	
	# Check if player coalition is 5x stronger
	situation:claim_province_situation = {
		var:target_country = {
			if = {
				limit = {
					root.var:player_coalition_strength >= {
						value = var:target_coalition_strength
						multiply = 5.0
					}
				}
				# AI will surrender
				situation:claim_province_situation = {
					set_variable = { name = ai_will_surrender value = yes }
				}
			}
			else = {
				# AI will fight
				situation:claim_province_situation = {
					set_variable = { name = ai_will_fight value = yes }
				}
			}
		}
	}
}

# Calculate AI Willingness to Support Player
# Used when player seeks rival support
# Scope: potential supporter country
# Formula: Base -100 + (Target Opinion/2) + (Player Opinion/2) + (DipRep*10) + (Allied?200:0)
dm_calculate_rival_support_willingness = {
	# Save the player and target for reference
	save_scope_as = potential_supporter
	situation:claim_province_situation.owner = { save_scope_as = claim_player }
	situation:claim_province_situation.var:target_country = { save_scope_as = claim_target }
	
	# Start with very negative base
	set_variable = { name = support_willingness value = -100 }
	
	# Add half of opinion toward target (negative opinion helps)
	change_variable = {
		name = support_willingness
		add = {
			value = opinion
			target = scope:claim_target
			divide = 2
		}
	}
	
	# Add half of opinion toward player (positive opinion helps)
	change_variable = {
		name = support_willingness
		add = {
			value = opinion
			target = scope:claim_player
			divide = 2
		}
	}
	
	# Add diplomatic reputation bonus (10 per point)
	change_variable = {
		name = support_willingness
		add = {
			value = scope:claim_player.modifier:diplomatic_reputation
			multiply = 10
		}
	}
	
	# Massive bonus if already allied
	if = {
		limit = { is_ally_of = scope:claim_player }
		change_variable = {
			name = support_willingness
			add = 200
		}
	}
}

# Execute War Declaration with Supporters
# Declares war with claim province CB and adds all supporters
dm_execute_claim_war_declaration = {
	situation:claim_province_situation = {
		owner = {
			# Declare war with claim province CB
			declare_war_with_cb = {
				target = situation:claim_province_situation.var:target_country
				type = casus_belli:cb_conquer_province
				province = situation:claim_province_situation.var:target_province
			}
			
			# Add all player supporters to the war
			if = {
				limit = { 
					situation:claim_province_situation = { 
						has_variable = player_supporters_list 
					}
				}
				situation:claim_province_situation = {
					every_in_list = {
						list = player_supporters_list
						limit = { exists = yes }
						# Join war as co-belligerent
						join_war = {
							war = situation:claim_province_situation.owner.current_war
							side = attacker
						}
					}
				}
			}
		}
	}
}

# Transfer Province Ownership
# Transfers the claimed province to the player
dm_transfer_claimed_province = {
	situation:claim_province_situation = {
		var:target_province = {
			every_location_in_province = {
				set_owner = situation:claim_province_situation.owner
			}
		}
	}
}
