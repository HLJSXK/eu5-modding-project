# Dynamic Missions: Choose Your Goal
dynamic_missions.1 = {
	type = country_event
	title = dynamic_missions.1.title
	desc = dynamic_missions.1.desc
	fire_only_once = no
	
	illustration_tags = {
		10 = happy
		10 = interior
	}
	
	trigger = {
		is_ai = no
		has_game_started = yes
		NOT = { has_variable = active_dynamic_mission }
	}
	
	# Option 1: Focus on Developing a City
	option = {
		name = dynamic_missions.1.option_a
		trigger = { can_start_develop_city_mission = yes }
		set_variable = { name = active_dynamic_mission value = yes }
		trigger_event = { id = dynamic_missions.2 }
		custom_tooltip = dynamic_missions.1.option_a.tooltip
	}

	# Option 2: Large Research Project
	option = {
		name = "Start a Large Research Project"
		trigger = { can_start_large_research_project = yes }
		set_variable = { name = active_dynamic_mission value = yes }
		activate_situation = large_research_project_situation
		custom_tooltip = "Embark on a 5-year research project to gain a significant technology boost."
	}
	
	# Option 3: Establish a New City
	option = {
		name = dynamic_missions.1.option_d
		trigger = { can_start_establish_new_city_mission = yes }
		set_variable = { name = active_dynamic_mission value = yes }
		trigger_event = { id = dynamic_missions.30 }
		custom_tooltip = dynamic_missions.1.option_d.tooltip
	}
	
	# Option 4: Not now
	option = {
		name = dynamic_missions.1.option_c
		custom_tooltip = dynamic_missions.1.option_c.tooltip
	}
}

	type = country_event
	title = dynamic_missions.2.title
	desc = dynamic_missions.2.desc
	fire_only_once = no
	
	illustration_tags = {
		10 = exterior
		10 = city
	}
	
	immediate = {
		every_owned_location = {
			limit = {
				OR = {
					location_rank = town
					location_rank = city
				}
			}
			add_to_list = eligible_cities
		}
	}
	
	option = {
		name = dynamic_missions.2.option_a
		capital = {
			save_scope_as = target_city
		}
		activate_situation = develop_city_situation
		situation:develop_city_situation = {
			set_variable = { name = target_location value = scope:target_city }
			scope:target_city = {
				set_variable = { name = initial_building_count value = num_buildings }
			}
		}
		custom_tooltip = dynamic_missions.2.option_a.tooltip
	}
	
	option = {
		name = dynamic_missions.2.option_b
		remove_variable = active_dynamic_mission
		custom_tooltip = dynamic_missions.2.option_b.tooltip
	}
}

# --- Large Research Project Events ---

country_event = {
    id = dynamic_missions.21
    title = "dm_conscript_university.t"
    desc = "dm_conscript_university.d"
    picture = GFX_evt_library
    is_triggered_only = yes

    dynamic_option = {
        for_each = every_owned_location
        limit = {
            has_building = university
            NOT = { has_location_modifier = dm_conscripted_university_debuff }
        }
        name = "[location.GetName]"
        effect = {
            add_location_modifier = { name = dm_conscripted_university_debuff }
            situation:large_research_project_situation = {
                if = {
                    limit = { NOT = { has_variable = conscripted_universities_count } }
                    set_variable = { name = conscripted_universities_count value = 0 }
                }
                change_variable = { name = conscripted_universities_count value = 1 }
            }
        }
    }
}

country_event = {
    id = dynamic_missions.22
    title = "dm_appoint_lead_scientist.t"
    desc = "dm_appoint_lead_scientist.d"
    picture = GFX_evt_character
    is_triggered_only = yes

    dynamic_option = {
        for_each = every_character
        limit = { is_in_court = yes }
        name = "[character.GetName]"
        effect = {
            set_variable = { name = temp_scientist_char value = this }
            this = { add_character_modifier = { name = dm_lead_scientist_debuff } }
            situation:large_research_project_situation = {
                set_variable = { name = lead_scientist_appointed value = yes }
                set_variable = { name = lead_scientist_char value = root.var:temp_scientist_char }
                set_variable = {
                    name = temp_progress_bonus
                    value = {
                        value = root.var:temp_scientist_char.skill_adm
                        add = root.var:temp_scientist_char.skill_dip
                        add = root.var:temp_scientist_char.skill_mil
                        divide = 10
                    }
                }
                change_variable = { name = research_progress value = var:temp_progress_bonus }
                remove_variable = temp_progress_bonus
            }
            root = { remove_variable = temp_scientist_char }
        }
    }
}

# Dynamic Missions: Select Target Town for New City
dynamic_missions.30 = {
	type = country_event
	title = dynamic_missions.30.title
	desc = dynamic_missions.30.desc
	fire_only_once = no
	
	illustration_tags = {
		10 = exterior
		10 = town
	}
	
	immediate = {
		# Build list of eligible towns
		every_owned_location = {
			limit = {
				location_rank = town
				is_occupied = no
				is_under_siege = no
			}
			add_to_list = eligible_towns
		}
	}
	
	# Dynamic options for each eligible town
	# Note: In actual implementation, you may need to use a fixed number of options
	# or implement a selection mechanism via variables
	
	# Option 1: Select Capital (if it's a town)
	option = {
		name = dynamic_missions.30.option_capital
		trigger = {
			capital = {
				location_rank = town
			}
		}
		
		capital = {
			save_scope_as = target_town
		}
		
		activate_situation = establish_new_city_situation
		
		situation:establish_new_city_situation = {
			set_variable = { 
				name = target_location 
				value = scope:target_town 
			}
		}
		
		custom_tooltip = dynamic_missions.30.option_capital.tooltip
	}
	
	# Option 2: Select First Eligible Town
	option = {
		name = dynamic_missions.30.option_first_town
		trigger = {
			any_in_list = {
				list = eligible_towns
				count >= 1
			}
		}
		
		random_in_list = {
			list = eligible_towns
			save_scope_as = target_town
		}
		
		activate_situation = establish_new_city_situation
		
		situation:establish_new_city_situation = {
			set_variable = { 
				name = target_location 
				value = scope:target_town 
			}
		}
		
		custom_tooltip = dynamic_missions.30.option_first_town.tooltip
	}
	
	# Option 3: Cancel
	option = {
		name = dynamic_missions.30.option_cancel
		remove_variable = active_dynamic_mission
		custom_tooltip = dynamic_missions.30.option_cancel.tooltip
	}
}

# Dynamic Missions: Select Character for Regional Governor
dynamic_missions.31 = {
	type = country_event
	title = dynamic_missions.31.title
	desc = dynamic_missions.31.desc
	fire_only_once = no
	
	illustration_tags = {
		10 = character
		10 = interior
	}
	
	# Dynamic option: Select a character from court
	dynamic_option = {
		for_each = every_character
		limit = {
			is_in_court = yes
		}
		name = "[character.GetName]"
		
		effect = {
			# Save character reference
			set_variable = { 
				name = temp_governor_char 
				value = this 
			}
			
			# Calculate control bonus based on character skills
			set_variable = {
				name = temp_control_bonus
				value = {
					value = this.skill_adm
					add = this.skill_dip
					add = this.skill_mil
					divide = 10
				}
			}
			
			# Apply character debuff
			this = {
				add_character_modifier = {
					name = dm_regional_governor_debuff
					duration = -1  # Until mission ends
				}
			}
			
			# Apply control bonus to target location
			situation:establish_new_city_situation = {
				set_variable = {
					name = governor_character
					value = root.var:temp_governor_char
				}
				
				var:target_location = {
					# Apply dynamic control bonus
					add_location_modifier = {
						name = dm_governor_control_bonus
						duration = -1  # Until mission ends
						# Note: The actual bonus value needs to be set here
						# This is a simplified version
					}
					change_control = root.var:temp_control_bonus
					change_max_control = root.var:temp_control_bonus
				}
			}
			
			# Lose legitimacy
			add_legitimacy = -5
			
			# Clean up temporary variables
			remove_variable = temp_governor_char
			remove_variable = temp_control_bonus
			
			custom_tooltip = dynamic_missions.31.effect.tooltip
		}
	}
	
	# Option: Cancel
	option = {
		name = dynamic_missions.31.option_cancel
		# Remove the "used" flag so player can try again
		remove_country_modifier = dm_appoint_governor_used
		custom_tooltip = dynamic_missions.31.option_cancel.tooltip
	}
}

	fire_only_once = no

		10 = happy
		10 = city
	}
	
	option = {
		name = dynamic_missions.33.option_a

		situation:establish_new_city_situation = {
			var:target_location = {
				add_location_modifier = {
					name = dm_country_pearl
					duration = -1  # Permanent
				}

		set_country_flag = has_completed_establish_city_mission
		
		# Remove active mission variable
		remove_variable = active_dynamic_mission
		
		# Trigger next mission selection after 30 days
		trigger_event = { id = dynamic_missions.1 days = 30 }
		
		custom_tooltip = dynamic_missions.33.option_a.tooltip
	}
}
