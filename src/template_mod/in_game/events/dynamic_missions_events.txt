namespace = dynamic_missions

# Dynamic Missions: Choose Your Goal
dynamic_missions.1 = {
	type = country_event
	title = dynamic_missions.1.title
	desc = dynamic_missions.1.desc
	fire_only_once = no
	
	illustration_tags = {
		10 = happy
		10 = interior
	}
	
	trigger = {
		is_ai = no
		has_game_started = yes
		NOT = { has_variable = active_dynamic_mission }
	}
	
	# Option 1: Focus on Developing a City
	option = {
		name = dynamic_missions.1.option_a
		trigger = { can_start_develop_city_mission = yes }
		set_variable = { name = active_dynamic_mission value = yes }
		trigger_event = { id = dynamic_missions.2 }
		custom_tooltip = dynamic_missions.1.option_a.tooltip
	}

	# Option 2: Large Research Project
	option = {
		name = "Start a Large Research Project"
		trigger = { can_start_large_research_project = yes }
		set_variable = { name = active_dynamic_mission value = yes }
		activate_situation = large_research_project_situation
		custom_tooltip = "Embark on a 5-year research project to gain a significant technology boost."
	}
	
	# Option 3: Not now
	option = {
		name = dynamic_missions.1.option_c
		custom_tooltip = dynamic_missions.1.option_c.tooltip
	}
}

# Dynamic Missions: Select Target City
dynamic_missions.2 = {
	type = country_event
	title = dynamic_missions.2.title
	desc = dynamic_missions.2.desc
	fire_only_once = no
	
	illustration_tags = {
		10 = exterior
		10 = city
	}
	
	immediate = {
		every_owned_location = {
			limit = {
				OR = {
					location_rank = town
					location_rank = city
				}
			}
			add_to_list = eligible_cities
		}
	}
	
	option = {
		name = dynamic_missions.2.option_a
		capital = {
			save_scope_as = target_city
		}
		activate_situation = develop_city_situation
		situation:develop_city_situation = {
			set_variable = { name = target_location value = scope:target_city }
			scope:target_city = {
				set_variable = { name = initial_building_count value = num_buildings }
			}
		}
		custom_tooltip = dynamic_missions.2.option_a.tooltip
	}
	
	option = {
		name = dynamic_missions.2.option_b
		remove_variable = active_dynamic_mission
		custom_tooltip = dynamic_missions.2.option_b.tooltip
	}
}

# --- Large Research Project Events ---

country_event = {
    id = dynamic_missions.21
    title = "dm_conscript_university.t"
    desc = "dm_conscript_university.d"
    picture = GFX_evt_library
    is_triggered_only = yes

    dynamic_option = {
        for_each = every_owned_location
        limit = {
            has_building = university
            NOT = { has_location_modifier = dm_conscripted_university_debuff }
        }
        name = "[location.GetName]"
        effect = {
            add_location_modifier = { name = dm_conscripted_university_debuff }
            situation:large_research_project_situation = {
                if = {
                    limit = { NOT = { has_variable = conscripted_universities_count } }
                    set_variable = { name = conscripted_universities_count value = 0 }
                }
                change_variable = { name = conscripted_universities_count value = 1 }
            }
        }
    }
}

country_event = {
    id = dynamic_missions.22
    title = "dm_appoint_lead_scientist.t"
    desc = "dm_appoint_lead_scientist.d"
    picture = GFX_evt_character
    is_triggered_only = yes

    dynamic_option = {
        for_each = every_character
        limit = { is_in_court = yes }
        name = "[character.GetName]"
        effect = {
            set_variable = { name = temp_scientist_char value = this }
            this = { add_character_modifier = { name = dm_lead_scientist_debuff } }
            situation:large_research_project_situation = {
                set_variable = { name = lead_scientist_appointed value = yes }
                set_variable = { name = lead_scientist_char value = root.var:temp_scientist_char }
                set_variable = {
                    name = temp_progress_bonus
                    value = {
                        value = root.var:temp_scientist_char.skill_adm
                        add = root.var:temp_scientist_char.skill_dip
                        add = root.var:temp_scientist_char.skill_mil
                        divide = 10
                    }
                }
                change_variable = { name = research_progress value = var:temp_progress_bonus }
                remove_variable = temp_progress_bonus
            }
            root = { remove_variable = temp_scientist_char }
        }
    }
}
