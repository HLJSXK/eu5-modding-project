### Changes to harder AI difficulties to ensure better integration with the rest of the mod. ###
# Removed population growth buffs to AI (should help prevent snowballing and reduce lag)
# Removed food consumption buffs (forcing the AI to deal with the mod's increased food consumption to the fullest extent)
# Tax efficiency buffs heavily nerfed. 
# Nerfed AI bonus to estate satisfaction on harder difficulties since, while it makes AI countries stronger, it also makes them boring.

TRY_REPLACE:difficulty_ai_hard = {
	game_data = {
		category = country
	}
	discipline = 0.1
	naval_damage_done = 0.1
	naval_damage_taken = -0.1
	tax_income_efficiency = 0.02 # 2% bonus | Vanilla: 0.1 (10%)
	research_speed_modifier = 0.1
	global_estate_target_satisfaction = 0.05 # 5% bonus | Vanilla: 0.1 (10% bonus)
	trade_efficiency = medium_trade_efficiency_bonus
	global_estate_satisfaction_recovery = 0.0005 # +0.05% bonus | Vanilla: 0.002 (0.20%)
	# global_population_growth = 0.0005

}
