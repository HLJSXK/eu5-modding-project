## Province-scope Little Ice Age mitigation — permanent SOL rebalance, NOT CMM-toggleable.
## Province auto_modifiers are not supported by EU5 (only country/location/dynasty/international_organization).
## These modifiers remain as TRY_REPLACE always-on, making LIA province events less punishing.
## The sol__lia CMM toggle controls country and location LIA relief; province relief is always active.

TRY_REPLACE:short_harvest_modifier = {
	game_data = {
		category = province
	}

	local_monthly_food_modifier = -0.25 # Vanilla: -0.5 (permanent SOL mitigation; cannot be CMM-toggled)
}

TRY_REPLACE:recent_white_storm_modifier = {
	game_data = {
		category = province
		decaying = yes
	}

	local_monthly_food_modifier = -0.25 # Vanilla: -0.5 (permanent SOL mitigation; cannot be CMM-toggled)
	local_population_growth = -0.005    # Same as vanilla
}
