# Can no longer reduce war exhaustion while at war

TRY_REPLACE:reduce_war_exhaustion = {
	ability = dip

	allow = {
		war_exhaustion > 0
		OR = {
			sol_as_is_on = no   # AS globally off: unrestricted
			sol_rwe_is_on = no  # rwe sub-toggle off: unrestricted
			at_war = no          # AS on + rwe on: restricted to peacetime
		}
	}
	
	is_finished = {
		war_exhaustion <= 0
	}

	country_modifier = {
		monthly_war_exhaustion = -0.05
	}
}
