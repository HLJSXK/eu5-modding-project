global_living_standard = {
	monthly_spawn_chance = 1
	hint_tag = hint_reformation

	can_start = {
		current_date >= 1337.2.1
	}

	can_end = {
		always = no
	}

	visible = {
		sol_sol_is_on = yes
	}

	on_start = {
		gls_recalculate_global_living_standard = yes
	}

	on_monthly = {
		if = {
			limit = { sol_sol_is_on = yes }
			# Pass A：地图依赖，所有国家（AI + 人类），每年一次
			# 更新 savings_pressure（地图着色依赖）和地点变量。
			every_country = {
				if = {
					limit = {
						total_population > 0
						OR = {
							current_month = 1
							NOT = { has_variable = gls_map_cache_initialized }
						}
					}
					gls_compute_savings_pressure = yes
					gls_update_location_map = yes
				}
			}
			# Pass B：面板，仅人类玩家国家，每月
			every_country = {
				if = {
					limit = {
						total_population > 0
						is_human = yes
					}
					gls_compute_savings_pressure = yes
					gls_compute_panel_display = yes
					if = {
						limit = {
							OR = {
								current_month = 1
								NOT = { has_variable = gls_country_sol_all }
							}
						}
						gls_accumulate_panel_stats = yes
					}
				}
			}
		}
	}

	tooltip = {
		if = {
			limit = {
				is_land = yes
				has_owner = yes
				has_variable = gls_location_sol_weighted
				var:gls_location_sol_weighted >= 50
			}
			custom_tooltip = GLS_LOCATION_SOL_HIGH_TT
			custom_tooltip = GLS_LOCATION_SOL_VALUE_TT
		}
		else_if = {
			limit = {
				is_land = yes
				has_owner = yes
				has_variable = gls_location_sol_weighted
				var:gls_location_sol_weighted >= 25
			}
			custom_tooltip = GLS_LOCATION_SOL_MID_TT
			custom_tooltip = GLS_LOCATION_SOL_VALUE_TT
		}
		else_if = {
			limit = {
				is_land = yes
				has_owner = yes
				has_variable = gls_location_sol_weighted
			}
			custom_tooltip = GLS_LOCATION_SOL_LOW_TT
			custom_tooltip = GLS_LOCATION_SOL_VALUE_TT
		}
		else_if = {
			limit = {
				is_land = yes
				has_owner = yes
			}
			custom_tooltip = GLS_LOCATION_SOL_MAP_TT
		}
	}

	map_color = {
		if = {
			limit = {
				sol_map_is_on = yes
				is_land = yes
				has_owner = yes
				has_variable = gls_location_avg_scale
			}
			lerp = {
				min_color = define:NMapColors|MAP_COLOR_LOW
				max_color = define:NMapColors|MAP_COLOR_HIGH
				factor = "gls_location_color_factor"
			}
		}
		else = {
			value = define:NMapColors|DEFAULT_COLOR
		}
	}

	legend_key = {
		desc = "GLS_MAP_LEGEND_HIGH"
		color = define:NMapColors|MAP_COLOR_HIGH
	}

	legend_key = {
		desc = "GLS_MAP_LEGEND_MID"
		color = define:NMapColors|MAP_COLOR_MID
	}

	legend_key = {
		desc = "GLS_MAP_LEGEND_LOW"
		color = define:NMapColors|MAP_COLOR_LOW
	}

}
