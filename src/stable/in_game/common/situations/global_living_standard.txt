global_living_standard = {
	monthly_spawn_chance = 1
	hint_tag = hint_reformation

	can_start = {
		current_date >= 1337.2.1
	}

	can_end = {
		always = no
	}

	visible = {
		always = yes
	}

	on_start = {
		gls_recalculate_global_living_standard = yes
	}

	on_monthly = {
		every_country = {
			if = {
				limit = { total_population > 0 }
				gls_update_country_visual_cache = yes
			}
		}
	}

	tooltip = {
		if = {
			limit = {
				is_land = yes
				has_owner = yes
				has_variable = gls_location_avg_scale
				var:gls_location_avg_scale >= 1.35
			}
			custom_tooltip = GLS_LOCATION_SOL_HIGH_TT
			custom_tooltip = GLS_LOCATION_SOL_VALUE_TT
		}
		else_if = {
			limit = {
				is_land = yes
				has_owner = yes
				has_variable = gls_location_avg_scale
				var:gls_location_avg_scale >= 1.15
			}
			custom_tooltip = GLS_LOCATION_SOL_MID_TT
			custom_tooltip = GLS_LOCATION_SOL_VALUE_TT
		}
		else_if = {
			limit = {
				is_land = yes
				has_owner = yes
				has_variable = gls_location_avg_scale
			}
			custom_tooltip = GLS_LOCATION_SOL_LOW_TT
			custom_tooltip = GLS_LOCATION_SOL_VALUE_TT
		}
		else_if = {
			limit = {
				is_land = yes
				has_owner = yes
			}
			custom_tooltip = GLS_LOCATION_SOL_MAP_TT
		}
	}

	map_color = {
		if = {
			limit = {
				is_land = yes
				has_owner = yes
				var:gls_location_avg_scale >= 1.35
			}
			value = define:NMapColors|MAP_COLOR_HIGH
		}
		else_if = {
			limit = {
				is_land = yes
				has_owner = yes
				var:gls_location_avg_scale >= 1.15
			}
			value = define:NMapColors|MAP_COLOR_MID
		}
		else_if = {
			limit = {
				is_land = yes
				has_owner = yes
				var:gls_location_avg_scale > 0
			}
			value = define:NMapColors|MAP_COLOR_LOW
		}
		else = {
			value = define:NMapColors|DEFAULT_COLOR
		}
	}

	legend_key = {
		desc = "GLS_MAP_LEGEND_HIGH"
		color = define:NMapColors|MAP_COLOR_HIGH
	}

	legend_key = {
		desc = "GLS_MAP_LEGEND_MID"
		color = define:NMapColors|MAP_COLOR_MID
	}

	legend_key = {
		desc = "GLS_MAP_LEGEND_LOW"
		color = define:NMapColors|MAP_COLOR_LOW
	}

}
