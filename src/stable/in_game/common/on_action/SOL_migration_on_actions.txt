# SOL Migration Notice
#
# Detection is entirely built-in via GUI widgets — no CMF dependency.
#   Widget 1 (sol_after_lobby_caller):   detects new game vs old save, stamps new games
#   Widget 2 (sol_load_after_lobby_caller): fires migration event for old saves
#   See: SOL_migration_caller.gui / SOL_migration_scripted_guis.txt
#
# Tracking variables (all global, persist in saves):
#   sol_game_started_after_lobby — set once our widget has run; marks "we've been past lobby"
#   sol_is_new_game              — transient; set by Widget 1, cleared by Widget 2
#   SOL_migration_v2_seen        — permanent stamp; prevents re-firing

# Fires migration event for all human players from global scope.
# Called by sol_load_after_lobby_caller scripted GUI via trigger_event_silently.
# Scope: none (global)
SOL_migration_fire_event = {
	effect = {
		set_global_variable = SOL_migration_v2_seen
		every_country = {
			limit = { is_ai = no }
			trigger_event_non_silently = sol_migration.1
		}
	}
}
