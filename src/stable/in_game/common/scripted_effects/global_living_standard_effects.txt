gls_refresh_country_pool = {
	clear_global_variable_list = gls_all_country_pool
	clear_global_variable_list = gls_nobles_country_pool
	clear_global_variable_list = gls_clergy_country_pool
	clear_global_variable_list = gls_burghers_country_pool
	clear_global_variable_list = gls_commoners_country_pool
	clear_global_variable_list = gls_tribesmen_country_pool
	every_country = {
		if = {
			limit = { total_population >= 1 }
			add_to_global_variable_list = { name = gls_all_country_pool target = this }
		}
		if = {
			limit = { num_pop_type_in_country:nobles >= 1 }
			add_to_global_variable_list = { name = gls_nobles_country_pool target = this }
		}
		if = {
			limit = { num_pop_type_in_country:clergy >= 1 }
			add_to_global_variable_list = { name = gls_clergy_country_pool target = this }
		}
		if = {
			limit = { num_pop_type_in_country:burghers >= 1 }
			add_to_global_variable_list = { name = gls_burghers_country_pool target = this }
		}
		if = {
			limit = {
				OR = {
					num_pop_type_in_country:laborers >= 1
					num_pop_type_in_country:peasants >= 1
					num_pop_type_in_country:soldiers >= 1
				}
			}
			add_to_global_variable_list = { name = gls_commoners_country_pool target = this }
		}
		if = {
			limit = { num_pop_type_in_country:tribesmen >= 1 }
			add_to_global_variable_list = { name = gls_tribesmen_country_pool target = this }
		}
	}
}

gls_calculate_country_scores = {
	# root = country
	set_variable = { name = gls_all_score value = 0 }
	set_variable = { name = gls_nobles_score value = 0 }
	set_variable = { name = gls_clergy_score value = 0 }
	set_variable = { name = gls_burghers_score value = 0 }
	set_variable = { name = gls_commoners_score value = 0 }
	set_variable = { name = gls_tribesmen_score value = 0 }

	if = {
		limit = { num_locations > 0 }
		every_owned_location = {
			set_variable = {
				name = gls_loc_dev
				value = { add = development }
			}

			set_variable = { name = gls_loc_nobles_score value = 0 }
			if = {
				limit = { owner = { num_pop_type_in_country:nobles > 0 } }
				set_variable = {
					name = gls_loc_nobles_score
					value = {
						add = local_noble_gdp_display
						multiply = var:gls_loc_dev
					}
				}
			}

			set_variable = { name = gls_loc_clergy_score value = 0 }
			if = {
				limit = { owner = { num_pop_type_in_country:clergy > 0 } }
				set_variable = {
					name = gls_loc_clergy_score
					value = {
						add = local_clergy_gdp_display
						multiply = var:gls_loc_dev
					}
				}
			}

			set_variable = { name = gls_loc_burghers_score value = 0 }
			if = {
				limit = { owner = { num_pop_type_in_country:burghers > 0 } }
				set_variable = {
					name = gls_loc_burghers_score
					value = {
						add = local_burghers_gdp_display
						multiply = var:gls_loc_dev
					}
				}
			}

			set_variable = { name = gls_loc_commoners_score value = 0 }
			if = {
				limit = {
					owner = {
						OR = {
							num_pop_type_in_country:laborers > 0
							num_pop_type_in_country:peasants > 0
							num_pop_type_in_country:soldiers > 0
						}
					}
				}
				set_variable = {
					name = gls_loc_commoners_score
					value = {
						add = local_commoner_gdp_display
						multiply = var:gls_loc_dev
					}
				}
			}

			set_variable = { name = gls_loc_tribesmen_score value = 0 }
			if = {
				limit = { owner = { num_pop_type_in_country:tribesmen > 0 } }
				set_variable = {
					name = gls_loc_tribesmen_score
					value = {
						add = local_tribesmen_gdp_display
						multiply = var:gls_loc_dev
					}
				}
			}

				set_variable = {
					name = gls_loc_all_score
					value = {
						add = var:gls_loc_nobles_score
						add = var:gls_loc_clergy_score
						add = var:gls_loc_burghers_score
						add = var:gls_loc_commoners_score
						add = var:gls_loc_tribesmen_score
					}
				}

			owner = {
					change_variable = { name = gls_all_score add = prev.var:gls_loc_all_score }
				change_variable = { name = gls_nobles_score add = prev.var:gls_loc_nobles_score }
				change_variable = { name = gls_clergy_score add = prev.var:gls_loc_clergy_score }
				change_variable = { name = gls_burghers_score add = prev.var:gls_loc_burghers_score }
				change_variable = { name = gls_commoners_score add = prev.var:gls_loc_commoners_score }
				change_variable = { name = gls_tribesmen_score add = prev.var:gls_loc_tribesmen_score }
			}

			remove_variable = gls_loc_dev
			remove_variable = gls_loc_nobles_score
			remove_variable = gls_loc_clergy_score
			remove_variable = gls_loc_burghers_score
			remove_variable = gls_loc_commoners_score
			remove_variable = gls_loc_tribesmen_score
				remove_variable = gls_loc_all_score
		}
	}

	if = {
		limit = { num_locations > 0 }

			if = {
				limit = { total_population > 0 }
				set_variable = {
					name = gls_all_score
					value = {
						add = var:gls_all_score
						divide = total_population
						divide = num_locations
					}
				}
			}

		if = {
			limit = { num_pop_type_in_country:nobles > 0 }
			set_variable = {
				name = gls_nobles_score
				value = {
					add = var:gls_nobles_score
					divide = num_pop_type_in_country:nobles
					divide = num_locations
				}
			}
		}

		if = {
			limit = { num_pop_type_in_country:clergy > 0 }
			set_variable = {
				name = gls_clergy_score
				value = {
					add = var:gls_clergy_score
					divide = num_pop_type_in_country:clergy
					divide = num_locations
				}
			}
		}

		if = {
			limit = { num_pop_type_in_country:burghers > 0 }
			set_variable = {
				name = gls_burghers_score
				value = {
					add = var:gls_burghers_score
					divide = num_pop_type_in_country:burghers
					divide = num_locations
				}
			}
		}

		if = {
			limit = {
				OR = {
					num_pop_type_in_country:laborers > 0
					num_pop_type_in_country:peasants > 0
					num_pop_type_in_country:soldiers > 0
				}
			}
			set_variable = {
				name = gls_commoners_score
				value = {
					add = var:gls_commoners_score
					divide = {
						add = num_pop_type_in_country:laborers
						add = num_pop_type_in_country:peasants
						add = num_pop_type_in_country:soldiers
					}
					divide = num_locations
				}
			}
		}

		if = {
			limit = { num_pop_type_in_country:tribesmen > 0 }
			set_variable = {
				name = gls_tribesmen_score
				value = {
					add = var:gls_tribesmen_score
					divide = num_pop_type_in_country:tribesmen
					divide = num_locations
				}
			}
		}
	}
}
