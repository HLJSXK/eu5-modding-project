gls_refresh_country_pool = {
	clear_global_variable_list = gls_country_pool
	every_country = {
		add_to_global_variable_list = { name = gls_country_pool target = this }
	}
}

gls_calculate_country_scores = {
	# root = country
	set_variable = { name = gls_avg_dev value = 1 }
	if = {
		limit = { num_locations > 0 }
		set_variable = {
			name = gls_avg_dev
			value = {
				add = total_development
				divide = num_locations
			}
		}
	}

	set_variable = { name = gls_nobles_income_sum value = 0 }
	set_variable = { name = gls_clergy_income_sum value = 0 }
	set_variable = { name = gls_burghers_income_sum value = 0 }
	set_variable = { name = gls_commoners_income_sum value = 0 }
	set_variable = { name = gls_tribesmen_income_sum value = 0 }

	every_owned_location = {
		set_variable = {
			name = gls_loc_nobles_income
			value = { add = local_noble_gdp_display }
		}
		set_variable = {
			name = gls_loc_clergy_income
			value = { add = local_clergy_gdp_display }
		}
		set_variable = {
			name = gls_loc_burghers_income
			value = { add = local_burghers_gdp_display }
		}
		set_variable = {
			name = gls_loc_commoners_income
			value = { add = local_commoner_gdp_display }
		}
		set_variable = {
			name = gls_loc_tribesmen_income
			value = { add = local_tribe_gdp_display }
		}
		owner = {
			change_variable = { name = gls_nobles_income_sum add = prev.var:gls_loc_nobles_income }
			change_variable = { name = gls_clergy_income_sum add = prev.var:gls_loc_clergy_income }
			change_variable = { name = gls_burghers_income_sum add = prev.var:gls_loc_burghers_income }
			change_variable = { name = gls_commoners_income_sum add = prev.var:gls_loc_commoners_income }
			change_variable = { name = gls_tribesmen_income_sum add = prev.var:gls_loc_tribesmen_income }
		}

		remove_variable = gls_loc_nobles_income
		remove_variable = gls_loc_clergy_income
		remove_variable = gls_loc_burghers_income
		remove_variable = gls_loc_commoners_income
		remove_variable = gls_loc_tribesmen_income
	}

	set_variable = {
		name = gls_nobles_score
		value = {
			add = var:gls_nobles_income_sum
			divide = {
				add = num_pop_type_in_country:nobles
				min = 0.001
			}
			multiply = var:gls_avg_dev
		}
	}

	set_variable = {
		name = gls_clergy_score
		value = {
			add = var:gls_clergy_income_sum
			divide = {
				add = num_pop_type_in_country:clergy
				min = 0.001
			}
			multiply = var:gls_avg_dev
		}
	}

	set_variable = {
		name = gls_burghers_score
		value = {
			add = var:gls_burghers_income_sum
			divide = {
				add = num_pop_type_in_country:burghers
				min = 0.001
			}
			multiply = var:gls_avg_dev
		}
	}

	set_variable = {
		name = gls_commoners_score
		value = {
			add = var:gls_commoners_income_sum
			divide = {
				add = num_pop_type_in_country:laborers
				add = num_pop_type_in_country:peasants
				add = num_pop_type_in_country:soldiers
				min = 0.001
			}
			multiply = var:gls_avg_dev
		}
	}

	set_variable = {
		name = gls_tribesmen_score
		value = {
			add = var:gls_tribesmen_income_sum
			divide = {
				add = num_pop_type_in_country:tribesmen
				min = 0.001
			}
			multiply = var:gls_avg_dev
		}
	}
}

			remove_variable = gls_tribesmen_top4_value
		}
	}

	if = {
		limit = {
			global_variable_list_size = {
				name = gls_tribesmen_candidates
				value > 0
			}
		}
		ordered_in_global_list = {
			variable = gls_tribesmen_candidates
			order_by = { add = { value = gls_tribesmen_score } }
			max = 1
			check_range_bounds = no
			situation:global_living_standard = {
				set_variable = { name = gls_tribesmen_top5_country value = prev }
				set_variable = { name = gls_tribesmen_top5_value value = prev.var:gls_tribesmen_score }
			}
			remove_list_global_variable = {
				name = gls_tribesmen_candidates
				target = prev
			}
		}
	}
	else = {
		situation:global_living_standard = {
			remove_variable = gls_tribesmen_top5_country
			remove_variable = gls_tribesmen_top5_value
		}
	}
}
