gls_update_location_map = {
	# 设置地图着色所需的地点变量，对 AI 和人类国家均执行。
	# 调用者：Pass A（所有国家，每年）
	# root = country
	if = {
		limit = { num_locations > 0 }
		every_owned_location = {
			limit = { is_land = yes }
			set_variable = {
				name = gls_location_avg_scale
				value = { add = local_average_gdp_per_capita_scale }
			}
			set_variable = {
				name = gls_location_sol_weighted
				value = {
					value = var:gls_location_avg_scale
					multiply = development
				}
			}
		}
	}
	set_variable = { name = gls_map_cache_initialized value = 1 }
}

gls_accumulate_panel_stats = {
	# 地点循环累加 + 聚合/SoL/需求计算，仅对人类国家执行。
	# 依赖：gls_institution_bonus（由 gls_compute_panel_display 设置）。
	# 调用者：Pass B 内年度条件块（人类，每年）
	# root = country
	set_variable = { name = gls_location_count value = 0 }
	set_variable = { name = gls_location_scale_sum value = 0 }
	set_variable = { name = gls_country_wealth value = 0 }
	set_variable = { name = gls_development_sum value = 0 }
	set_variable = { name = gls_control_sum value = 0 }

	set_variable = { name = gls_sum_nobles_offset value = 0 }
	set_variable = { name = gls_sum_clergy_offset value = 0 }
	set_variable = { name = gls_sum_burghers_offset value = 0 }
	set_variable = { name = gls_sum_commoners_offset value = 0 }
	set_variable = { name = gls_sum_tribesmen_offset value = 0 }
	set_variable = { name = gls_sum_all_offset value = 0 }

	set_variable = { name = gls_pop_nobles value = 0 }
	set_variable = { name = gls_pop_clergy value = 0 }
	set_variable = { name = gls_pop_burghers value = 0 }
	set_variable = { name = gls_pop_commoners value = 0 }
	set_variable = { name = gls_pop_tribesmen value = 0 }
	set_variable = { name = gls_pop_all value = 0 }

	if = {
		limit = { num_locations > 0 }
		every_owned_location = {
			limit = { is_land = yes }
			save_scope_as = gls_cached_location

			owner = {
				change_variable = { name = gls_location_count add = 1 }
				change_variable = { name = gls_location_scale_sum add = prev.var:gls_location_avg_scale }
				change_variable = { name = gls_country_wealth add = scope:gls_cached_location.local_wealth_display }
				change_variable = { name = gls_development_sum add = scope:gls_cached_location.development }
				change_variable = { name = gls_control_sum add = scope:gls_cached_location.local_control }
			}

			if = {
				limit = { num_pop_type:nobles > 0 }
				owner = {
					change_variable = {
						name = gls_sum_nobles_offset
						add = {
							value = scope:gls_cached_location.local_noble_demand_scale_offset
							multiply = scope:gls_cached_location.num_pop_type:nobles
						}
					}
					change_variable = { name = gls_pop_nobles add = scope:gls_cached_location.num_pop_type:nobles }
				}
			}

			if = {
				limit = { num_pop_type:clergy > 0 }
				owner = {
					change_variable = {
						name = gls_sum_clergy_offset
						add = {
							value = scope:gls_cached_location.local_clergy_demand_scale_offset
							multiply = scope:gls_cached_location.num_pop_type:clergy
						}
					}
					change_variable = { name = gls_pop_clergy add = scope:gls_cached_location.num_pop_type:clergy }
				}
			}

			if = {
				limit = { num_pop_type:burghers > 0 }
				owner = {
					change_variable = {
						name = gls_sum_burghers_offset
						add = {
							value = scope:gls_cached_location.local_burghers_demand_scale_offset
							multiply = scope:gls_cached_location.num_pop_type:burghers
						}
					}
					change_variable = { name = gls_pop_burghers add = scope:gls_cached_location.num_pop_type:burghers }
				}
			}

			if = {
				limit = { local_commoner_population > 0 }
				owner = {
					change_variable = {
						name = gls_sum_commoners_offset
						add = {
							value = scope:gls_cached_location.local_commoner_demand_scale_offset
							multiply = scope:gls_cached_location.local_commoner_population
						}
					}
					change_variable = { name = gls_pop_commoners add = scope:gls_cached_location.local_commoner_population }
				}
			}

			if = {
				limit = { num_pop_type:tribesmen > 0 }
				owner = {
					change_variable = {
						name = gls_sum_tribesmen_offset
						add = {
							value = scope:gls_cached_location.local_tribesmen_demand_scale_offset
							multiply = scope:gls_cached_location.num_pop_type:tribesmen
						}
					}
					change_variable = { name = gls_pop_tribesmen add = scope:gls_cached_location.num_pop_type:tribesmen }
				}
			}
		}
	}

	set_variable = {
		name = gls_pop_all
		value = {
			add = var:gls_pop_nobles
			add = var:gls_pop_clergy
			add = var:gls_pop_burghers
			add = var:gls_pop_commoners
			add = var:gls_pop_tribesmen
		}
	}

	set_variable = {
		name = gls_sum_all_offset
		value = {
			add = var:gls_sum_nobles_offset
			add = var:gls_sum_clergy_offset
			add = var:gls_sum_burghers_offset
			add = var:gls_sum_commoners_offset
			add = var:gls_sum_tribesmen_offset
		}
	}

	set_variable = { name = gls_country_location_avg_scale value = 0 }
	if = {
		limit = { var:gls_location_count > 0 }
		set_variable = {
			name = gls_country_location_avg_scale
			value = {
				add = var:gls_location_scale_sum
				divide = var:gls_location_count
			}
		}
	}

	set_variable = { name = gls_avg_development value = 0 }
	if = {
		limit = { var:gls_location_count > 0 }
		set_variable = {
			name = gls_avg_development
			value = {
				add = var:gls_development_sum
				divide = var:gls_location_count
			}
		}
	}

	set_variable = { name = gls_avg_control value = 0 }
	if = {
		limit = { var:gls_location_count > 0 }
		set_variable = {
			name = gls_avg_control
			value = {
				add = var:gls_control_sum
				divide = var:gls_location_count
			}
		}
	}

	# Development demand multiplier = avg_development / 10 (display + demand formula)
	set_variable = {
		name = gls_development_multiplier
		value = {
			value = 0
			if = {
				limit = { var:gls_avg_development > 0 }
				add = var:gls_avg_development
				divide = 10
			}
		}
	}

	set_variable = { name = gls_country_sol_nobles value = 0 }
	if = {
		limit = { var:gls_pop_nobles > 0 }
		set_variable = {
			name = gls_country_sol_nobles

			name = gls_country_sol_clergy

				add = {
					value = var:gls_sum_clergy_offset
					divide = var:gls_pop_clergy
				}
			}
		}
	}

	set_variable = { name = gls_country_sol_burghers value = 0 }
	if = {
		limit = { var:gls_pop_burghers > 0 }
		set_variable = {
			name = gls_country_sol_burghers
			value = {
				add = 1
				add = {
					value = var:gls_sum_burghers_offset
					divide = var:gls_pop_burghers
				}
			}
		}
	}

	set_variable = { name = gls_country_sol_commoners value = 0 }
	if = {
		limit = { var:gls_pop_commoners > 0 }
		set_variable = {
			name = gls_country_sol_commoners
			value = {
				add = 1
				add = {
					value = var:gls_sum_commoners_offset
					divide = var:gls_pop_commoners
				}
			}
		}
	}

	set_variable = { name = gls_country_sol_tribesmen value = 0 }
	if = {
		limit = { var:gls_pop_tribesmen > 0 }
		set_variable = {
			name = gls_country_sol_tribesmen
			value = {
				add = 1
				add = {
					value = var:gls_sum_tribesmen_offset
					divide = var:gls_pop_tribesmen
				}
			}
		}
	}

	set_variable = { name = gls_country_sol_all value = 0 }
	if = {
		limit = { var:gls_pop_all > 0 }
		set_variable = {
			name = gls_country_sol_all
			value = {
				add = 1
				add = {
					value = var:gls_sum_all_offset
					divide = var:gls_pop_all
				}
			}
		}
	}

	# Nonlinear income component per estate (= sol - 1 - savings_pressure).
	# Derived after per-estate SOL is known; monthly cache must run first.
	set_variable = {
		name = gls_nobles_nonlinear
		value = {
			value = 0
			if = {
				limit = { has_variable = gls_country_sol_nobles }

				subtract = 1
				if = {
					limit = { has_variable = gls_nobles_savings_pressure }
