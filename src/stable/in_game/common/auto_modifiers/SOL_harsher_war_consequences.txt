## Harsher War Consequences (hwc) — location-scope auto_modifiers.
# Controlled by CMM toggle sol__hwc (default ON).
# When ON: applies the SOL delta on top of vanilla location modifiers.
# When OFF: vanilla static modifiers apply without modification.

# is_blockaded_by_enemies delta:
# Vanilla: prosperity -0.025, maritime -2.0, max_control -0.10, construction -0.10
# SOL:     prosperity -0.05,  maritime -5.0, max_control -0.30, construction -0.30, migration -0.5 (new)
# Delta (this modifier): prosperity -0.025, maritime -3.0, max_control -0.20, construction -0.20, migration -0.5
sol_hwc_blockade_extras = {
	category = location
	type = location
	potential_trigger = {
		sol_hwc_is_on = yes
		has_location_modifier = is_blockaded_by_enemies
	}
	local_monthly_prosperity = -0.025
	local_maritime_presence = -3.0
	local_max_control = -0.2
	local_construction_speed = -0.2
	local_migration_attraction = -0.5
}

# under_siege: change prosperity mechanism from rate-decay to flat monthly.
# Vanilla: local_prosperity_decay = 0.05 (rate-based, much harsher at high prosperity)
# SOL:     local_monthly_prosperity = -0.05 (flat monthly hit)
# Delta: neutralize vanilla decay, apply flat monthly version.
sol_hwc_siege_prosperity = {
	category = location
	type = location
	potential_trigger = {
		sol_hwc_is_on = yes
		has_location_modifier = under_siege
	}
	local_prosperity_decay = -0.05   # Neutralizes vanilla 0.05 decay
	local_monthly_prosperity = -0.05 # SOL flat monthly replacement
}
