## Location Base Values (lb) and Inverse Control (ic) — location-scope auto_modifiers.
# lb toggle (sol__lb): controls RGO size reduction and control gain improvement.
# ic toggle (sol__ic): controls building cost penalty registration (see note below).
#
# Technical note on inverse_control:
# The vanilla inverse_control static modifier scales with (1 - control). Our mod adds
# local_build_buildings_cost = 0.5 to TRY_REPLACE:inverse_control, which is correctly
# scaled. This scaling cannot be cleanly replicated in an auto_modifier without confirmed
# access to the inverse_control scalar. Therefore, sol__ic is registered in CMM for
# future use, but the building cost effect remains in TRY_REPLACE (always-on with
# correct control scaling). See A_SOL_location.txt for the TRY_REPLACE definition.

# location_base_values delta:
# Vanilla: local_max_rgo_size = 2, local_monthly_control = 0.0005
# SOL:     local_max_rgo_size = 1, local_monthly_control = 0.001
# Delta (this modifier): rgo_size -1, monthly_control +0.0005
# Note: location_base_values applies to every location always (no condition check needed
# in potential_trigger beyond the CMM toggle).
sol_lb_base_values_delta = {
	category = location
	type = location
	potential_trigger = {
		sol_lb_is_on = yes
	}
	local_max_rgo_size = -1
	local_monthly_control = 0.0005
}

# total_population delta:
# Vanilla: local_max_rgo_size = 0.025 per population unit
# SOL:     local_max_rgo_size = 0.020 per population unit
# Delta (this modifier): -0.005 per population unit (scales with total_population)
sol_lb_population_delta = {
	category = location
	type = location
	potential_trigger = {
		sol_lb_is_on = yes
	}
	scales_with = total_population
	local_max_rgo_size = -0.005
}
