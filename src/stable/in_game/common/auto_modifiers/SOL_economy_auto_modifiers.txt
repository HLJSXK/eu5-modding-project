### ============================================================
### SECTION: Base Tax Efficiency
### ============================================================
# Every country receives a base -15% tax efficiency penalty.
# This is stricter than the original mod's -10%, but the mod is
# more generous with bonuses elsewhere, so it can be overcome.
# Scaled by sol_as_bte_scale (CMM-adjustable).

base_tax_efficiency_602 = {
	potential_trigger = {
		exists = yes
		sol_as_is_on = yes
	}
	scales_with = sol_as_bte_scale
	tax_income_efficiency = 1
}

### ============================================================
### SECTION: Diplomatic Spending
### ============================================================
# Adds a dynamic diplomatic spending component for all real countries.
# Final diplomatic spending cost: 0.1 (vanilla) + (-0.05 + used_cap/200)
# = 0.05 + used_diplomatic_capacity/200
# Scaled by sol_as_ds_scale (CMM-adjustable).

sol_diplomatic_spending_dynamic = {
	potential_trigger = {
		exists = yes
		sol_as_is_on = yes
	}

	scales_with = sol_as_ds_scale

	diplomatic_spending_cost = 1
}

### ============================================================
### SECTION: Stability Cost Discount
### ============================================================
# Reduces stability investment cost based on cultural tradition.
# Formula: 0.05 - 0.05 / (cultural_tradition + 1)
# Applied as a negative modifier on stability_cost.
# Scaled by sol_cultural_tradition_stability_discount (script value).

sol_cultural_tradition_stability_discount = {
	potential_trigger = {
		exists = yes
		sol_as_is_on = yes
	}

	scales_with = sol_cultural_tradition_stability_discount

	stability_cost = -1
}

### ============================================================
### SECTION: Difficulty Rebalance
### ============================================================
# Applies a negative delta to tax_income_efficiency for AI Hard/Very Hard
# and Player Easy/Very Easy, halving the vanilla difficulty bonus.
#
# CMM toggle: sol__difficulty (default ON)
# When ON:  vanilla bonus is reduced (SOL balanced values apply)
# When OFF: vanilla difficulty static modifiers apply unmodified

sol_difficulty_ai_hard_mod = {
	potential_trigger = {
		exists = yes
		is_ai = yes
		has_game_rule = ai_hard_difficulty
		sol_as_is_on = yes
		sol_difficulty_is_on = yes
	}

	tax_income_efficiency = -0.025  # Vanilla AI Hard: +0.05 → SOL: +0.025
}

	potential_trigger = {
