# Adds a dynamic diplomatic spending component for all real countries.
# Final diplomatic spending cost becomes:
#   0.1 (vanilla base) + (-0.05 + used_diplomatic_capacity/200)
# = 0.05 + used_diplomatic_capacity/200

sol_diplomatic_spending_dynamic = {
    potential_trigger = {
        exists = yes
        sol_as_is_on = yes
    }

    # Additive modifier on top of vanilla 0.1 base:
    # -0.05 + used_diplomatic_capacity / 200
    scales_with = {
        value = used_diplomatic_capacity
        multiply = 0.005
        add = -0.05
    }

    diplomatic_spending_cost = 1
}
