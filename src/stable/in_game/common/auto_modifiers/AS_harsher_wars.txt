## Changes are from the Harsher Wars mod. 
# War exhaustion is more impactful.

TRY_REPLACE:war_exhaustion_impact = {
	scales_with = war_exhaustion

	land_morale_modifier = -0.02
	naval_morale_modifier = -0.02
	global_levy_size_modifier = -0.05 # Vanilla: None
	
	
	global_production_efficiency = scaled_production_efficiency_penalty
	trade_efficiency = scaled_trade_efficiency_penalty
	global_population_growth = -0.0003
    global_monthly_control = -0.0001
    stability_decay = 0.0001 # Vanilla: None
	monthly_legitimacy = -0.01 # Vanilla: None
	global_defensive = -0.05 # Vanilla: None | Should make sieges much easier if the country is exhausted enough.
}

TRY_REPLACE:capital_occupied = {
	potential_trigger = { 
		at_war = yes
		OR = {
			country_type = location
			country_type = army
		}
		capital ?= {
			AND = {
				exists = owner
				owner != controller
			}
		}
	}
	
	monthly_war_exhaustion = 0.3 # Vanilla: 0.1 (Original Harsher Wars mod sets this to 0.5)
}
