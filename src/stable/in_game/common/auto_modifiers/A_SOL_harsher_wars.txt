## Changes are from the Harsher Wars mod.
# War exhaustion is more impactful.
# Vanilla war_exhaustion_impact and capital_occupied always apply.
# These modifiers add the AS delta on top when sol_as_is_on = yes.

# total_occupied: WE delta controlled by CMM slider sol__hw_we.
# Vanilla total_occupied = 0.4/month. Slider 0 → +0.0 (vanilla). Default 16 → +1.6 → net 2.0/month.
# Note: potential_trigger has no sol_as_is_on check intentionally — this is a standalone hw feature.
sol_hw_total_occupied_we = {
	potential_trigger = {
		exists = yes
		modifier:total_occupied = yes
	}
	scales_with = sol_hw_we_delta
	monthly_war_exhaustion = 1
}

sol_as_war_exhaustion_extras = {
	potential_trigger = {
		exists = yes
		sol_as_is_on = yes
		sol_hwe_is_on = yes
	}

	scales_with = war_exhaustion

	global_levy_size_modifier = -0.05 # Vanilla: None
	global_monthly_control = -0.0001  # Vanilla: None
	stability_decay = 0.0001          # Vanilla: None
	monthly_legitimacy = -0.01        # Vanilla: None
	global_defensive = -0.05          # Vanilla: None
}

sol_as_capital_occupied_extras = {
	potential_trigger = {
		sol_as_is_on = yes
		sol_iwe_is_on = yes
		at_war = yes
		OR = {
			country_type = location
			country_type = army
		}
		capital ?= {
			AND = {
				exists = owner
				owner != controller
			}
		}
	}

	# Delta vs vanilla (0.1/month): net result 0.3/month
	monthly_war_exhaustion = 0.2
}
