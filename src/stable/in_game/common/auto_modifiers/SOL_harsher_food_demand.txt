## Harsher Food Demand (hfd) — location-scope auto_modifiers.
# Controlled by CMM toggle sol__hfd (default ON).
# When ON: applies the SOL delta on top of vanilla winter/levy modifiers.
# When OFF: vanilla static modifiers apply without modification.

# winter_mild food delta:
# Vanilla: max_attrition 2, local_monthly_food -0.25
# SOL:     max_attrition 2, local_monthly_food -0.50
# Delta (this modifier): food -0.25
sol_hfd_winter_mild = {
	category = location
	type = location
	potential_trigger = {
		sol_hfd_is_on = yes
		has_location_modifier = winter_mild
	}
	local_monthly_food_modifier = -0.25
}

# winter_normal food delta:
# Vanilla: army_attrition 1, navy_attrition 5, max_attrition 2, food -0.50, food_decay 0.0025, construction -0.5
# SOL:     army_attrition 1, navy_attrition 5, max_attrition 2, food -1.00, (no food_decay), construction -0.5
# Delta (this modifier): food -0.50, neutralize food_decay
sol_hfd_winter_normal = {
	category = location
	type = location
	potential_trigger = {
		sol_hfd_is_on = yes
		has_location_modifier = winter_normal
	}
	local_monthly_food_modifier = -0.5
	local_food_decay = -0.0025  # Neutralizes vanilla 0.0025 food decay
}

# raised_levies extra penalties:
# Vanilla: raw_material -0.2, food -0.2
# SOL:     raw_material -0.3, food -0.3
# Delta (this modifier): raw_material -0.1, food -0.1
sol_hfd_raised_levies = {
	category = location
	type = location
	potential_trigger = {
		sol_hfd_is_on = yes
		has_location_modifier = raised_levies
	}
	local_raw_material_output = -0.1
	local_monthly_food_modifier = -0.1
}
