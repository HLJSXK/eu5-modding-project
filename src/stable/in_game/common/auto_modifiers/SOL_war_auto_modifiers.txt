## Changes are from the Harsher Wars mod.
# War exhaustion is more impactful.
# Vanilla war_exhaustion_impact and capital_occupied always apply.
# These modifiers add the SOL delta on top when sol_hwe_is_on / sol_iwe_is_on = yes.
# Both triggers include the war acceleration master check (sol_hw_is_on).

sol_as_war_exhaustion_extras = {
	potential_trigger = {
		exists = yes
		sol_hwe_is_on = yes
	}

	scales_with = war_exhaustion

	global_levy_size_modifier = -0.05 # Vanilla: None
	global_monthly_control = -0.0001  # Vanilla: None
	stability_decay = 0.0001          # Vanilla: None
	monthly_legitimacy = -0.01        # Vanilla: None
	global_defensive = -0.05          # Vanilla: None
}

# total_occupied war exhaustion delta:
# Vanilla: monthly_war_exhaustion 0.4    SOL target: 2.0
# Delta (this modifier): +1.6/month when total_occupied is active and hwe is ON
sol_hw_total_occupied_we = {
	potential_trigger = {
		exists = yes
		sol_hwe_is_on = yes
		has_country_modifier = total_occupied
	}

	monthly_war_exhaustion = 1.6
}

sol_as_capital_occupied_extras = {
	potential_trigger = {
		sol_iwe_is_on = yes
		at_war = yes
		OR = {
			country_type = location
			country_type = army
		}
		capital ?= {
			AND = {
				exists = owner
				owner != controller
			}
		}
	}

	# Delta vs vanilla (0.1/month): net result 0.3/month
	monthly_war_exhaustion = 0.2
}
