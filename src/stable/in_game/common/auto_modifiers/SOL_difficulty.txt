# SOL Difficulty Tax Nerf — Auto-Modifiers
#
# Applies a negative delta to tax_income_efficiency on AI Hard/Very Hard
# and Player Easy/Very Easy, halving the vanilla bonus.
#
# CMM toggle: sol__difficulty (default ON)
# When ON:  vanilla bonus is reduced (SOL balanced values apply)
# When OFF: vanilla difficulty static modifiers apply unmodified

sol_difficulty_ai_hard_mod = {
	potential_trigger = {
		exists = yes
		is_ai = yes
		has_game_rule = ai_hard_difficulty
		sol_as_is_on = yes
		sol_difficulty_is_on = yes
	}

	tax_income_efficiency = -0.025  # Vanilla AI Hard: +0.05 → SOL: +0.025
}
