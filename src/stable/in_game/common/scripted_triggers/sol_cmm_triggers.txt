# Scope: country
# True when CMM anti-snowballing toggle is ON.
# Falls back to ON when CMM is not installed (has_global_variable_map = cmm is false).
sol_as_is_on = {
	custom_description = {
		text = sol_as_is_on_desc
		OR = {
			NOT = { has_global_variable_map = cmm }  # CMM not installed → AS defaults ON
			has_global_variable = sol_as_enabled      # CMM installed → alias set when ON
		}
	}
}
