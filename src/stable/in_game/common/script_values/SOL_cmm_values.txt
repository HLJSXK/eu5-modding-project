# Script values for CMM-adjustable AS modifier settings.
# All values fall back to current hardcoded defaults when CMM is not installed.

# bte: returns global_var:sol_bte × 0.01
# Used as scales_with for base_tax_efficiency_602 (tax_income_efficiency = 1).
# Default sol_bte = -15 → -0.15 (matches original hardcoded value).
sol_as_bte_scale = {
	if = {
		limit = { has_global_variable = sol_bte }
		value = global_var:sol_bte
		multiply = 0.01
	}
	else = { value = -0.15 }
}

# cts: returns the coefficient for the cultural tradition formula.
# sol_cts is negative (e.g., -5), so coefficient = -sol_cts × 0.01 = 0.05.
# Used in sol_cultural_tradition_stability_discount script_value.
sol_as_cts_coeff = {
	if = {
		limit = { has_global_variable = sol_cts }
		value = global_var:sol_cts
		multiply = -0.01
	}
	else = { value = 0.1 }
}

# ds_coeff: base term of diplomatic spending formula = ds × 0.01.
# Default sol_ds = -5 → -0.05 (matches original formula constant).
sol_as_ds_coeff = {
	if = {
		limit = { has_global_variable = sol_ds }
		value = global_var:sol_ds
		multiply = 0.01
	}
	else = { value = -0.05 }
}

# ds_cap_factor: shape = (1 - used_diplomatic_capacity / 10).
# At cap=0: 1.0. At cap=10: 0.0. At cap=20: -1.0.
sol_as_ds_cap_factor = {
	value = used_diplomatic_capacity
	multiply = -0.1
	add = 1
}

# Combined diplomatic spending scale: ds_coeff × (1 - cap/10).
# At ds=-5, cap=0:  -0.05 ×  1.0 = -0.05 (5% reduction) ✓
# At ds=-5, cap=10: -0.05 ×  0.0 =  0.00 (neutral)       ✓
# At ds=-5, cap=20: -0.05 × -1.0 = +0.05 (5% increase)
sol_as_ds_scale = {
	value = sol_as_ds_coeff
	multiply = sol_as_ds_cap_factor
}
