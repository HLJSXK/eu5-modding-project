
local_nobles_tax_share = {

    multiply = 150

    add = num_pop_type:soldiers

    multiply = 1

local_tribesmen_tax_share = {
    add = num_pop_type:tribesmen
    multiply = 0.01
}

local_peasant_enfranchisment_clamped = {
    add = modifier:local_peasant_enfranchisment
    min = 0
    max = 1
}

local_commoner_share_transferred_to_nobles = {
    add = local_commoner_tax_share
    multiply = {
        value = 1
        subtract = local_peasant_enfranchisment_clamped

gdp_per_capita_scale = {
	value = 1
	# scale with GDP per capita
	if = {
		limit = { pop_type = pop_type:nobles }
		add = location.local_noble_sol
	}
	else_if = {
		limit = { pop_type = pop_type:clergy }
		add = location.local_clergy_sol
	}
	else_if = {
		limit = { pop_type = pop_type:burghers }
		add = location.local_burghers_sol
	}
	else_if = {
		limit = { pop_type = pop_type:laborers }
		add = location.local_commoner_sol
	}
    else_if = {
        limit = { pop_type = pop_type:soldiers }
        add = location.local_commoner_sol
    }
	else_if = {
		limit = { pop_type = pop_type:peasants }
		add = location.local_commoner_sol
	}
	else_if = {
		limit = { pop_type = pop_type:tribesmen }
		add = location.local_tribesmen_sol
	}
}
