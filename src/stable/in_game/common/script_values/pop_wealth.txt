gdp_per_capita_scale = {
    value = 1
    # Each embraced institution increases all pop goods demand by 5%.
    if = {
        limit = { location = { exists = owner } }
        add = {
            value = location.owner.num_embraced_institutions
            multiply = 0.05
        }
    }
    if = {
        limit = { pop_type = pop_type:nobles }
        add = location.local_noble_demand_scale_offset
    }
    else_if = {
        limit = { pop_type = pop_type:clergy }
        add = location.local_clergy_demand_scale_offset
    }
    else_if = {
        limit = { pop_type = pop_type:burghers }
        add = location.local_burghers_demand_scale_offset
    }
    else_if = {
        limit = { pop_type = pop_type:laborers }
        add = location.local_commoner_demand_scale_offset
    }
    else_if = {
        limit = { pop_type = pop_type:soldiers }
        add = location.local_commoner_demand_scale_offset
    }
    else_if = {
        limit = { pop_type = pop_type:peasants }
        add = location.local_commoner_demand_scale_offset
    }
    else_if = {
        limit = { pop_type = pop_type:tribesmen }
        add = location.local_tribesmen_demand_scale_offset
    }
}

local_average_gdp_per_capita_scale = {
    value = 0
    add = {
        value = local_noble_demand_scale_offset
        multiply = num_pop_type:nobles
    }
    add = {
        value = local_clergy_demand_scale_offset
        multiply = num_pop_type:clergy
    }
    add = {
        value = local_burghers_demand_scale_offset
        multiply = num_pop_type:burghers
    }
    add = {
        value = local_commoner_demand_scale_offset
        multiply = local_commoner_population
    }
    add = {
        value = local_tribesmen_demand_scale_offset
        multiply = num_pop_type:tribesmen
    }
    divide = {
        add = num_pop_type:nobles
        add = num_pop_type:clergy
        add = num_pop_type:burghers
        add = local_commoner_population
        add = num_pop_type:tribesmen
        min = 0.001
    }
    add = 1
    if = {
        limit = { exists = owner }
        add = {
            value = owner.num_embraced_institutions
            multiply = 0.05
        }
    }
}
