SOL_wine_demand = {
	value = 1
	# == Substitute Group: Alcohol ==
	if = {
		limit = { location.market ?= { is_target_in_global_variable_list = { name = SOL_wine_scarce_markets target = this } } }
		multiply = { value = 0.4 desc = "SOL_SUB_ALCOHOL_SCARCE" }
	}
	if = {
		limit = { location.market ?= { is_target_in_global_variable_list = { name = SOL_liquor_scarce_markets target = this } } }
		add = { value = 0.375 desc = "SOL_SUB_ALCOHOL_SHIFT" }
	}
	if = {
		limit = { location.market ?= { is_target_in_global_variable_list = { name = SOL_beer_scarce_markets target = this } } }
		add = { value = 0.3 desc = "SOL_SUB_ALCOHOL_SHIFT" }
	}
	if = {
		limit = {
			religion = {
				OR = {
					this = religion:jain
					this = religion:sikhism
					group = religion_group:muslim
				}
			}
		}
		multiply = {
			value = 0
			desc = "POP_DEMAND_RELIGION_FORBIDS"
		}
	}

	#Some Cultures have preferences
	if = {
		limit = {
			#Andeans favor "Chicha" that is a beer
			culture = { has_culture_group = culture_group:andean_group }
		}
		multiply = {
			desc = "POP_DEMAND_CULTURAL_PREFERENCES"
			value = 0.5
		}
	}

	if = {
		limit = {
			#Japanese favor "Sake" that is a liquor
			culture = { has_culture_group = culture_group:japanese_group }
		}
		multiply = {
			desc = "POP_DEMAND_CULTURAL_PREFERENCES"
			value = 0.5
		}
	}

	if = {
		limit = {
			location.region = {
				OR = {
					this = region:iberia_region
					this = region:italy_region
					this = region:balkan_region
				}
			}
		}
		multiply = {
			value = 2
			desc = "POP_DEMAND_MEDITERRANEAN"
		}
	}

	if = {
		limit = { 
			location.sub_continent = sub_continent:south_asia
		}
		multiply = {
			value = 0.6
			desc = "POP_DEMAND_INDIA"
		}
	}

	# some poptypes wants more...

	if = {
		limit = { 
			pop_type = pop_type:clergy
			religion = {
				OR = {
					this = religion:shinto # rice wine
					group = {
						OR = {
							this = religion_group:christian
							this = religion_group:israelite_group
						}
					}
				}
			}
		}
		
		multiply = {
			value = 10
			desc = "POP_DEMAND_RITUAL_WINE"
		}
	}
	if = {
		limit = { #Baltics prefer mead (wine)
			culture = { has_culture_group = culture_group:baltic_group }
		}
		multiply = {
			value = 1.2
			desc = "POP_DEMAND_BALTIC"
		}
	}

	# scale with GDP per capita
	multiply = {
		value = sol_gdp_per_capita_scale
	}
	

}
