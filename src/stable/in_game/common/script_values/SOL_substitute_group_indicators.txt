### ============================================================
### SECTION: Substitute Group Scarcity Indicators (GUI helpers)
### ============================================================
# Each value returns 1 if any good in the group is currently scarce
# at the evaluated location market (market is in SOL_<good>_scarce_markets).
#
# Used by location_window.gui demography_item blockoverride to drive
# visibility of per-stratum substitute-goods icons in the Population tab.
#
# Scope when evaluated: location (via Location.MakeScope.ScriptValue)

sol_grp_precious_scarce = {
    value = 0
    if = {
        limit = {
            OR = {
                market ?= { is_target_in_global_variable_list = { name = SOL_goods_gold_scarce_markets target = this } }
                market ?= { is_target_in_global_variable_list = { name = SOL_silver_scarce_markets target = this } }
                market ?= { is_target_in_global_variable_list = { name = SOL_pearls_scarce_markets target = this } }
                market ?= { is_target_in_global_variable_list = { name = SOL_amber_scarce_markets target = this } }
                market ?= { is_target_in_global_variable_list = { name = SOL_gems_scarce_markets target = this } }
                market ?= { is_target_in_global_variable_list = { name = SOL_ivory_scarce_markets target = this } }
            }
        }
        value = 1
    }
}
