namespace = sol_ae

# Age 1 — Traditions: notification fires at game start via on_game_start
sol_ae.1 = {
	type = country_event
	title = sol_ae.1.title
	desc = sol_ae.1.desc

	trigger = {
		sol_ae_is_on = yes
	}

	option = {
		name = sol_ae.1.a
		show_as_tooltip = {
			add_country_modifier = {
				modifier = sol_ae_age_1_modifier
				days = -1
			}
		}
	}
}
